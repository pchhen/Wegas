55      @Test
56      public void PlayerJoinTest() {
57          NumberDescriptor testNumber;
58          testNumber = new NumberDescriptor("number");
59          testNumber.setDefaultInstance(new NumberInstance(0));
60          testNumber.setScope(new TeamScope());
61  
62          NumberDescriptor testNumber2;
63          testNumber2 = new NumberDescriptor("number2");
64          testNumber2.setDefaultInstance(new NumberInstance(0));
65          testNumber2.setScope(new GameScope());
66  
67          TriggerDescriptor trigger = new TriggerDescriptor();
68          trigger.setScope(new TeamScope());
69          trigger.setDefaultInstance(new TriggerInstance());
70          trigger.setTriggerEvent(new Script("1===1"));
71          trigger.setPostTriggerEvent(new Script("number.value = " + FINAL_VALUE));
72          trigger.setOneShot(Boolean.TRUE);
73  
74          TriggerDescriptor trigger2 = new TriggerDescriptor();
75          trigger2.setScope(new TeamScope());
76          trigger2.setDefaultInstance(new TriggerInstance());
77          trigger2.setTriggerEvent(new Script("true"));
78          trigger2.setPostTriggerEvent(new Script("number2.value += 1 "));
79          trigger2.setOneShot(Boolean.FALSE);
80  
81          descriptorFacade.create(gameModel.getId(), testNumber);
82          descriptorFacade.create(gameModel.getId(), testNumber2);
83          descriptorFacade.create(gameModel.getId(), trigger);
84          descriptorFacade.create(gameModel.getId(), trigger2);
85  
86          Team team3 = new Team("test-team3");
87          Team team4 = new Team("test-team4");
88  
89          team3.setGame(game);
90  
91          team4.setGame(game);
92  
93          team4.setToken(TEAM4_TOKEN);
94  
95          teamFacade.create(team3);
96  
97          teamFacade.create(team4);
98          Player testPlayer0 = new Player("TestPlayer0");
99          Player testPlayer1 = new Player("TestPlayer1");
100 
101         playerFacade.create(team.getId(), testPlayer0);
102         playerFacade.create(team4.getId(), testPlayer1);
103 
104 
105         NumberDescriptor number = (NumberDescriptor) descriptorFacade.find(testNumber.getId());
106         /* CONTEXT? */
107         assert FINAL_VALUE == number.getValue(testPlayer0);
108         assert FINAL_VALUE == number.getValue(testPlayer1);
109 
110         /* REFRESH CONTEXT */
111         Assert.assertEquals(FINAL_VALUE, ((NumberInstance) instanceFacade.find(testNumber.getId(), testPlayer0)).getValue(), 0.0);
112         Assert.assertEquals(FINAL_VALUE, ((NumberInstance) instanceFacade.find(testNumber.getId(), testPlayer1)).getValue(), 0.0);
113 
114         /*
115          * Player created before variable -> state machines don't execute
116          */
117         Assert.assertEquals(0.0, ((NumberInstance) instanceFacade.find(testNumber.getId(), player2)).getValue(), 0.0);
118         /*
119          * Game Scope trigger increase for each player added after trigger creation
120          */
121         Assert.assertEquals(2, ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
122         /*
123          * add a player in not empty team then Reset, trigger will execute
124          */
125         playerFacade.create(team4.getId(), new Player("TestPlayer5"));
126         gameModelFacade.reset(gameModel.getId());
127         Assert.assertEquals(FINAL_VALUE, ((NumberInstance) instanceFacade.find(testNumber.getId(), testPlayer0)).getValue(), 0.0);
128         Assert.assertEquals(FINAL_VALUE, ((NumberInstance) instanceFacade.find(testNumber.getId(), testPlayer1)).getValue(), 0.0);
129         Assert.assertEquals(FINAL_VALUE, ((NumberInstance) instanceFacade.find(testNumber.getId(), player2)).getValue(), 0.0);
130 
131         /*
132          * trigger2 will execute numberOfPlayers times after a reset -> increment testNumber2 by the same amount.
133          * For each player
134          */
135         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
136 
137         playerFacade.create(team4.getId(), new Player("TestPlayer6"));
138         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
139         gameModelFacade.reset(gameModel.getId());
140         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
141         /*
142          * Player added in empty team.
143          */
144         playerFacade.create(team3.getId(), new Player("TestPlayer7"));
145         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
146         gameModelFacade.reset(gameModel.getId());
147         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
148         playerFacade.create(team3.getId(), new Player("TestPlayer8"));
149         playerFacade.create(team3.getId(), new Player("TestPlayer9"));
150         playerFacade.create(team3.getId(), new Player("TestPlayer10"));
151         Assert.assertEquals(playerFacade.find(testPlayer0.getId()).getGame().getPlayers().size(), ((NumberInstance) instanceFacade.find(testNumber2.getId(), testPlayer0)).getValue(), 0.0);
152     }
