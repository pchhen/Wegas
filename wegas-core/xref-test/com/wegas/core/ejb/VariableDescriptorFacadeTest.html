293         Assert.assertEquals(SUBNAME1, ((ListDescriptor) findByGameModelId.get(2)).item(0).getName());

297         Assert.assertEquals(SUBNAME1, findByGameModelId.get(0).getName());
298         Assert.assertEquals(0, ((ListDescriptor) findByGameModelId.get(3)).size());
299 
300         vdf.remove(vd1.getId());
301         vdf.remove(vd2.getId());
302         vdf.remove(vd3.getId());
303         vdf.remove(sub1.getId());

320         list1.setScope(new TeamScope());
321         vdf.create(gameModel.getId(), list1);
322 
323         StringDescriptor vd1 = new StringDescriptor(VARIABLENAME1);
324         vd1.setDefaultInstance(new StringInstance(VALUE1));

326         vdf.createChild(list1.getId(), vd1);
327 
328         StringDescriptor vd2 = new StringDescriptor(VARIABLENAME2);

334 
335         vdf.move(vd1.getId(), list1.getId(), 1);                                // Move first item to second position
336         childrenDescriptors = vdf.findByGameModelId(gameModel.getId());         // Refresh
337         Assert.assertEquals(VARIABLENAME2, ((ListDescriptor) childrenDescriptors.get(0)).item(0).getName());
338 
339         // 2nd case: from root to descriptor
340         StringDescriptor vd3 = new StringDescriptor(VARIABLENAME3);

342         vd3.setScope(new TeamScope());
343         vdf.create(gameModel.getId(), vd3);
344 
345         vdf.move(vd3.getId(), list1.getId(), 0);                                // Move first item to index 0
346         childrenDescriptors = vdf.findByGameModelId(gameModel.getId());         // Refresh
347         Assert.assertEquals(VARIABLENAME3, ((ListDescriptor) childrenDescriptors.get(0)).item(0).getName());
348         Assert.assertEquals(1, childrenDescriptors.size());
349 
350         // 3rd case: from one descriptor to another
351         ListDescriptor list2 = new ListDescriptor(LISTNAME2);
352         list2.setDefaultInstance(new ListInstance());
353         list2.setScope(new TeamScope());
354         vdf.create(gameModel.getId(), list2);
355 
356         vdf.move(vd3.getId(), list2.getId(), 0);                                // Move first item to index 0
357         childrenDescriptors = vdf.findByGameModelId(gameModel.getId());         // Refresh
358         Assert.assertEquals(VARIABLENAME3, ((ListDescriptor) childrenDescriptors.get(1)).item(0).getName());
359         Assert.assertEquals(2, ((ListDescriptor) childrenDescriptors.get(0)).size());
360 
361         vdf.remove(vd1.getId());                                                // Clean up
362         vdf.remove(vd2.getId());
363         vdf.remove(vd3.getId());
364         vdf.remove(list1.getId());
365         vdf.remove(list2.getId());
366     }

370         final String VARIABLENAME1 = "test_variable";
371         final String LISTNAME1 = "list1";
372         final String LISTNAME2 = "list2";
373         final String LISTNAME3 = "list3";
374         final String LISTNAME4 = "list4";
375 
376         VariableDescriptorFacade vdf = lookupBy(VariableDescriptorFacade.class);
377 
378         ListDescriptor list1 = new ListDescriptor(LISTNAME1, new ListInstance());// Create a hierarchy of lists
379         vdf.create(gameModel.getId(), list1);
380 
381         ListDescriptor list2 = new ListDescriptor(LISTNAME2, new ListInstance());
382         vdf.createChild(list1.getId(), list2);
383 
384         ListDescriptor list3 = new ListDescriptor(LISTNAME3, new ListInstance());
385         vdf.createChild(list1.getId(), list3);
386 
387         ListDescriptor list4 = new ListDescriptor(LISTNAME4, new ListInstance());
388         vdf.createChild(list3.getId(), list4);
389 
390         NumberDescriptor nb = new NumberDescriptor(VARIABLENAME1, new NumberInstance(10));
391         vdf.createChild(list4.getId(), nb);
392 
393         DescriptorListI duplicate =
394                 (DescriptorListI) vdf.duplicate(list1.getId());                 // Duplicate a root variable
395         Assert.assertEquals(10.0, ((NumberDescriptor) ((DescriptorListI) ((DescriptorListI) duplicate.item(1)).item(0)).item(0)).getInstance(player).getValue());
396 
397         duplicate = (DescriptorListI) vdf.duplicate(list3.getId());             // Duplicate a sub child variable
398         Assert.assertEquals(10.0, ((NumberDescriptor) ((DescriptorListI) duplicate.item(0)).item(0)).getInstance(player).getValue());
399 
400         GameModel duplicateGm = gameModelFacade.duplicate(gameModel.getId());
401         DescriptorListI find = (DescriptorListI) vdf.find(duplicateGm, LISTNAME1);
402         Assert.assertEquals(10.0, ((NumberInstance) ((NumberDescriptor) ((DescriptorListI) ((DescriptorListI) find.item(1)).item(0)).item(0)).getScope().getVariableInstances().values().iterator().next()).getValue());
403     }
