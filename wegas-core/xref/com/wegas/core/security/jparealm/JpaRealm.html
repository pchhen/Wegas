39       */
40      public JpaRealm() {
41          setName("JpaRealm");                                                    //This name must match the name in the User class's getPrincipals() method
42      }

44      @Override
45      protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken authcToken) throws AuthenticationException {
46          UsernamePasswordToken token = (UsernamePasswordToken) authcToken;
47          try {
48              JpaAccount account = (JpaAccount) accountFacade().findByEmail(token.getUsername());
49              SimpleAuthenticationInfo info = new SimpleAuthenticationInfo(account.getId(), account.getPasswordHex(), getName());
50              info.setCredentialsSalt(new SimpleByteSource(account.getSalt()));
51              return info;
52          } catch (EJBException e) {
53              return null;
54          } catch (NamingException ex) {
55              logger.error("Unable to find AocountFacade EJB", ex);
56              return null;
57          }
58      }
59  
60      @Override
61      protected AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection principals) {
62          try {
63  //            if (principals.fromRealm(this.getName()).size() > 0) {
64  //                Long accountId = (Long) principals.fromRealm(getName()).iterator().next();
65  //                AbstractAccount account = accountFacade().find(accountId);
66  //            }
67  
68              AbstractAccount account = accountFacade().find((Long) principals.getPrimaryPrincipal());
69              SimpleAuthorizationInfo info = new SimpleAuthorizationInfo();
70              for (Role role : account.getRoles()) {
71                  info.addRole(role.getName());
72  
73                  for (Permission p : role.getPermissions()) {
74                      addPermissions(info, p);
75                  }
76              }
77  
78              for (Permission p : account.getPermissions()) {
79                  addPermissions(info, p);
80              }
81              return info;
82          } catch (EJBException e) {
83              return null;
84          } catch (NamingException ex) {
85              logger.error("Unable to find AocountFacade EJB", ex);
86              return null;
87          }
88      }
89  
90      /**
91       *
92       * @return @throws NamingException
93       */
94      public AccountFacade accountFacade() throws NamingException {
95          return Helper.lookupBy(AccountFacade.class);
96      }
97  
98      public static void addPermissions(SimpleAuthorizationInfo info, Permission p) {
99          info.addStringPermission(p.getValue());
100         if (p.getInducedPermission() != null && !p.getInducedPermission().isEmpty()) {
101             info.addStringPermission(p.getInducedPermission());
102         }
103     }
