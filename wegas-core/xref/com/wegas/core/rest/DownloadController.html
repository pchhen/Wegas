8   package com.wegas.core.rest;

10  import com.sun.jersey.multipart.FormDataParam;
11  import java.io.IOException;
12  import java.io.OutputStream;
13  import java.nio.charset.Charset;
14  import javax.ejb.Stateless;
15  import javax.ws.rs.Consumes;
16  import javax.ws.rs.POST;
17  import javax.ws.rs.Path;
18  import javax.ws.rs.PathParam;
19  import javax.ws.rs.WebApplicationException;
20  import javax.ws.rs.core.MediaType;
21  import javax.ws.rs.core.Response;
22  import javax.ws.rs.core.StreamingOutput;
23  
24  /**
25   *
26   * @author Cyril Junod <cyril.junod at gmail.com>
27   */
28  @Stateless
29  @Path("Download")
30  public class DownloadController {
31  
32      @POST
33      @Consumes(MediaType.MULTIPART_FORM_DATA)
34      @Path("{filename}")
35      public static Response forward(@FormDataParam("ctype") String contentType, @PathParam("filename") String filename, @FormDataParam("data") final String data) {
36          return Response.ok(new StreamingOutput() {
37              @Override
38              public void write(OutputStream out) throws IOException, WebApplicationException {
39                  out.write(data.getBytes(Charset.forName("ISO-8859-1")));
40              }
41          }, contentType).header("Content-Disposition", "attachment; filename=" + filename).build();
42      }
43  }
