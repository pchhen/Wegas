96      private void runForPlayer(Player player) {
97          List<StateMachineDescriptor> statemachines = variableDescriptorFacade.findByClass(player.getGameModel(), StateMachineDescriptor.class);
98          List<Transition> passed = new ArrayList<>();
99          stateMachineEventsCounter = new InternalStateMachineEventCounter();
100         Integer steps = this.doSteps(player, passed, statemachines, 0);
101         logger.info("#steps[" + steps + "] - Player {} triggered transition(s):{}", player.getName(), passed);
102     }

104     private Integer doSteps(Player player, List<Transition> passedTransitions, List<StateMachineDescriptor> stateMachineDescriptors, Integer steps) {
105 
106         List<Script> impacts = new ArrayList<>();
107         List<Script> preImpacts = new ArrayList<>();
108         List<Transition> transitions;
109         StateMachineInstance smi;
110         Boolean validTransition;
111         Boolean transitionPassed = false;
112 
113         for (VariableDescriptor sm : stateMachineDescriptors) {
114             validTransition = false;
115             smi = (StateMachineInstance) sm.getInstance(player);
116             if (!smi.getEnabled() || smi.getCurrentState() == null) { // a state may not be defined : remove statemachine's state when a player is inside that state
117                 continue;
118             }
119             transitions = smi.getCurrentState().getTransitions();
120             for (Transition transition : transitions) {
121                 if (transition instanceof DialogueTransition) {                 // Dialogue, don't eval
122                     continue;
123                 } else if (this.isNotDefined(transition.getTriggerCondition())) {
124                     validTransition = true;
125                 } else if (transition.getTriggerCondition().getContent().contains("Event.fired")) { //TODO: better way to find out which are event transition.
126                     if (passedTransitions.contains(transition)) {
127                         /*
128                          * Loop prevention : that player already passed through
129                          * this transiton
130                          */
131                         logger.debug("Loop detected, already marked {} IN {}", transition, passedTransitions);
132                     } else {
133                         try {
134                             if (this.eventTransition(player, transition, smi)) {
135                                 passedTransitions.add(transition);
136                                 transitionPassed = true;
137                                 smi.transitionHistoryAdd(transition.getId());
138                             }
139                         } catch (ScriptException ex) {
140                             //maybe throw a WegasException
141                             //validTransition still false
142                         }
143 
144                     }
145 
146                 } else {
147                     try {
148                         validTransition = (Boolean) scriptManager.eval(player, transition.getTriggerCondition());
149                     } catch (ScriptException ex) {
150                         //validTransition still false
151                     }
152                 }
153                 if (validTransition == null) {
154                     throw new WegasException("Please review condition [" + sm.getLabel() + "]:\n"
155                             + transition.getTriggerCondition().getContent());
156                 } else if (validTransition) {
157                     if (passedTransitions.contains(transition)) {
158                         /*
159                          * Loop prevention : that player already passed through
160                          * this transiton
161                          */
162                         logger.debug("Loop detected, already marked {} IN {}", transition, passedTransitions);
163                     } else {
164                         passedTransitions.add(transition);
165                         smi.setCurrentStateId(transition.getNextStateId());
166                         preImpacts.add(transition.getPreStateImpact());
167                         impacts.add(smi.getCurrentState().getOnEnterEvent());
168                         smi.transitionHistoryAdd(transition.getId());
169                         transitionPassed = true;
170                     }
171                 }
172             }
173         }
174         if (transitionPassed) {
175             /*@DIRTY, @TODO : find something else : Runing scripts overrides previous state change Only for first Player (resetEvent). */
176             variableDescriptorFacade.findByClass(player.getGameModel(), StateMachineDescriptor.class);
177             preImpacts.addAll(impacts);
178             try {
179                 scriptManager.eval(player, preImpacts);
180             } catch (ScriptException | WegasException ex) {
181                 logger.warn("Script failed ", ex);
182             }
183             steps++;
184             steps = this.doSteps(player, passedTransitions, stateMachineDescriptors, steps);
185         }
186         return steps;
187 
188     }

190     /**
191      * Manage event transition
192      *
193      * @param player current {@link Player}
194      * @param transition the event transition
195      * @param smi current {@link StateMachineInstance} passing transition
196      * @return
197      * @throws ScriptException
198      */
199     private Boolean eventTransition(Player player, Transition transition, StateMachineInstance smi) throws ScriptException {
200         String script = transition.getTriggerCondition().getContent();
201         Pattern pattern = Pattern.compile("Event\\.fired\\((['\"])(\\w+)\\1\\)");
202         Matcher matcher = pattern.matcher(script);
203         String event;
204         if (matcher.find()) {
205             event = matcher.group(2);
206         } else {
207             return false;
208         }
209         Object[] firedParams = scriptEvent.getFiredParameters(event);
210         /* events and more ? */
211         if (!(Boolean) scriptManager.eval(player, transition.getTriggerCondition())) {
212             return false;
213         }
214         final Integer instanceEventCount = stateMachineEventsCounter.count(smi, event);
215         if (firedParams.length > instanceEventCount) {
216             smi.setCurrentStateId(transition.getNextStateId());
217             stateMachineEventsCounter.increase(smi, event);
218             if (!this.isNotDefined(transition.getPreStateImpact())) {
219                 this.evalEventImpact(player, transition.getPreStateImpact(), firedParams[instanceEventCount]);
220             }
221             if (!this.isNotDefined(smi.getCurrentState().getOnEnterEvent())) {
222                 this.evalEventImpact(player, smi.getCurrentState().getOnEnterEvent(), firedParams[instanceEventCount]);
223             }
224 
225             return true;
226         } else {
227             return false;
228         }
229     }

231     /**
232      * Run given script with given parameter. Script may use named parameter
233      * {@value #EVENT_PARAMETER_NAME} to access passed parameter.
234      *
235      * @param script the script to run
236      * @param param the parameter to pass to the script
237      * @see #EVENT_PARAMETER_NAME
238      */
239     private void evalEventImpact(Player player, final Script script, final Object param) throws ScriptException {
240         try {
241             final Object impactFunc;
242             if (script.getLanguage().toLowerCase().equals("javascript")) {
243                 impactFunc = scriptManager.eval(new Script(script.getLanguage(),
244                         String.format("function(%s){%s}", EVENT_PARAMETER_NAME, script.getContent()) // A JavaScript Function. should check for engine type
245                 ));
246             } else {
247                 return; // define other language here
248             }
249             if (param instanceof ScriptEventFacade.EmptyObject) {
250                 ((Invocable) requestManager.getCurrentEngine()).invokeMethod(impactFunc, "call", impactFunc);
251             } else {
252                 ((Invocable) requestManager.getCurrentEngine()).invokeMethod(impactFunc, "call", impactFunc, param);
253             }
254         } catch (NoSuchMethodException ex) {
255             logger.debug("Event transition script failed", ex);
256         }
257     }

270     /**
271      * Used to store Events during run. Prevent passing multiple event
272      * transitions with same event if less events where thrown.
273      * StateMachineInstance dependant.
274      */
275     private class InternalStateMachineEventCounter {
276 
277         private final Map<StateMachineInstance, Map<String, Integer>> smEvents;
278 
279         private InternalStateMachineEventCounter() {
280             this.smEvents = new HashMap<>();
281         }
282 
283         private Integer count(StateMachineInstance instance, String event) {
284             if (!smEvents.containsKey(instance)) {
285                 smEvents.put(instance, new HashMap<String, Integer>());
286             }
287             if (smEvents.get(instance).containsKey(event)) {
288                 return smEvents.get(instance).get(event);
289             } else {
290                 return 0;
291             }
292         }
293 
294         private void increase(StateMachineInstance instance, String event) {
295             if (!smEvents.containsKey(instance)) {
296                 smEvents.put(instance, new HashMap<String, Integer>());
297             }
298             smEvents.get(instance).put(event, this.count(instance, event) + 1);
299         }

301     }
