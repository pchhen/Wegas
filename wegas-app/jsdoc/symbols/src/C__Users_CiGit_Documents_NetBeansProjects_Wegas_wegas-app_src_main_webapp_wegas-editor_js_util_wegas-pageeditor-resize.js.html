  9 /**
 10  * @fileOverview PageEditor resize Extension
 11  * @author Cyril Junod <cyril.junod at gmail.com>
 12  */
 13 YUI.add("wegas-pageeditor-resize", function(Y) {
 14     "use strict";
 15     var CONTENTBOX = "contentBox",
 16             BOUNDINGBOX = "boundingBox",
 17             WIDTH = "width",
 18             HEIGHT = "height";
 19     /**
 20      * PageEditor Plugin Extension enabling widget resize.
 21      * @constructor
 22      */
 23     function PageEditorResize() {
 24         Y.onceAfter(this._initResize, this, "render");
 25         Y.after(this._alignResize, this, "showOverlay");
 26         Y.once(this._resizeDestruct, this, "destructor");
 27         Y.after(this._setupIcon, this, "fixedOverlay");
 28     }
 29     PageEditorResize.NAME = "wegas-pageeditor-resize";
 30     PageEditorResize.prototype = {
 31         /* @lends Y.Wegas.PageEditorResize# */
 32         /**
 33          * @private
 34          * @returns {undefined}
 35          */
 36         _initResize: function() {
 37             var bindedFixedOverlay;
 38             if (!this.highlightOverlay) {
 39                 Y.error("PageEditorResize is an extension for PageEditor.");
 40             }
 41             bindedFixedOverlay = Y.bind(this.fixedOverlay, this);
 42             this._resizeNode = Y.Node.create("<div class='pageeditor-resizenode'></div>");
 43             this._iconResizeNode = Y.Node.create("<div class='pageeditor-resizenode-icon'></div>");
 44             this._resizeNode.setStyles({
 45                 top: 0,
 46                 left: 0
 47             });
 48             this.highlightOverlay.get(CONTENTBOX).append(this._resizeNode);
 49             this._resizeNode.hide();
 50             this.shownOverlay.get(CONTENTBOX).append(this._iconResizeNode);
 51             this.highlightOverlay.after("visibleChange", function(e) {
 52                 if (e.newVal && this.overlayWidget && this.overlayWidget.CSSSize) {
 53                     this._resizeNode.show();
 54                 } else {
 55                     this._resizeNode.hide();
 56                 }
 57             }, this);
 58             this._resize = new Y.DD.Drag({
 59                 node: this._resizeNode
 60             });
 61             this._resize.plug(Y.Plugin.DDConstrained, {
 62                 constrain: this.get("host").get(CONTENTBOX),
 63                 cacheRegion: false                                              //scroll changes region
 64             }).plug(Y.Plugin.DDNodeScroll, {
 65                 node: this.get("host").get(CONTENTBOX)
 66             });
 67             this._resizeNode.before("mousedown", function(e) {
 68                 this.detach();
 69                 this._resizeNode.show();
 70                 this._dd.con.set("constrain", this.get("host").get("widget").get(CONTENTBOX));
 71                 this._resize._widget = this.shownOverlay._widget.get(BOUNDINGBOX);
 72             }, this);
 73             this._resizeNode.after("mouseup", function(e) {
 74                 this.bind();
 75             }, this);
 76             this._resize.on("drag:drag", function(e) {
 77                 var bb = this._resize._widget;
 78                 bb.setStyles({
 79                     width: parseInt(bb.getComputedStyle(WIDTH), 10) + e.info.delta[0],
 80                     height: parseInt(bb.getComputedStyle(HEIGHT), 10) + e.info.delta[1]
 81                 });
 82                 this.fixedOverlay(this.shownOverlay._widget);
 83             }, this);
 84 //            this._resize.before("drag:start", function(e) {
 85 //                var bb = this._widget;
 86 //                bb.setStyles({
 87 //                    width: bb.getComputedStyle(WIDTH),
 88 //                    height: bb.getComputedStyle(HEIGHT)
 89 //                });
 90 //                bindedFixedOverlay(Y.Widget.getByNode(bb));
 91 //            });
 92             this._resize.on("drag:end", function(e) {
 93                 var widget = this.shownOverlay._widget, bb = widget.get(BOUNDINGBOX);
 94                 widget.CSSSize.setAttrs({
 95                     "styles": {
 96                         width: bb.getComputedStyle(WIDTH),
 97                         height: bb.getComputedStyle(HEIGHT)
 98                     }
 99                 });
100                 this._resize._widget = null;
101                 this.fixedOverlay(widget);
102                 this.saveCurrentPage();
103                 this.bind();
104                 this._syncWidgetEdition();
105             }, this);
106         },
107         /**
108          * Align drag node with PageEditor's overlay
109          * @private
110          * @returns {undefined}
111          */
112         _alignResize: function() {
113             var bb = this.shownOverlay.get(BOUNDINGBOX), pos = bb.getXY();
114             if (pos) {  //widget not destroyed
115                 pos[0] = pos[0] + bb.getDOMNode().offsetWidth - 12;
116                 pos[1] = pos[1] + bb.getDOMNode().offsetHeight - 12;
117                 this._resizeNode.setXY(pos);
118             }
119         },
120         _setupIcon: function() {
121             if (this.shownOverlay._widget.CSSSize) {
122                 this._iconResizeNode.show();
123             }else{
124                 this._iconResizeNode.hide();
125             }
126         },
127         /**
128          * self destructor called after PageEditor's destructor
129          * @private
130          * @returns {undefined}
131          */
132         _resizeDestruct: function() {
133             Y.detach(this._alignResize, this, "showOverlay");
134              Y.detach(this._setupIcon, this, "fixedOverlay");
135             this._resize.detachAll();
136             this._resizeNode.detachAll();
137             this._resize.destroy();
138             this._resizeNode.destroy(true);
139             this._iconResizeNode.destroy(true);
140         }
141 
142     };
143     Y.namespace("Wegas").PageEditorResize = PageEditorResize;
144 });
