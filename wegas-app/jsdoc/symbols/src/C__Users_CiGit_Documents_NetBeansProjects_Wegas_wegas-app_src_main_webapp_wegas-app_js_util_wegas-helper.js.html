 12 YUI.add('wegas-helper', function(Y) {
 13     "use strict";
 14     /**
 15      * @name Y.Wegas.Helper
 16      * @class
 17      * @constructor
 18      */
 19 
 20     var Helper = {
 21         /**
 22          * Generate ID an unique id based on current time.
 23          * @function
 24          * @static
 25          * @return {Number} time
 26          * @description
 27          */
 28         genId: function() {
 29             var now = new Date();
 30             return now.getHours() + now.getMinutes() + now.getSeconds();
 31         },
 32         /**
 33          * Escape a html string by replacing <, > and " by their html entities.
 34          *
 35          * @function
 36          * @static
 37          * @param {String} str
 38          * @return {String} Escaped string
 39          */
 40         htmlEntities: function(str) {
 41             return String(str).replace(/&/g, '&')
 42                     .replace(/</g, '<')
 43                     .replace(/>/g, '>')
 44                     .replace(/"/g, '"');
 45         },
 46         /**
 47          * Replace any text line return
 48          * @function
 49          * @static
 50          * @param {String} str the string to escape
 51          * @param {String} replaceBy The value to replace with, default is \<br \/\>
 52          * @return {String} Escaped string
 53          */
 54         nl2br: function(str, replaceBy) {
 55             replaceBy = replaceBy || '<br />';
 56             return (str + '').replace(/([^>\r\n]?)(\r\n|\n\r|\r|\n)/g, '$1' + replaceBy + '$2');
 57         },
 58         escapeJSString: function(str) {
 59             return str.replace(/"/g, '\\"').replace(/(\r\n|\n\r|\r|\n)/g, "\\n");
 60             //return Helper.nl2br(str.replace(/"/g, '\\"'), "\\n");
 61         },
 62         unesacapeJSString: function(str) {
 63             return str.replace(/\\"/g, '"');
 64         },
 65         escapeCSSClass: function(str) {
 66             return str.replace(/ /g, "-").toLowerCase();
 67         },
 68         stripHtml: function(html) {
 69             var div = document.createElement("div");
 70             div.innerHTML = html;
 71             return div.textContent || div.innerText || "";
 72         },
 73         /**
 74          * Format a date, using provided format string.
 75          *
 76          * @function
 77          * @static
 78          * @argument {Number} timestamp
 79          * @argument {String} fmt the format to apply, ex. '%d.%M.%Y at %H:%i:%s' <br />
 80          * d	Day of the month, 2 digits with leading zeros <br />
 81          * m	Numeric representation of a month, with leading zeros <br />
 82          * M	A short textual representation of a month, three letters <br />
 83          * Y	A full numeric representation of a year, 4 digits <br />
 84          * H	24-hour format of an hour with leading zeros <br />
 85          * i	Minutes with leading zeros <br />
 86          * s	Seconds, with leading zeros <br />
 87          * @returns {String} formated date
 88          */
 89         formatDate: function(timestamp, fmt) {
 90             var date = new Date(timestamp),
 91                     months = ["Jan", "Feb", "Mar", "Apr", "Mai", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"];
 92             function pad(value) {
 93                 return (value.toString().length < 2) ? '0' + value : value;
 94             }
 95             return fmt.replace(/%([a-zA-Z])/g, function(_, fmtCode) {
 96                 switch (fmtCode) {
 97                     case 'Y':
 98                         return date.getFullYear();
 99                     case 'M':
100                         return months[date.getMonth()];
101                     case 'm':
102                         return pad(date.getMonth() + 1);
103                     case 'd':
104                         return pad(date.getDate());
105                     case 'H':
106                         return pad(date.getHours());
107                     case 'i':
108                         return pad(date.getMinutes());
109                     case 's':
110                         return pad(date.getSeconds());
111                     default:
112                         throw new Error('Unsupported format code: ' + fmtCode);
113                 }
114             });
115         },
116         /**
117          * Returns a time lapse between provided timestamp and now, e.g. "a month ago",
118          * "2 hours ago", "10 minutes ago"
119          * @function
120          * @static
121          * @argument {Number} timestamp
122          * @return {String} The formatted time
123          */
124         smartDate: function(timestamp) {
125             var date = new Date(timestamp),
126                     now = new Date(),
127                     diffN = now.getTime() - timestamp,
128                     oneMinute = 60 * 1000,
129                     oneHour = 60 * oneMinute,
130                     oneDay = 24 * oneHour;
131             // oneMonth =  30 * oneDay,
132             // oneYear =  365 * oneDay;
133 
134             if (!date.getTime()) {
135                 return "undefined";
136             }
137 
138             if (diffN < oneMinute) {                                           // last minute
139                 return Math.round(diffN / 1000) + " seconds ago";
140             } else if (diffN < oneHour) {                                      // last hour
141                 return  Math.round(diffN / oneMinute) + " minutes ago";
142             } else if (diffN < oneDay
143                     && now.getDay() === date.getDay()) {                            // Today
144                 return Helper.formatDate(timestamp, "%H:%i");
145             } else if (date.getYear() === now.getYear()) {                      // This year
146                 return Helper.formatDate(timestamp, "%d %M");
147             } else {                                                             // Older
148                 return Helper.formatDate(timestamp, "%d %M %Y");
149             }
150 
151         },
152         /**
153          * Java hashCode implementation
154          * @param {String} value to hash
155          * @returns {Number}
156          */
157         hashCode: function(value) {
158             return Y.Array.reduce(value.split(""), 0, function(prev, curr) {
159                 prev = ((prev << 5) - prev) + curr.charCodeAt(0);
160                 return prev |= 0;                                               //Force 32 bits
161             });
162         },
163         /**
164          * Return an object with functions (first level only, not objects in object...)
165          *  that will execute the supplied function in the supplied object's context,
166          *  optionally adding any additional supplied parameters to the beginning of
167          *  the arguments collection the supplied to the function.
168          * @param {Object} o the object with in functions to execute on the context object.
169          * @param {Object} c the execution context.
170          * @param {any} 0..n arguments to include before the arguments the function is executed with.
171          * @returns An object with the wrapped functions.
172          */
173         superbind: function(o, c) {
174             var i, args = arguments.length > 0 ? Y.Array(arguments, 0, true) : null;
175             for (i in o) {
176                 args[0] = o[i];
177                 o[i] = Y.bind.apply(c, args);
178             }
179             return o;
180         },
181         getURLParameter: function(name) {
182             var param = (RegExp(name + '=' + '(.+?)(&|$)').exec(location.search) || [, null])[1];
183             return param ? decodeURIComponent(param) : param;
184         }
185     };
186     Y.namespace("Wegas").Helper = Helper;
187     Y.namespace("Wegas").superbind = Y.Wegas.Helper.superbind;
188 
189     Y.namespace("Wegas").Timer = Y.Base.create("wegas-timer", Y.Base, [], {
190         initializer: function() {
191 
192         },
193         start: function() {
194             if (!this.handler) {
195                 this.handler = Y.later(this.get("duration"), this, this.timeOut);
196             }
197         },
198         reset: function() {
199             if (this.handler) {
200                 this.handler.cancel();
201             }
202             this.handler = Y.later(this.get("duration"), this, this.timeOut);
203         },
204         timeOut: function() {
205             this.handler = null;
206             this.fire("timeOut");
207         }
208     }, {
209         ATTRS: {
210             duration: {
211                 value: 0
212             }
213         }
214     });
215 });
