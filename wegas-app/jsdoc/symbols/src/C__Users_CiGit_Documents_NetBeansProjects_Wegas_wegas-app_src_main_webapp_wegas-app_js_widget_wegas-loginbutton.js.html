 12 YUI.add("wegas-loginbutton", function(Y) {
 13     "use strict";
 14 
 15     var LoginButton;
 16 
 17     /**
 18      * @name Y.Wegas.LoginButton
 19      * @extends Y.Wegas.Button
 20      * @class  Button with a defined behavior.
 21      * @constructor
 22      * @description Button with special label and menu with two
 23      * options : set user preferences or logout
 24      */
 25     LoginButton = Y.Base.create("wegas-login", Y.Wegas.Button, [], {
 26         /** @lends Y.Wegas.LoginButton# */
 27 
 28         // *** Lifecycle Methods *** //
 29         /**
 30          * @function
 31          * @private
 32          * @description bind function to events.
 33          * Call widget parent to execute its proper bind function.
 34          * When UserFacade is updated, do syncUI
 35          * Add plugin menu with 2 options : open page "user preferences" and logout
 36          */
 37         bindUI: function() {
 38             Y.Wegas.LoginButton.superclass.bindUI.apply(this, arguments);
 39             Y.Wegas.Facade.User.after("update", this.syncUI, this);
 40 
 41             if (this.menu) {                                                    // Don't add the plugin if it already exist.
 42                 return;
 43             }
 44 
 45             this.plug(Y.Plugin.WidgetMenu, {
 46                 children: [{
 47                         type: "Button",
 48                         label: "Preferences",
 49                         plugins: [{
 50                                 fn: "OpenPageAction",
 51                                 cfg: {
 52                                     subpageId: this.get("preferencePageId"), // @fixme
 53                                     targetPageLoaderId: this.get("targetPageLoader")
 54                                 }
 55                             }]
 56                     }, {
 57                         type: "Button",
 58                         label: "Logout",
 59                         plugins: [{
 60                                 fn: "OpenUrlAction",
 61                                 cfg: {
 62                                     url: "wegas-app/logout",
 63                                     target: "self"
 64                                 }
 65                             }]
 66                     }]
 67             });
 68         },
 69         /**
 70          * @function
 71          * @private
 72          * @description Call widget parent to execute its proper sync function.
 73          * Set label of this button with team and/or player name.
 74          */
 75         syncUI: function() {
 76             Y.Wegas.LoginButton.superclass.syncUI.apply(this, arguments);
 77 
 78             var cUser = Y.Wegas.Facade.User.cache.get("currentUser"),
 79                     cPlayer = Y.Wegas.Facade.Game.cache.getCurrentPlayer(),
 80                     cTeam = Y.Wegas.Facade.Game.cache.getCurrentTeam(),
 81                     name = cUser.get("name") || "undefined",
 82                     mainAccount = cUser.getMainAccount(),
 83                     gameModel = Y.Wegas.Facade.GameModel.cache.getCurrentGameModel();
 84 
 85             if (mainAccount) {
 86                 name = "<img src=\"http://www.gravatar.com/avatar/" + mainAccount.get("hash") + "?s=28&d=mm\" />" + name;
 87             }
 88 
 89             if (mainAccount instanceof Y.Wegas.persistence.GuestJpaAccount) {   // If current account is a Guest,
 90                 this.menu.getMenu().item(0).hide();                             // hide the "Preference" button
 91             }
 92 
 93             if (!this.get('labelIsUser')) {
 94                 if (cPlayer) {
 95                     name = cPlayer.get("name");
 96                 }
 97                 if (cTeam) {
 98                     name = cTeam.get("name") + " : " + name;
 99                 }
100                 if (gameModel && gameModel.get("properties.freeForAll")){
101                     name = cPlayer.get("name");
102                 }
103             }
104             this.set("label", name);
105         }
106     }, {
107         /** @lends Y.Wegas.LoginButton */
108 
109         /**
110          * @field
111          * @static
112          * @description
113          * <p><strong>Attributes</strong></p>
114          * <ul>
115          *    <li>labelIsUser: Select what kind of label you want (user/team  or team/player)</li>
116          *    <li>preferencePageId: Id of the the page which contains widget userPreferences</li>
117          *    <li>targetPageLoader: Zone to display the page which contains widget userPreferences</li>
118          * </ul>
119          */
120         ATTRS: {
121             label: {
122                 "transient": true
123             },
124             /**
125              * Select what kind of label you want (user/team  or team/player)
126              */
127             labelIsUser: {
128                 value: false,
129                 validator: function(b) {
130                     return (b === 'true' || b === true);
131                 }
132             },
133             /**
134              * Id of the the page which contains widget userPreference
135              */
136             preferencePageId: {
137                 value: 1000                                                     //@fixme
138             },
139             /**
140              * targetPageLoader: Zone to display the page which contains widget userPreferences
141              */
142             targetPageLoader: {
143                 value: "maindisplayarea"
144             }
145         }
146     });
147     Y.namespace('Wegas').LoginButton = LoginButton;
148 
149 });
