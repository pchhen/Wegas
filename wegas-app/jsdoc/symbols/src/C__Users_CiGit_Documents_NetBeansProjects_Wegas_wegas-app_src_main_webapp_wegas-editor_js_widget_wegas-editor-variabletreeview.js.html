 12 YUI.add('wegas-editor-variabletreeview', function(Y) {
 13     "use strict";
 14 
 15     var ID = "id",
 16             CLASS = "@class",
 17             NAME = "name",
 18             EDITBUTTONTPL = "<span class=\"wegas-treeview-editmenubutton\"></span>",
 19             Wegas = Y.Wegas,
 20             VariableTreeView;
 21 
 22     /**
 23      * @name Y.Wegas.VariableTreeView
 24      * @extends Y.Wegas.EditorTreeView
 25      * @constructor
 26      * @class
 27      */
 28     VariableTreeView = Y.Base.create("wegas-editor-treeview", Wegas.EditorTreeView, [], {
 29         /** @lends Y.Wegas.VariableTreeView# */
 30         // *** Private fields ** //
 31         CONTENT_TEMPLATE: "<div class=\"wegas-editor-variabletreeview\"></div>",
 32         // ** Lifecycle methods ** //
 33         renderUI: function() {
 34             VariableTreeView.superclass.renderUI.apply(this);
 35 
 36             this.treeView.plug(Y.Plugin.TreeViewSortable, {
 37                 nodeGroups: [{
 38                         nodeClass: "wegas-editor-questionitem",
 39                         parentNode: ".wegas-editor-question"
 40                     }, {
 41                         nodeClass: "wegas-editor-listitem",
 42                         parentNode: ".wegas-editor-list"
 43                     }]
 44             });                                                                 // Add sortable plugin to the treeview
 45             this.treeView.sortable.on("sort", function(e) {                     // On sort event,
 46                 var entity = e.dragWidget.get("data.entity"),
 47                         dropEntity = e.dropWidget.get("data.entity");
 48 
 49                 Wegas.Facade.VariableDescriptor.cache.move(entity, dropEntity, e.index);// call facade method
 50             });
 51         },
 52         syncUI: function() {
 53             VariableTreeView.superclass.syncUI.call(this);
 54             this.treeView.syncUI();                                             // Needed by treeview sort plugin
 55         },
 56         //
 57         // *** Private Methods *** //
 58         /**
 59          * @function
 60          * @private
 61          */
 62         genTreeViewElements: function(elements) {
 63             var ret = [], i, el, elClass, text, collapsed, selected,
 64                     l, result, children = [], tooltip;
 65 
 66             for (i in elements) {
 67                 if (elements.hasOwnProperty(i)) {
 68                     el = elements[i];
 69                     elClass = el.get(CLASS);
 70                     collapsed = !this.isNodeExpanded(el);
 71                     selected = (this.currentSelection === el.get(ID)) ? 2 : 0;
 72                     text = el.getEditorLabel()/* + "  <span class='treeview-sub'>" + el.getType().replace("Descriptor", "") + "</span>"*/;
 73                     tooltip = "Type: " + el.getType().replace("Descriptor", "");
 74 
 75                     switch (elClass) {
 76                         case 'StringDescriptor':
 77                         case 'TextDescriptor':
 78                         case 'NumberDescriptor':
 79                         case 'InboxDescriptor':
 80                         case 'TriggerDescriptor':
 81                         case 'FSMDescriptor':
 82                         case 'TaskDescriptor':
 83                         case 'ObjectDescriptor':
 84                         case 'ResourceDescriptor':
 85                         case 'DialogueDescriptor':
 86                             var els = this.genScopeTreeViewElements(el);
 87                             ret.push({
 88                                 type: 'TreeNode',
 89                                 label: text,
 90                                 tooltip:tooltip,
 91                                 children: els,
 92                                 //children: (els.length >= 1) ? els : null, //no children now, loaded on expands
 93                                 //children: null, //no children now, loaded on expands
 94                                 data: {
 95                                     entity: el
 96                                 },
 97                                 collapsed: collapsed,
 98                                 selected: selected,
 99                                 rightWidget: Y.Node.create(EDITBUTTONTPL),
100                                 iconCSS: "wegas-icon-variabledescriptor wegas-icon-" + elClass.toLowerCase(),
101                                 cssClass: "wegas-editor-listitem"
102                                         //iconCSS: "wegas-icon-" + el.get(CLASS)
103                             });
104                             break;
105 
106                         case 'ListDescriptor':
107                             ret.push({
108                                 type: 'TreeNode',
109                                 label: text,
110                                 tooltip:tooltip,
111                                 collapsed: collapsed,
112                                 selected: selected,
113                                 children: (!collapsed) ? this.genTreeViewElements(el.get("items")) : [],
114                                 data: {
115                                     entity: el
116                                 },
117                                 rightWidget: Y.Node.create(EDITBUTTONTPL),
118                                 cssClass: "wegas-editor-listitem wegas-editor-list"
119                             });
120                             break;
121 
122                         case 'QuestionDescriptor':
123                             ret.push({
124                                 type: 'TreeNode',
125                                 label: text,
126                                 tooltip:tooltip,
127                                 collapsed: collapsed,
128                                 selected: selected,
129                                 children: (!collapsed) ? this.genTreeViewElements(el.get("items")) : [],
130                                 data: {
131                                     entity: el
132                                 },
133                                 iconCSS: "wegas-icon-questiondescriptor",
134                                 rightWidget: Y.Node.create(EDITBUTTONTPL),
135                                 cssClass: "wegas-editor-listitem wegas-editor-question"
136                             });
137                             break;
138 
139                         case 'ChoiceDescriptor':
140                             children = [];
141 
142                             for (l = 0; l < el.get("results").length; l += 1) {
143                                 result = el.get("results")[l];
144                                 children.push({
145                                     label: "Result: " + result.get(NAME),
146                                     selected: (result.get(ID) === this.currentSelection) ? 2 : 0,
147                                     data: {
148                                         entity: result,
149                                         parentEntity: el
150                                     },
151                                     rightWidget: Y.Node.create(EDITBUTTONTPL),
152                                     iconCSS: "wegas-icon-variabledescriptor"
153                                 });
154                             }
155 
156                             ret.push({
157                                 type: 'TreeNode',
158                                 label: text,
159                                 tooltip:tooltip,
160                                 children: children,
161                                 data: {
162                                     entity: el
163                                 },
164                                 collapsed: collapsed,
165                                 selected: selected,
166                                 rightWidget: Y.Node.create(EDITBUTTONTPL),
167                                 iconCSS: "wegas-icon-choicedescriptor",
168                                 cssClass: "wegas-editor-questionitem"
169                             });
170                             break;
171 
172                         case 'SingleResultChoiceDescriptor':
173                             ret.push({
174                                 type: 'TreeLeaf',
175                                 label: text,
176                                 tooltip:tooltip,
177                                 selected: selected,
178                                 data: {
179                                     entity: el
180                                 },
181                                 rightWidget: Y.Node.create(EDITBUTTONTPL),
182                                 iconCSS: "wegas-icon-choicedescriptor",
183                                 cssClass: "wegas-editor-questionitem"
184                             });
185                             break;
186 
187                         default:
188                             text = el.get(CLASS) + ': ' + el.get(NAME);
189                             ret.push({
190                                 label: text,
191                                 tooltip:tooltip,
192                                 data: el
193                             });
194                             break;
195                     }
196                 }
197             }
198             return ret;
199         },
200         /**
201          * @function
202          * @private
203          */
204         genScopeTreeViewElements: function(el) {
205             var children = [], i, label, team, player, instance,
206                     instances = el.get("scope").get("variableInstances");
207 
208             for (i in instances) {
209                 if (instances.hasOwnProperty(i)) {
210                     instance = instances[i];
211                     label = '';
212                     switch (el.get("scope").get(CLASS)) {
213                         case 'PlayerScope':
214                             player = Wegas.Facade.Game.cache.getPlayerById(i);
215 
216                             if (!player)
217                                 continue;
218 
219                             label = (player) ? player.get(NAME) : "undefined";
220                             break;
221                         case 'TeamScope':
222                             team = Wegas.Facade.Game.cache.getTeamById(i);
223 
224                             if (!team)
225                                 continue;
226 
227                             label = (team) ? team.get(NAME) : "undefined";
228                             break;
229                         case 'GameScope':
230                         case 'GameModelScope':
231                             label = 'Global';
232                             break;
233                     }
234                     children.push(this.genVariableInstanceElements(label, instance));
235                 }
236             }
237             return children;
238         },
239         /**
240          * @function
241          * @private
242          */
243         genVariableInstanceElements: function(label, el) {
244             var l,
245                     selected = (this.currentSelection == el.get(ID)) ? 2 : 0;
246 
247             switch (el.get(CLASS)) {
248                 case 'StringInstance':
249                 case 'TextInstance':
250                 case 'NumberInstance':
251                 case 'ListInstance':
252                     return {
253                         label: label + ': ' + el.get("value"),
254                         selected: selected,
255                         data: {
256                             entity: el
257                         }
258                     };
259 
260                 case 'QuestionInstance':
261                     l = label + ((el.get("replies").length > 0) ? ': ' + el.get("replies").get(NAME) : ': unanswered');
262                     return {
263                         label: l,
264                         selected: selected,
265                         data: {
266                             entity: el
267                         }
268                     };
269 
270                 case 'InboxInstance':
271                     var k, children = [], collapsed = !this.isNodeExpanded(el);
272 
273                     label += "(" + el.get("messages").length + ")";
274 
275                     for (k = 0; k < el.get("messages").length; k += 1) {
276                         children.push({
277                             label: el.get("messages")[k].get("subject")
278                         });
279                     }
280                     return {
281                         type: 'TreeNode',
282                         label: label,
283                         selected: selected,
284                         collapsed: collapsed,
285                         data: {
286                             entity: el
287                         },
288                         children: children
289                     };
290 
291                 default:
292                     return {
293                         label: label,
294                         selected: selected,
295                         data: {
296                             entity: el
297                         }
298                     };
299             }
300         }
301     });
302     Y.namespace('Wegas').VariableTreeView = VariableTreeView;

330         fillsLeaf: function(e) {
331             var node = e.node,
332                     entity = node.get("data.entity"),
333                     id = entity.get(ID);
334 
335             if (entity instanceof Wegas.persistence.ListDescriptor) {
336                 if (node.size() > 0) {
337                     return;
338                 }
339                 node.add(this.get("host").genTreeViewElements(entity.get("items")));
340             } else if (entity instanceof Wegas.persistence.VariableDescriptor
341                     && !(Wegas.persistence.ChoiceDescriptor && entity instanceof Wegas.persistence.ChoiceDescriptor)) { // @hack
342 
343                 if (node.size() > 1) {  /* @fixme @hack What if there is only 1 player in the game ? */
344                     return;
345                 }
346                 node.removeAll();
347                 node.set("loading", true);
348 
349                 Wegas.Facade.VariableDescriptor.sendRequest({
350                     request: "/" + id + "?view=Editor"
351                 });
352             }
353         }

356         NAME: "EditorTVNodeLoader"
