 51      * @name Y.Plugin.EditEntityAction
 52      * @extends Y.Plugin.EntityAction
 53      * @constructor
 54      */
 55     var EditEntityAction = function() {
 56         EditEntityAction.superclass.constructor.apply(this, arguments);
 57     };
 58     Y.extend(EditEntityAction, EntityAction, {
 59         /**
 60          * @function
 61          * @private
 62          */
 63         execute: function() {
 64             var entity = this.get(ENTITY);
 65 
 66             EditEntityAction.hideRightTabs();                                   // Destroy all tabs in the right panel
 67             EditEntityAction.getEditionTab();                                   // Create the edition tab (and the left panel won't pop in and out)
 68 
 69             if (entity instanceof Y.Wegas.persistence.VariableDescriptor
 70                     || entity instanceof Y.Wegas.persistence.JpaAccount
 71                     || entity instanceof Y.Wegas.persistence.GameModel
 72                     || entity instanceof Y.Wegas.persistence.VariableInstance) {// @fixme we may get extended mode for every entities,
 73                 EditEntityAction.showEditFormOverlay();
 74                 this.get("dataSource").cache.getWithView(entity, "EditorExtended", {// just need to check if it causes bugs
 75                     on: {
 76                         success: function(e) {
 77                             EditEntityAction.hideEditFormOverlay();
 78                             EditEntityAction.showUpdateForm(e.response.entity, e.callback.ds);
 79                         },
 80                         ds: this.get("dataSource")
 81                     }
 82                 });
 83             } else {
 84                 EditEntityAction.showUpdateForm(this.get(ENTITY), this.get("dataSource"));
 85             }
 86         }
 87     }, {
 88         NS: "editentity",
 89         NAME: "EditEntityAction",
 90         ATTRS: {
 91             formCfg: {
 92             }
 93         },
 94         /**
 95          *
 96          */
 97         tab: null,
 98         /**
 99          *
100          */
101         form: null,
102         /**
103          * Show edition form in the target div
104          */
105         showEditForm: function(entity, callback, cancelCallback, formCfg) {
106             if (EditEntityAction.cancelCallback) {                              // A cancel action was defined. By changing form, assume cancel
107                 try {
108                     EditEntityAction.cancelCallback(EditEntityAction.currentEntity);
109                 } finally {
110                     EditEntityAction.cancelCallback = null;
111                 }
112             }
113 
114             EditEntityAction.callback = callback;
115             EditEntityAction.currentEntity = entity;
116             EditEntityAction.cancelCallback = cancelCallback;
117 
118             var tab = EditEntityAction.getEditionTab(),
119                     prefix = (entity.get("id")) ? "Edit " : "New ";             // No id -> new entity
120 
121             tab.setAttrs({
122                 label: prefix + entity.getType().replace("Descriptor", "").replace("Instance", "").toLowerCase(),
123                 selected: 2
124             });
125             tab.form.setAttrs({
126                 values: entity.toObject(),
127                 cfg: formCfg || entity.getFormCfg()
128             });
129             tab.form.toolbar.setStatusMessage("");
130             //tab.form.saveButton.set("disabled", false);
131 
132             Y.fire("rightTabShown");
133 
134             return tab.form;
135         },
136         getEditionTab: function() {
137             if (!EditEntityAction.tab || EditEntityAction.tab.get("destroyed")) {// First make sure the edit tab exists
138 //                Y.Widget.getByNode("#rightTabView").destroyAll();
139                 EditEntityAction.tab = Wegas.TabView.createTab("Edit", '#rightTabView', {}, 0);
140                 var form = new Wegas.Form();
141                 //EditEntityAction.tab = Wegas.TabView.createTab("Edit", '#centerTabView');
142                 form.on("submit", function(e) {
143                     this.form.showOverlay();
144                     //EditEntityAction.form.saveButton.set("disabled", true);
145                     this.callback(e.value, this.currentEntity);
146                 }, EditEntityAction);
147                 form.on("cancel", function() {
148                     this.tab.remove();
149                     this.tab.destroy();
150                     if (Y.Lang.isFunction(this.cancelCallback)) {
151                         this.cancelCallback(this.currentEntity);
152                     }
153                     this.cancelCallback = this.currentEntity = this.callback = null;
154                     //Wegas.app.widget.hidePosition("right");                   // Hide the right layout
155                 }, EditEntityAction);
156                 form.before("updated", function(e) {
157                     EditEntityAction.form.toolbar.setStatusMessage("*");
158                     //EditEntityAction.form.saveButton.set("disabled", false);
159                 });
160                 EditEntityAction.tab.form = EditEntityAction.form = form;
161                 EditEntityAction.tab.add(form);
162                 if (!Y.Wegas.App.hideRightPanelCloseButton) {
163                     form.addButton({
164                         type: "Button",
165                         action: "cancel",
166                         label: "<span class=\"wegas-icon wegas-icon-cancel\" ></span>Close"
167                     });
168                 }
169             }
170             return EditEntityAction.tab;
171         },
172         /**
173          *
174          */
175         hideFormFields: function() {
176             EditEntityAction.form.destroyForm();
177         },
178         /**
179          *
180          */
181         showEditFormOverlay: function() {
182             EditEntityAction.form.showOverlay();
183         },
184         /**
185          *
186          */
187         hideEditFormOverlay: function() {
188             EditEntityAction.form.hideOverlay();
189         },
190         hideRightTabs: function() {
191             Y.Widget.getByNode("#rightTabView").destroyAll();
192         },
193         /**
194          *
195          */
196         showFormMessage: function(level, msg, timeout) {
197             EditEntityAction.form.showMessageBis(level, msg);
198         },
199         /**
200          *
201          */
202         showUpdateForm: function(entity, dataSource) {
203             var dataSource = dataSource;
204 
205             EditEntityAction.showEditForm(entity, function(cfg) {           // Display the edit form
206                 // entity.setAttrs(cfg);
207                 dataSource.cache.put(cfg, {
208                     success: function() {
209                         EditEntityAction.showFormMessage("success", "Item has been updated");
210                         EditEntityAction.hideEditFormOverlay();
211                     },
212                     failure: Y.bind(EditEntityAction.form.defaultFailureHandler, EditEntityAction.form)
213                 });
214             });
215         }
216     });

227     Y.extend(NewEntityAction, EditEntityAction, {
228         showAddForm: function(entity) {
229             var dataSource = this.get("dataSource");
230             EditEntityAction.hideRightTabs();
231             EditEntityAction.showEditForm(entity, function(newVal) {
232                 dataSource.cache.post(newVal, null, {
233                     success: function(e) {
234                         EditEntityAction.hideEditFormOverlay();
235                         EditEntityAction.showUpdateForm(e.response.entity, dataSource);
236                         EditEntityAction.showFormMessage("success", "Item has been added");
237                     },
238                     failure: Y.bind(EditEntityAction.form.defaultFailureHandler, EditEntityAction.form)
239                 });
240             }, null, this.get("formCfg"));
241         },
242         execute: function() {
243             Wegas.Editable.useAndRevive({
244                 "@class": this.get("targetClass")                               // Load target class dependencies
245             }, Y.bind(function(entity) {
246                 this.showAddForm(entity);                                       // and display the edition form
247             }, this));
248         }
249     }, {
250         NS: "NewEntityAction",
251         NAME: "NewEntityAction",
252         ATTRS: {
253             targetClass: {},
254             dataSource: {
255                 getter: function(val) {
256                     if (!val) {
257                         return Wegas.Facade.VariableDescriptor;
258                     }
259                     if (Lang.isString(val)) {
260                         return Wegas.Facade[val];
261                     }
262                     return val;
263                 }
264             }
265         }

274     var EditEntityArrayFieldAction = function() {
275         EditEntityArrayFieldAction.superclass.constructor.apply(this, arguments);
276     };
277     Y.extend(EditEntityArrayFieldAction, EntityAction, {
278         execute: function() {
279             var entity = (this.get("method").toLowerCase() === "post") ? this.get(ENTITY) : this.get("parentEntity");
280 
281             if (entity instanceof Y.Wegas.persistence.VariableDescriptor) {
282                 this.get("dataSource").cache.getWithView(entity, "EditorExtended", {// just need to check if it causes bugs
283                     on: {
284                         success: Y.bind(function(e) {
285                             this.doExecute(e.response.entity);
286                         }, this)
287                     }
288                 });
289             } else {
290                 this.doExecute(entity);
291             }
292         },
293         doExecute: function(descriptor) {
294             var entity = this.get(ENTITY),
295                     dataSource = this.get("dataSource"),
296                     newEntity, targetArray;
297 
298             switch (this.get("method").toString().toLowerCase()) {
299                 case "put":
300                     var child = Y.Array.find(descriptor.get(this.get("attributeKey")), function(i) {
301                         return i.get("id") === entity.get("id");
302                     });
303 
304                     EditEntityAction.showEditForm(child, function(newVal) {
305                         child.setAttrs(newVal);
306                         dataSource.cache.put(descriptor.toObject(), {
307                             success: function() {
308                                 EditEntityAction.hideEditFormOverlay();
309                                 EditEntityAction.showFormMessage("success", "Item has been updated");
310                             },
311                             failure: Y.bind(EditEntityAction.form.defaultFailureHandler, EditEntityAction.form)
312                         });
313                     });
314                     break;
315                 case "post":
316                     newEntity = Wegas.Editable.revive({
317                         "@class": this.get("targetClass")
318                     });
319                     EditEntityAction.showEditForm(newEntity, Y.bind(function(newVal) {
320                         newEntity.setAttrs(newVal);
321                         descriptor.get(this.get("attributeKey")).push(newEntity);
322                         dataSource.cache.put(descriptor.toObject(), {
323                             success: function() {
324                                 EditEntityAction.hideEditFormOverlay();
325                                 EditEntityAction.showFormMessage("success", "Item has been added");
326                                 EditEntityAction.hideFormFields();
327                             },
328                             failure: Y.bind(EditEntityAction.form.defaultFailureHandler, EditEntityAction.form)
329                         });
330                     }, this));
331                     break;
332                 case "delete":
333                     if (confirm("Are your sure your want to delete this item ?")) {
334                         targetArray = descriptor.get(this.get("attributeKey"));
335                         Y.Array.find(targetArray, function(e, i, a) {
336                             if (e.get("id") === entity.get("id")) {
337                                 a.splice(i, 1);
338                                 return true;
339                             }
340                             return false;
341                         });
342                         this.get("host").showOverlay();
343                         dataSource.cache.put(descriptor.toObject());
344                     } else {
345                         return;
346                     }
347                     break;
348             }
349         }
350     }, {
351         NS: "editentitarrayfieldaction",
352         NAME: "EditEntityArrayFieldAction",
353         ATTRS: {
354             /**
355              * Can be put, post or delete
356              */
357             method: {
358                 value: "put"
359             },
360             parentEntity: {},
361             targetClass: {},
362             attributeKey: {}
363         }
364     });

372     var AddEntityChildAction = function() {
373         AddEntityChildAction.superclass.constructor.apply(this, arguments);
374     };
375     Y.extend(AddEntityChildAction, NewEntityAction, {
376         showAddForm: function(entity, parentData) {
377             var dataSource = this.get("dataSource");
378             EditEntityAction.hideRightTabs();                                   // Hide all active tabs
379 
380             EditEntityAction.showEditForm(entity, function(newVal) {
381                 //@Hack since the server return the parent list,
382                 // and we have no way to identify the newly created descriptor
383                 // we need to look for the one that was not there before
384                 var idBack = [];
385                 Y.Array.each(parentData.get("items"), function(e) {
386                     idBack.push(e.get("id"));
387                 });
388                 dataSource.cache.post(newVal, parentData.toObject(), {
389                     success: function(e) {
390                         EditEntityAction.hideEditFormOverlay();
391 
392                         var newDescriptor = e.response.entity;
393                         if (newDescriptor instanceof Y.Wegas.persistence.VariableDescriptor
394                                 && newDescriptor.get("items")) {                // If the parent list of the edited item was returned,
395                             newDescriptor = Y.Array.find(newDescriptor.get("items"), function(e) {// need to look up for the edited entity
396                                 return idBack.indexOf(e.get("id")) === -1;
397                             });
398                         }
399 
400                         EditEntityAction.showUpdateForm(newDescriptor, dataSource);
401                         EditEntityAction.showFormMessage("success", "Item has been added");
402                     },
403                     failure: Y.bind(EditEntityAction.form.defaultFailureHandler, EditEntityAction.form)
404                 });
405             }, null, this.get("formCfg"));
406         },
407         execute: function() {
408             Wegas.Editable.useAndRevive({// Load target class dependencies
409                 "@class": this.get("targetClass")
410             }, Y.bind(function(entity) {
411                 this.showAddForm(entity, this.get(ENTITY)); // and display the edition form
412             }, this));
413         }
414     }, {
415         NS: "wegas",
416         NAME: "AddEntityChildAction",
417         ATTRS: {
418             targetClass: {}
419         }
420     });

567                 plugins: [{
568                         fn: "WidgetToolbar"
569                     }]
570             }), Y.bind(function(entity, widget, tab) {
571                 tab.set("selected", 2);
572                 widget.showOverlay();
573                 EditEntityAction.showEditFormOverlay();
574                 this.get("dataSource").cache.getWithView(entity, "EditorExtended", {// just need to check if it causes bugs
575                     on: {
576                         success: function(e) {
577                             widget.set("entity", e.response.entity);
578                             EditEntityAction.hideEditFormOverlay();
579                         }
580                     }
581                 });
582             }, this, this.get("entity")));
583         }
584     }, {
585         NS: "wegas",
586         NAME: "EditFSMAction",
587         ATTRS: {

589                 value: {}
590             }
591         }
