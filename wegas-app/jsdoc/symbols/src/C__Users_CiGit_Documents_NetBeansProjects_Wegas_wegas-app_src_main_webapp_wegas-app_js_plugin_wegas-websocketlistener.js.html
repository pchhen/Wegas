 11 YUI.add('wegas-websocketlistener', function(Y) {
 12     "use strict";
 13 
 14     var WebSocketListener = Y.Base.create("WebSocketListener", Y.Plugin.Base, [], {
 15         initializer: function() {
 16             Y.Wegas.Facade[this.get("dataSource")].on("*:EntityUpdatedEvent", this.onVariableInstanceUpdate, this);
 17         },
 18        
 19         onVariableInstanceUpdate: function(data) {
 20             this.get("host").cache.onResponseRevived({
 21                 serverResponse: Y.Wegas.Editable.revive({
 22                     "@class": "ManagedModeResponseFilter$ServerResponse",
 23                     events: [data]
 24                 })
 25             });
 26         },
 27         destructor: function() {
 28         }
 29     }, {
 30         ATTRS: {
 31             dataSource: {
 32                 initOnly:true
 33             }  
 34         },
 35         NS: "ws",
 36         NAME: "WebSocketListener"
 37     });
 38     Y.namespace('Plugin').WebSocketListener = WebSocketListener;
 39 
 40 });
