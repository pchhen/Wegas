 12 YUI.add('wegas-pmg-assignment', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class color occupation in datatable
 17      *  @name Y.Plugin.Assignment
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var CONTENTBOX = "contentBox", HOST = "host",
 22             Wegas = Y.Wegas,
 23             Assignment = Y.Base.create("wegas-pmg-assignment", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 24         handlers: null,
 25         menu: null,
 26         menuDetails: null,
 27         sortable: null,
 28         /** @lends Y.Plugin.Assignment */
 29 
 30         /**
 31          * Lifecycle methods
 32          * @function
 33          * @private
 34          */
 35         initializer: function() {
 36             this.handlers = {};
 37             this.sortable = [];
 38 
 39             this.addAssignmentColumn();
 40 
 41             this.onceAfterHostEvent("render", function() {
 42                 this.sync();
 43                 this.bind();
 44             });
 45         },
 46         bind: function() {
 47             this.handlers.update = Y.Wegas.Facade.VariableDescriptor.after("update", this.sync, this);
 48 
 49             this.handlers.createMenu = this.get(HOST).datatable.delegate('click', function(e) {            // fill the "add" menu on click
 50                 this.createMenu(e, true);
 51             }, '.yui3-datatable-data .assignment .assign', this);
 52             this.handlers.showDelete = this.get(HOST).datatable.delegate('hover', function(e) {
 53                 e.currentTarget.all(".remove").show();
 54             }, function(e) {
 55                 e.currentTarget.all(".remove").hide();
 56             }, '.tasks .task', this);
 57 
 58             this.handlers.remove = this.get(HOST).datatable.delegate('click', function(e) {
 59                 var node = e.target.getDOMNode().parentElement;
 60                 Wegas.Facade.VariableDescriptor.sendRequest({
 61                     request: "/ResourceDescriptor/RemoveAssignment/" + node.getAttribute("assignmentid"),
 62                     cfg: {
 63                         method: "DELETE"
 64                     }
 65                 });
 66             }, '.task .remove', this);
 67 
 68             this.get("host").datatable.after("sort", this.sync, this);
 69         },
 70         /**
 71          * Destructor methods.
 72          * @function
 73          * @private
 74          */
 75         sync: function() {
 76             Y.log("sync()", "log", "Wegas.Assignment");
 77             this.addButtonsAssignment();
 78             this.syncSortable();
 79         },
 80         addAssignmentColumn: function() {
 81             this.get(HOST).datatable.addColumn({
 82                 key: 'assignments',
 83                 label: "Assignments"
 84             }, this.get("columnPosition"));
 85         },
 86         addButtonsAssignment: function() {
 87             var cb = this.get(HOST).get(CONTENTBOX);
 88             cb.all(".yui3-datatable-data tr .yui3-datatable-col-assignments").each(function(node) {
 89                 node.append("<div class='assignment'></div>");
 90                 node.one('.assignment').append("<span class='assign'></span>");
 91 //                node.addClass('noDescription');
 92             });
 93         },
 94         createMenu: function(e) {
 95             var i, tasks, resources, resourceDesc, resourceId;
 96             resourceId = this.get(HOST).datatable.getRecord(e.target).get("id");
 97             resources = Y.Wegas.Facade.VariableDescriptor.cache.find("name", this.get(HOST).get('variable'));
 98             for (i = 0; i < resources.get('items').length; i += 1) {
 99                 if (resources.get('items')[i].get('id') === resourceId) {
100                     resourceDesc = resources.get('items')[i];
101                     break;
102                 }
103             }
104             if (this.menu === null) {
105                 this.menu = new Y.Wegas.Menu();
106                 this.menuDetails = new Y.Wegas.Menu({
107                     width: "250px"
108                 });
109 
110                 this.handlers.moveMenu = this.menu.on("button:mouseenter", function(e) {           // align the menu
111                     var timer;
112                     this.menuDetails.hide();
113                     this.menuDetails.set("align", {
114                         node: this.menu.get("boundingBox"),
115                         points: (e.details[0].domEvent.clientX > Y.DOM.winWidth() / 2) ?
116                                 ["tr", "tl"] : ["tl", "tr"]
117                     });
118                     timer = new Y.Wegas.Timer({
119                         duration: "500"
120                     });
121                     timer.on("timeOut", function() {
122                         this.menuDetails.show();
123                         this.getTaskDescription(e.target.get("data").assignement.taskDescriptor);
124                     }, this);
125                     timer.start();
126                 }, this);
127 
128                 this.handlers.hideMenu = this.menu.on("visibleChange", function(e) {                 // When the menu is hidden, hide the details panel
129                     if (!e.newVal) {
130                         this.menuDetails.hide();
131                     }
132                 }, this);
133 
134                 this.handlers.assignTask = this.menu.on("button:click", this.onTaskMenuClick, this);
135             }
136             this.menu.removeAll();
137             tasks = this.getTasks(resourceDesc);
138             if (!tasks || tasks.lenght <= 0) {
139                 return;
140             }
141             this.menu.add(tasks);
142             this.menu.attachTo(e.target);
143         },
144         getTasks: function(resourceDesc) {
145             //add is a boolean to determine if target is remove or add a task
146             //you can only add a task which isn't already added.
147             //you can only remove a task which is added.
148             var i, tasks, items, taskDesc, label, array = [], no, taskExist,
149                     assignments = resourceDesc.getInstance().get("assignments"),
150                     taskExistence = function(item) {
151                 return taskDesc.get("id") === item.get("taskDescriptorId");
152             };
153             if (!this.get("taskList")) {
154                 return;
155             }
156             tasks = Y.Wegas.Facade.VariableDescriptor.cache.find("name", this.get("taskList"));
157             items = tasks.get('items');
158             for (i = 0; i < items.length; i += 1) {
159                 taskExist = false;
160                 taskDesc = items[i];
161                 taskExist = Y.Array.find(assignments, taskExistence);
162                 if (!taskExist && taskDesc.getInstance().get("active")) {
163                     no = taskDesc.get("index");
164                     label = (taskDesc.get("title") || taskDesc.get("label") || taskDesc.get("name") || "undefined");
165                     array.push({
166                         type: "Button",
167                         label: no + ". " + label,
168                         data: {
169                             assignement: {
170                                 "@class": "Assignment",
171                                 taskDescriptor: taskDesc
172                             },
173                             resourceDesc: resourceDesc
174                         }
175                     });
176                 }
177             }
178             return array;
179         },
180         onTaskMenuClick: function(e) {
181             var data = e.target.get("data");
182             Wegas.Facade.VariableDescriptor.sendRequest({
183                 request: "/ResourceDescriptor/AbstractAssign/" + data.resourceDesc.getInstance().get("id"),
184                 cfg: {
185                     method: "POST",
186                     data: Y.JSON.stringify(data.assignement)
187                 }
188             });
189         },
190         getTaskDescription: function(taskDescriptor) {
191             if (taskDescriptor.get("description")) {
192                 this.descriptionToDisplay(taskDescriptor, taskDescriptor.get("description"));
193                 return;
194             }
195             Y.Wegas.Facade.VariableDescriptor.cache.getWithView(taskDescriptor, "Extended", {// Retrieve the object from the server in Export view
196                 on: Y.Wegas.superbind({
197                     success: function(e) {
198                         taskDescriptor.set("description", e.response.entity.get("description"));
199                         this.descriptionToDisplay(taskDescriptor, e.response.entity.get("description"));
200                     },
201                     failure: function(e) {
202                         this.menuDetails.get(CONTENTBOX).setHTML('<div style="padding:5px 10px"><i>Error loading description</i></div>');
203                     }
204                 }, this)
205             });
206         },
207         descriptionToDisplay: function(descriptor, fieldValue) {
208             var dataToDisplay, i, requirements;
209             dataToDisplay = '<div class="field" style="padding:5px 10px"><p class="popupTitel">Description</p><p>' + fieldValue + '</p></div><div style="padding:5px 10px" class="requirements"><p class="popupTitel">Requirements</p>';
210             requirements = descriptor.getInstance().get("requirements");
211             for (i = 0; i < requirements.length; i += 1) {
212                 dataToDisplay = dataToDisplay + "<p>" + requirements[i].get("quantity") + "x " + requirements[i].get("work")
213                         + " " + Y.Wegas.PmgDatatable.TEXTUAL_SKILL_LEVEL[requirements[i].get("level")];
214             }
215             dataToDisplay = dataToDisplay + "</div>";
216             this.menuDetails.get(CONTENTBOX).setHTML(dataToDisplay);
217         },
218         syncSortable: function() {
219             var i, node, dt = this.get(HOST).datatable, assignments,
220                     resourceDesc, resourceInstance, iResource, iAssign, taskDesc, assignementCell;
221 
222             //the drag and drop feature to reorganise tasks.
223             for (i = 0; i < this.sortable.length; i += 1) {
224                 this.sortable[i].destroy();
225             }
226 
227             // get assignments
228             for (iResource = 0; iResource < dt.data._items.length; iResource += 1) {
229                 resourceDesc = Y.Wegas.Facade.VariableDescriptor.cache.find("id", dt.data._items[iResource].get("id"));
230                 resourceInstance = resourceDesc.getInstance();
231                 assignments = resourceInstance.get("assignments");
232                 node = "<div class='tasks'>";
233                 for (iAssign = 0; iAssign < assignments.length; iAssign += 1) {
234                     taskDesc = Y.Wegas.Facade.VariableDescriptor.cache.find("id", assignments[iAssign].get("taskDescriptorId"));
235                     node = node + "<em class='task' assignmentid=" + assignments[iAssign].get("id") + "><span class='remove' style='display:none'></span><span>" + taskDesc.get("index") + "</span></em>";
236                 }
237                 node = node + "</div>";
238                 assignementCell = dt.getCell([iResource, this.get("columnPosition")]);
239                 assignementCell.append(node);
240                 this.sortable.push(new Y.Sortable({
241                     container: assignementCell.one(".tasks"),
242                     nodes: '.task',
243                     opacity: '.1'
244                 }));
245             }
246 
247             for (i = 0; i < this.sortable.length; i += 1) {
248                 this.sortable[i].delegate.after('drag:end', this.setPosition, this);
249             }
250         },
251         setPosition: function(e) {
252             var node = e.currentTarget.get("currentNode"), i = node.get("parentNode").get("children").indexOf(node);
253             Wegas.Facade.VariableDescriptor.sendRequest({
254                 request: "/ResourceDescriptor/MoveAssignment/" + node.getAttribute("assignmentid") + "/" + i,
255                 cfg: {
256                     method: "POST"
257                 }
258             });
259         },
260         destructor: function() {
261             Y.log("destructor()", "log", "Wegas.Assignment");
262             var k, i;
263             for (k in this.handlers) {
264                 if (this.handlers.hasOwnProperty(k)) {
265                     this.handlers[k].detach();
266                 }
267             }
268             if (this.menu) {
269                 this.menu.destroy();
270             }
271             if (this.menuDetails) {
272                 this.menuDetails.destroy();
273             }
274 
275             for (i = 0; i < this.sortable.length; i += 1) {
276                 this.sortable[i].destroy();
277             }
278         }
279     }, {
280         ATTRS: {
281             taskList: {
282                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
283                 _inputex: {
284                     _type: "variableselect",
285                     label: "Task list"
286                 }
287             },
288             columnPosition: {
289                 value: 5,
290                 _inputex: {
291                     _type: "integer",
292                     label: "Column position"
293                 }
294             }
295         },
296         NS: "assignment",
297         NAME: "Assignment"
298     });
299     Y.namespace("Plugin").Assignment = Assignment;
300 });
