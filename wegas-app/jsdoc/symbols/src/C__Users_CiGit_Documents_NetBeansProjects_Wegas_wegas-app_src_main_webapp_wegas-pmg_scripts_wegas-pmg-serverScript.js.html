176 /**
177  * Set gauge value (and restrict them between their bounds (max and min values))
178  */
179 function updateGauges() {
180     var i, j, taskInst, tasks = VariableDescriptorFacade.findByName(gm, 'tasks'),
181             costs = VariableDescriptorFacade.findByName(gm, 'costs'),
182             delay = VariableDescriptorFacade.findByName(gm, 'delay'),
183             quality = VariableDescriptorFacade.findByName(gm, 'quality'),
184             planedValue = VariableDescriptorFacade.findByName(gm, 'planedValue'),
185             earnedValue = VariableDescriptorFacade.findByName(gm, 'earnedValue'),
186             actualCost = VariableDescriptorFacade.findByName(gm, 'actualCost'),
187             qualityImpacts = VariableDescriptorFacade.findByName(gm, 'qualityImpacts'),
188             tasksQuality = 0, nomberOfBeganTasks = 0, tasksScale = 0, nomberOfEmployeeRequired,
189             costsJaugeValue, qualityJaugeValue, delayJaugeValue, qualityJaugeValue = 0;
190 
191     //Calculate task quality and task scale                                     @to check change quality calcul
192     for (i = 0; i < tasks.items.size(); i++) {
193         taskInst = tasks.items.get(i).getInstance(self);
194         if (isTrue(taskInst.getActive())) {//if task is active
195             nomberOfBeganTasks += 1;
196             nomberOfEmployeeRequired = 0;
197             for (j = 0; j < taskInst.getRequirements().size(); j++) {
198                 nomberOfEmployeeRequired += parseInt(taskInst.getRequirements().get(j).getQuantity());
199             }
200             tasksScale += parseInt(taskInst.getDuration()) * nomberOfEmployeeRequired;
201             if (parseInt(taskInst.getProperty('completeness')) > 0) { //...and started
202                 //TO check
203                 tasksQuality += parseInt(taskInst.getProperty('quality')) * parseInt(taskInst.getDuration()) * nomberOfEmployeeRequired;
204             } else {
205                 tasksQuality += (100 + parseInt(taskInst.getProperty('quality'))) * parseInt(taskInst.getDuration()) * nomberOfEmployeeRequired;
206             }
207         }
208     }
209 
210     //costs = EV / AC * 100
211     if (parseInt(planedValue.getInstance(self).getValue()) > 0) {
212         costsJaugeValue = Math.round((parseInt(earnedValue.getInstance(self).getValue()) / parseInt(actualCost.getInstance(self).getValue())) * 100);
213     }
214     costsJaugeValue = (costsJaugeValue > parseInt(costs.getMinValue())) ? costsJaugeValue : parseInt(costs.getMinValue());
215     costsJaugeValue = (costsJaugeValue < parseInt(costs.getMaxValue())) ? costsJaugeValue : parseInt(costs.getMaxValue());
216     costs.getInstance(self).setValue(costsJaugeValue);
217 
218     //delay = EV / PV * 100
219     delayJaugeValue = Math.round((parseInt(earnedValue.getInstance(self).getValue()) / parseInt(planedValue.getInstance(self).getValue())) * 100);
220     delayJaugeValue = (delayJaugeValue > parseInt(delay.getMinValue())) ? delayJaugeValue : parseInt(delay.getMinValue());
221     delayJaugeValue = (delayJaugeValue < parseInt(delay.getMaxValue())) ? delayJaugeValue : parseInt(delay.getMaxValue());
222     delay.getInstance(self).setValue(delayJaugeValue);
223 
224     //quality
225     //with weighting of task's scale = sum each task -> task quality / task scale
226     if (tasksScale > 0) {
227         qualityJaugeValue = (tasksQuality / tasksScale);
228     }
229     //whitout weighting of task's scale
230     //    if (nomberOfBeganTasks > 0) {
231     //        qualityJaugeValue = tasksQuality / nomberOfBeganTasks;
232     //    }
233     qualityJaugeValue += parseInt(qualityImpacts.getInstance(self).getValue()) / 2;
234     qualityJaugeValue = (qualityJaugeValue > parseInt(quality.getMinValue())) ? qualityJaugeValue : parseInt(quality.getMinValue());
235     qualityJaugeValue = (qualityJaugeValue < parseInt(quality.getMaxValue())) ? qualityJaugeValue : parseInt(quality.getMaxValue());
236     quality.getInstance(self).setValue(qualityJaugeValue);
237 }

239 /**
240  * calculate how many percent is completed based on current period and planified
241  *  length of the task.
242  * @param {TaskInstance} taskInst
243  * @returns {Number} number between 0 and 100 (both including)
244  */
245 function getPlannifiedCompleteness(taskInst) {
246     var i, planif = taskInst.getPlannification(), planifArray = [],
247             time = getCurrentInGameTime(), pastPeriods = 0;
248     for (i = 0; i < planif.size(); i++) { //transform list to array to use function ''indexOf''
249         planifArray.push(parseInt(planif.get(i)));
250     }
251     for (i = 0; i <= time.period; i++) {
252         if (planifArray.indexOf(i) > -1) {
253             pastPeriods += 1;
254         }
255     }
256     return (planifArray.length > 0) ? (pastPeriods / planifArray.length) * 100 : 0;
257 }

259 /**
260  * return the delay based on the difference (in percent) between
261  *  plannifiedcompleteness and real completeness (completeness / planifiedCompleteness * 100)
262  * if given task isn't started then delay = completeness + 100
263  * planified completeness is based on function ''getPlannifiedCompleteness''
264  * @param {taskDescriptor} taskDesc
265  * @returns {Number} delay between 0 and 100
266  */
267 function getCurrentTaskDelay(taskDesc) {
268     var planif, delay = 0, planif = taskDesc.getInstance(self).getPlannification(),
269             completeness = parseInt(taskDesc.getInstance(self).getProperty('completeness')),
270             planifiedCompleteness = getPlannifiedCompleteness(taskDesc.getInstance(self));
271     if (completeness > 0 && planif.length > 0) {
272         if (planifiedCompleteness <= 0) {
273             delay = completeness + 100;
274         }
275         else {
276             delay = completeness / planifiedCompleteness * 100;
277         }
278     }
279     return delay;

282 /**
283  * Call fonction to creat activities (createActivities) then get each
284  *  activities (but only one per task's requirement). for each activities,
285  *   Call the function to calculate the progression of each requirement
286  *   (calculateProgressOfNeed).
287  *   Then, calculate and set the quality and the completeness for each tasks
288  *   Then, check the end of a requirement inactivities (function ''checkEnd'');
289  * @param {Number} currentStep
290  */
291 function calculTasksProgress(currentStep) {
292     var i, work, activitiesAsNeeds, oneTaskPerActivity, allCurrentActivities,
293             taskProgress, allCurrentActivities,
294             requirementsByWork, taskInst;
295     //create activities
296     allCurrentActivities = createActivities(currentStep);
297     //get one unique requirement by activities and calculate its progression
298     activitiesAsNeeds = getActivitiesWithEmployeeOnDifferentNeeds(allCurrentActivities);
299     for (i = 0; i < activitiesAsNeeds.length; i++) { //for each need
300         calculateProgressOfNeed(activitiesAsNeeds[i], allCurrentActivities, currentStep);
301     }
302     //get each modified task and calculate is new quality and completeness
303     oneTaskPerActivity = getUniqueTasksInActivities(activitiesAsNeeds);
304     for (i = 0; i < oneTaskPerActivity.length; i++) {
305         taskProgress = 0;
306         taskInst = oneTaskPerActivity[i].getTaskDescriptor().getInstance(self);
307         requirementsByWork = getRequirementsByWork(taskInst.getRequirements());
308         for (work in requirementsByWork) {
309             taskProgress += requirementsByWork[work].completeness;
310         }
311         taskProgress = (taskProgress > 97) ? 100 : taskProgress; //>97 yes, don t frustrate the players please.
312         taskInst.setProperty('completeness', Math.round(taskProgress));
313         taskInst.setProperty('quality', calculateTaskQuality(oneTaskPerActivity[i].getTaskDescriptor()));
314     }
315     checkEnd(allCurrentActivities, currentStep);
316 }

318 /**
319  * Sort an Array of activities to keep only one occurence of task Descriptor
320  * in each activities. So in the returned list, two activities can't have the same task.
321  * @param {Array} activities  an Array of Activity
322  * @returns {Array of activities}

325     var i, j, oneTaskPerActivity = [], wasAdded;

327         oneTaskPerActivity.push(activities[0]);
328     }

330         wasAdded = false;
331         for (j = 0; j < oneTaskPerActivity.length; j++) {

333                 wasAdded = true;
334                 break;
335             }
336         }
337         if (!wasAdded) {
338             oneTaskPerActivity.push(activities[i]);
339         }
340     }
341     return oneTaskPerActivity;

538     var i, ii, taskDesc, assignables = [], work, exist;

540         work = null;
541         exist = false;

543         if (parseInt(taskDesc.getInstance(self).getProperty('completeness')) < 100 && getPredecessorFactor(taskDesc) >= 0.2) { //if the task isn t terminated and average of predecessors advancement is upper than 20%

549                 }

686 /**
687  * Return an object which requirements are gathered by work.
688  * this object contains (by work) :
689  * - maxLimit, the bigger limit found on this requirement job
690  * - typesOfLevels, an Array of all found levels on this requirement job
691  * - totalOfEmployees, the sum of all employees required on this requirement job.
692  * - completeness the sum of each (completeness * quantity of required employees) divided by the total of employee on this requirement job.
693  * @param {WRequirement} requirements
694  * @returns {Object} works
695  */
696 function getRequirementsByWork(requirements) {
697     var i, req, works = {}, work, needsCompletion = 0, totalOfEmployees = 0;
698     for (i = 0; i < requirements.size(); i++) {
699         req = requirements.get(i);
700         //keep an occurance of each kind of work needed
701         if (works[req.getWork()]) {
702             work = works[req.getWork()];
703             work.totalByWork += req.getQuantity();
704         } else {
705             work = works[req.getWork()] = {
706                 maxLimit: 0,
707                 typesOfLevels: [],
708                 totalOfEmployees: 0,
709                 completeness: 0,
710                 totalByWork: req.getQuantity(),
711                 dimFactorAdvancement: false
712             };
713         }
714         //keep the highest limit of all limits from each kind of work needed
715         if (work.maxLimit < parseInt(req.getLimit())) {
716             work.maxLimit = parseInt(req.getLimit());
717         }
718         //keep all kind of levels for each kind of work needed
719         if (work.typesOfLevels.indexOf(req.getLevel()) <= -1) {
720             work.typesOfLevels.push(req.getLevel());
721         }
722         //keep the summe of personns needed for each kind of work needed
723         totalOfEmployees += parseInt(req.getQuantity());
724         //is needed for next calcul
725         needsCompletion += (parseFloat(req.getCompleteness()) * parseInt(req.getQuantity()));
726         work.completeness += parseFloat(req.getCompleteness());
727     }
728     for (work in works) {
729         //keep the summe of personns needed for each kind of work needed
730         works[work].totalOfEmployees = totalOfEmployees;
731 
732         if (works[work].completeness > (works[work].totalByWork / totalOfEmployees) * 100) {
733             works[work].dimFactorAdvancement = true;
734         }
735     }
736     return works;
737 }

920  * @param {TaskInstance} taskInst
921  * @returns {Number} a factor between 0 and 2

923 function getRandomFactorFromTask(taskInst) {
924     var rn = Math.floor(Math.random() * 100), //number 0 to 100 (0 inclusive, 100 exclusive);
925             randomDurationSup = parseFloat(taskInst.getProperty('randomDurationSup')),
926             randomDurationInf = parseFloat(taskInst.getProperty('randomDurationInf')),
927             duration = parseInt(taskInst.getDuration()), delta,

929

932     } else if (rn < 10) {
933         delta = -(0.25 * x + 0.50) * randomDurationInf;
934     } else if (rn < 25) {
935         delta = -(0.25 * x + 0.25) * randomDurationInf;
936     } else if (rn < 50) {
937         delta = -0.25 * x * randomDurationInf;
938     } else if (rn < 75) {
939         delta = 0.25 * x * randomDurationSup;
940     } else if (rn < 90) {
941         delta = (0.25 * x + 0.25) * randomDurationSup;
942     } else if (rn < 97) {
943         delta = (0.25 * x + 0.50) * randomDurationSup;
944     } else {
945         delta = (0.25 * x + 0.75) * randomDurationSup;
946     }
947

949 
950     if (randomFactor < minTaskDuration) {
951         randomFactor = minTaskDuration;
952     }
953
