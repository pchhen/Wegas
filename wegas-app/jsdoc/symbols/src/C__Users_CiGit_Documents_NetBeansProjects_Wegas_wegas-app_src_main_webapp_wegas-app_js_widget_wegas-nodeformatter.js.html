 13 YUI.add('wegas-nodeformatter', function (Y) {
 14     "use strict";
 15 
 16     var CONTENTBOX = 'contentBox', NodeFormatter;
 17 
 18     /**
 19      * @name Y.Wegas.NodeFormatter
 20      * @extends Y.Widget
 21      * @borrows Y.Wegas.Widget
 22      * @class return value in specifique node.
 23      * @constructor
 24      * @description returm value in specifique node. Available type:
 25      * Text node, image node, value box node, and position node.
 26      */
 27     NodeFormatter = Y.Base.create("wegas-nodeformatter", Y.Widget, [Y.Wegas.Widget], {
 28         /** @lends Y.Wegas.NodeFormatter# */
 29 
 30         // ** Lifecycle Methods ** //
 31         /**
 32          * @function
 33          * @private
 34          * @param value
 35          * @param label
 36          * @param className
 37          * @return node
 38          * @description return a div node containing a node value and a node
 39          *  label corresponding with the given parameters.
 40          *  Returned node have the given class name.
 41          */
 42         makeNodeText: function (value, label, className) {
 43             var node = Y.Node.create('<div class="nodeformatter-properties"></div>');
 44             value = (value !== null) ? value : 'undefine';
 45             if (className) {
 46                 node.addClass(className);
 47             }
 48             if (label) {
 49                 node.append('<span class="label">' + label + '</span>');
 50             }
 51             node.append('<span class="value">' + value + '</span>');
 52             return node;
 53         },
 54 
 55         /**
 56          * @function
 57          * @private
 58          * @param value
 59          * @param label
 60          * @param className
 61          * @return node
 62          * @description return a div node containing an image node. This image
 63          * will have the given attrs parameters as attribute.
 64          * Returned node have the given class name.
 65          */
 66         makeNodeImage: function (attrs, className) {
 67             var k, node = new Y.Node.create('<div class="nodeformatter-img"></div>');
 68             node.append('<img></img>');
 69             if (className) {
 70                 node.one('img').addClass(className);
 71             }
 72             if (typeof attrs !== 'object') {
 73                 return node;
 74             }
 75             for (k in attrs) {
 76                 node.one('img').setAttribute(k, attrs[k]);
 77             }
 78             return node;
 79         },
 80 
 81         /**
 82          * @function
 83          * @private
 84          * @param value
 85          * @param maxVal
 86          * @param label
 87          * @param className
 88          * @return node
 89          * @description return a div node containing a node label, a node
 90          *  containing as many nodes as shown in value and a node shown the
 91          *  value and the max value.
 92          *  Returned node have the given class name.
 93          */
 94         makeNodeValueBox: function (value, maxVal, label, className) {
 95             var i, acc = [], node = new Y.Node.create('<div class="nodeformatter-valuebox"></div>');
 96             value = (typeof parseInt(value) === 'number') ? parseInt(value) : 0;
 97             maxVal = (maxVal || 'undefine');
 98             label = (label || 'undefine');
 99             if (className) {
100                 node.addClass(className);
101             }
102             for (i = 0; i < value; i += 1) {
103                 acc.push('<div class="box-unit"></div>');
104             }
105             node.append('<div class="label">' + label + '</div>');
106             node.append('<span class="box-units">' + acc.join('') + '</span>');
107             node.append('<span class="box-value">(' + value + '<span class="box-valueMax">/' + maxVal + '</span>)</span>');
108             return node;
109         },
110 
111         /**
112          * @function
113          * @private
114          * @param html
115          * @param selector
116          * @param value
117          * @param minVal
118          * @param invert
119          * @param className
120          * @return node
121          * @description return a div node containing the given html and
122          *  highlight a nodes according to the given values (a number).
123          *  To highlight a node this function count same type node as the
124          *   "selector" parameter (like 'li', 'div', 'p', 'img', etc...).
125          * The first value depending of the given 'minVal'
126          * The 'invert' parameter allow to count reversely.
127          * Returned node have the given class name.
128          */
129         makeNodePosition: function (html, selector, value, minVal, invert, className) {
130             var node = new Y.Node.create('<div class="nodeformatter-position"></div>');
131             minVal = (typeof parseInt(minVal) === 'number') ? parseInt(minVal) : 0;
132             value = (typeof parseInt(value) === 'number') ? parseInt(value) - minVal : -1 - minVal; // if value isn't a number, value egal minVal - 1 (and thus never selected)
133             invert = (invert === 'true' || invert === true) ? true : false;
134             if (className) {
135                 node.addClass(className);
136             }
137             node.append(html);
138             node.all(selector).each(function (n, i, q) { //n = node, i = iteration, q = number of iterations.
139                 i = (invert) ? (q.size() - 1 - i) : i;
140                 n.setAttribute('data-position', i + minVal); //keep a reference to real number position
141                 if (i < value) {
142                     n.addClass('previous');
143                 } else if (i === value) {
144                     n.addClass('current');
145                 } else {
146                     n.addClass('next');
147                 }
148             });
149             return node;
150         }
151 
152     }, {
153         /*
154          * @lends Y.Wegas.NodeFormatter#
155          */
156         /**
157          * @field
158          * @static
159          * @description
160          * <p><strong>Method (none)</strong></p>
161          */
162         ATTRS: {}
163     });
164 
165     Y.namespace('Wegas').NodeFormatter = NodeFormatter;
166 });
