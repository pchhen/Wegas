 13 YUI.add("wegas-gallery", function(Y) {
 14     "use strict";
 15     var WegasGallery,
 16             CONTENT_BOX = "contentBox",
 17             BOUNDING_BOX = "boundingBox";
 18     /*
 19      * ATTRS:<br/>
 20      * gallery: {Array} an array of picture to display, format : {srcUrl: ...[, description: ...]}
 21      * fullScreen : {Boolean} toggles fullScreen<br/>
 22      * lightGallery : {Boolean} An instance which will read DOM for "light-picture" or "light-gallery" classes<br/>
 23      * <br/>
 24      * *light-picture class nodes : "href"|"src" attribute (url) , "title" attribute (description). <br/>
 25      *  Create a gallery with all elements with the same data-gallery attribute <br/>
 26      * *light-gallery class nodes : child nodes with "href"|"src" attribute (url), "title" attribute (description)
 27      */
 28     /**
 29      * @name Y.Wegas.WegasGallery
 30      * @extends Y.Widget
 31      * @borrows Y.Wegas.Widget
 32      * @class class to display image in lightbox
 33      * @constructor
 34      * @description Display image in lightbox with classic lightbox functions
 35      */
 36     WegasGallery = Y.Base.create("wegas-gallery", Y.Widget, [], {
 37         /**
 38          * @lends Y.Wegas.WegasGallery#
 39          */
 40 // *** Private fields *** //
 41         /**
 42          * Content box of this widget, static
 43          * @field
 44          * @private
 45          */
 46         CONTENT_TEMPLATE: "<ul></ul>",
 47         /**
 48          * Store reference to fullscreen node
 49          * @field
 50          * @private
 51          */
 52         FULLSCREENNODE: null,
 53         /**
 54          * Reference to the ScrollView widget
 55          * @field
 56          * @private
 57          */
 58         scrollView: null,
 59         /**
 60          * Current display mode of this widget (boolean fullscreen)
 61          * @field
 62          * @private
 63          */
 64         isFullScreen: false,
 65         /**
 66          * Reference to each used functions
 67          * @field
 68          * @private
 69          */
 70         handlers: null,
 71         /**
 72          * image loaded in the widget
 73          * @field
 74          * @private
 75          */
 76         images: null,
 77         /**
 78          * @function
 79          * @private
 80          * @description Set variables with initials values.
 81          * Create a ScrollView
 82          */
 83         initializer: function() {
 84             if (!this.constructor.prototype.FULLSCREENNODE) {
 85                 this.constructor.prototype.FULLSCREENNODE = new Y.Node.create("<div class='gallery-fullscreen'><span><span></div>");
 86                 this.constructor.prototype.FULLSCREENNODE.appendTo(Y.one("body"));
 87                 this.constructor.prototype.FULLSCREENNODE.on('contextmenu', function(e) {
 88                     e.preventDefault();
 89                 });
 90                 this.constructor.prototype.FULLSCREENNODE = this.FULLSCREENNODE.get("firstChild");
 91             }
 92 
 93             this.images = {};
 94             this.handlers = [];
 95             this.isFullScreen = false;
 96 
 97             this.scrollView = new Y.ScrollView({
 98                 width: (parseInt(this.get("selectedWidth")) + 30),
 99                 srcNode: this.get(BOUNDING_BOX),
100                 axis: "x",
101                 flick: false,
102                 bounce: 0,
103                 drag: false
104             });
105             if (this.get("lightGallery")) {
106                 this.set("render", "body");
107                 this.set("gallery", []);
108                 this.render();
109             }
110 
111             Y.StyleSheet.register(new Y.StyleSheet(), this.get("id"));
112         },
113         /**
114          * @function
115          * @private
116          * @description Render DOM skeleton of this widget.
117          */
118         renderUI: function() {
119 // LAZY init
120             this.scrollView.get(BOUNDING_BOX).append("<div class='gallery-mask gallery-mask-left'><div></div></div>");
121             this.scrollView.get(BOUNDING_BOX).append("<div class='gallery-mask gallery-mask-right'><div></div></div>");
122             this.scrollView.get(BOUNDING_BOX).append("<div class='gallery-toggle'></div>");
123             this.scrollView.get(BOUNDING_BOX).append("<div class='gallery-scroll-indicator'></div>");
124             this.get(CONTENT_BOX).appendChild("<li class='img-loading'></li>");
125             if (this.get("lightGallery")) {
126                 this.scrollView.get(BOUNDING_BOX).addClass("wegas-lightGallery");
127             } else if (this.get("gallery").length > 0) {
128                 this.loadImage(0);
129             }
130             this.galleryChangeEvent({newVal: this.get("gallery")});
131         },
132         /**
133          * @function
134          * @public
135          * @description set the css of widget to adjust display to the
136          *  loaded images
137          *  @returns {undefined}
138          */
139         syncUI: function() {
140             var galleryId,
141                     selW,
142                     selH,
143                     smaH,
144                     smaW,
145                     container,
146                     styleSheet = Y.StyleSheet(this.get("id"));
147             if (!this.scrollView.get("rendered")) {
148                 return;
149             }
150             this.set("selectedWidth", this.get("selectedWidth"));
151             this.set("selectedHeight", this.get("selectedHeight"));
152             galleryId = "#" + this.scrollView.get("id") + " ", // prefix css with id, allow multiple instance with different style
153                     selW = parseInt(this.get("selectedWidth")),
154                     selH = this.get("selectedHeight"),
155                     smaH,
156                     smaW;
157             if (styleSheet) {
158                 styleSheet.disable();
159             }
160             if (this.get("fullScreen") || this.get("lightGallery")) {
161                 selH = (Y.one("body").get("winHeight") - 100);
162                 selW = Y.one("body").get("winWidth") / 1.7;
163                 if (!this.isFullScreen) {
164                     this.isFullScreen = true;
165                     this.scrollView.get(BOUNDING_BOX).swap(this.FULLSCREENNODE);
166                 }
167             } else {
168                 if (this.isFullScreen) {
169                     this.isFullScreen = false;
170                     this.scrollView.get(BOUNDING_BOX).swap(this.FULLSCREENNODE);
171                 }
172             }
173             container = this.scrollView.get(BOUNDING_BOX).get("parentNode");
174             smaW = (parseInt(container.getComputedStyle("width")) - selW - 90) / 2;
175             smaH = selH * 0.5;
176             //smaW = selW * 0.6;
177             if (this.get("gallery").length > 0) {
178                 this.scrollView.show();
179             } else {
180                 this.scrollView.hide();
181                 return;
182             }
183             this.scrollView.set("width", selW + 30);
184             styleSheet.set(galleryId + ".wegas-gallery li", {width: (selW + 30) + "px", height: (selH + 30) + "px"});
185             styleSheet.set(galleryId + ".gallery-mask", {width: (smaW + 30) + "px"});
186             styleSheet.set(galleryId + ".gallery-text", {width: (selW + 24) + "px", "max-height": selH + "px"});
187             styleSheet.set(galleryId + ".wegas-gallery img", {"max-width": smaW + "px", "max-height": (selH + 30) + "px"});
188             styleSheet.set(galleryId + ".wegas-gallery .gallery-selected img", {"max-width": selW + "px", "max-height": selH + "px"});
189 
190             styleSheet.enable();
191             this.scrollView.get(BOUNDING_BOX).setStyles({
192                 padding: "0 " + (smaW + 45) + "px 0 " + (smaW + 15) + "px"
193             });
194             this.scrollView.syncUI();
195             if (this.scrollView.get("rendered")) {
196                 this.scrollView.pages.scrollToIndex(+this.scrollView.pages.get("index"));
197             }
198             if (this.get("lightGallery") && !this.get("fullScreen")) {
199                 this.scrollView.hide();
200             }
201             this.genScrollIndicator();
202         },
203         /**
204          * @function
205          * @private
206          * @description bind function to events.
207          * When full screen change, if it's the light gallery, show the
208          *  scrollView (fullscreen) or hide it. if it's not the light Gallery,
209          *  just do sync.
210          * When node ".gallery-mask-left" is clicked, stop loading and display previous picture.
211          * When node ".gallery-mask-right" is clicked, stop loading and display next picture.
212          * When node ".gallery-toggle" is clicked, toggle fullscreen
213          * When node ".gallery-mask-left" capte event "mousedown", prevent scrollview scrolling.
214          * When node ".gallery-mask-right" capte event "mousedown", prevent scrollview scrolling.
215          * When scrollEnd and after 600 milisecondes, load next pictures.
216          * When window is resized, do windowResizeEvent.
217          * When gallery Change, reset scrollview and select picture 0.
218          * When node 'light-gallery' is clicked and if it's lightGallery,
219          *  active full screen and load pictures.
220          */
221         bindUI: function() {
222             /* CLICK events */
223             this.scrollView.get(BOUNDING_BOX).one(".gallery-scroll-indicator").delegate("click", function(e) {
224                 e.halt(true);
225                 this.scrollView.pages.scrollToIndex(+e.currentTarget.getData("index"));
226             }, "span", this);
227             this.scrollView.get(BOUNDING_BOX).one('.gallery-mask-left').on("click", function(e) {
228                 e.halt(true);
229                 this.prev();
230             }, this);
231             this.scrollView.get(BOUNDING_BOX).one('.gallery-mask-right').on("click", function(e) {
232                 e.halt(true);
233                 this.next();
234             }, this);
235             this.scrollView.get(BOUNDING_BOX).one('.gallery-toggle').on("click", function(e) {
236                 e.halt(true);
237                 this.set("fullScreen", !this.get("fullScreen"));
238             }, this);
239             this.handlers.push(this.get(CONTENT_BOX).delegate("click", function(e) {
240                 e.halt(true);
241                 this.set("fullScreen", true);
242             }, "li > *", this));
243             this.handlers.push(this.scrollView.get(BOUNDING_BOX).on("click", function(e) {
244                 e.halt(true);
245                 this.set("fullScreen", false);
246             }, this));
247             /* SYSTEM events */
248             this.scrollView.after("scrollEnd", function(e) {
249                 Y.later(50, this, function() {                                  // Let some time before loading next one
250                     var index = +this.scrollView.pages.get("index");
251                     if (this.images[index + 1] && !this.images[index + 1].loaded) {
252                         this.loadImage(index + 1);
253                     }
254                     if (this.images[index - 1] && !this.images[index - 1].loaded) {
255                         this.loadImage(index - 1);
256                     }
257                 });
258             }, this);
259             this.after("fullScreenChange", function(e) {
260                 if (this.get("lightGallery")) {
261                     if (this.get("fullScreen")) {
262                         this.scrollView.show();
263                     } else {
264                         this.scrollView.hide();
265                     }
266                 } else {
267                     this.syncUI();
268                 }
269             });
270             this.handlers.push(Y.on("windowresize", Y.bind(this.windowResizeEvent, this)));
271             this.after("galleryChange", this.galleryChangeEvent);
272             /* LIGHTMODE events */
273             if (this.get("lightGallery")) {
274                 this.handlers.push(Y.one("body").delegate("click", function(e) {
275                     var children, gallery = [];
276                     e.halt(true);
277                     children = e.target.get("children");
278                     children.each(function() {
279                         if (this.getAttribute("href") || this.getAttribute("src")) {
280                             gallery.push({
281                                 srcUrl: this.getAttribute("href") || this.getAttribute("src"),
282                                 description: this.getAttribute("title")
283                             });
284                         }
285                     });
286                     this.set("fullScreen", true);
287                     this.set("gallery", gallery);
288                 }, '.light-gallery', this));
289                 this.handlers.push(Y.one("body").delegate("click", function(e) {
290                     var gallery = [], index;
291                     e.halt(true);
292                     if (e.target.getAttribute("href") || e.target.getAttribute("src")) {
293                         this.set("fullScreen", true);
294                         if (e.target.hasAttribute("data-gallery")) {            /* group same data-gallery together */
295                             Y.all("[data-gallery='" + e.target.getAttribute("data-gallery") + "']").each(function(item, i) {
296                                 if (item === e.target) {
297                                     index = i;
298                                 }
299                                 gallery.push({
300                                     srcUrl: item.getAttribute("href") || item.getAttribute("src"),
301                                     description: item.getAttribute("title")
302                                 });
303                             });
304                             this.set("gallery", gallery);
305                             this.scrollView.pages.scrollToIndex(index);
306                         } else {
307                             this.set("gallery", [{
308                                     srcUrl: e.target.getAttribute("href") || e.target.getAttribute("src"),
309                                     description: e.target.getAttribute("title")
310                                 }]);
311                         }
312                     }
313                 }, '.light-picture', this));
314             }
315         },
316         galleryChangeEvent: function(event) {
317             if (event.newVal.length > 0) {
318                 this.loadImage(0);
319                 if (event.newVal.length > 1) {
320                     this.loadImage(1);
321                 }
322                 if (!this.scrollView.get("rendered")) {
323                     this.scrollView.plug(Y.Plugin.ScrollViewPaginator, {
324                         selector: 'li'
325                     });
326                     this.scrollView.render();
327                     this.scrollView.pages.after("indexChange", function(e) {
328                         this.setSelected(e.target.get("index"));
329                     }, this);
330                 } else {
331                     this.scrollView.syncUI();
332                 }
333 
334                 this.scrollView.pages.set("index", 0);
335                 this.syncUI();
336                 this.setSelected(0);
337             }
338         },
339         /**
340          * @function
341          * @private
342          * @description disable stylesheet, remove full screen node and detach
343          *  all functions created by this widget.
344          */
345         destructor: function() {
346             if (Y.StyleSheet(this.get("id"))) {
347                 Y.StyleSheet(this.get("id")).disable();
348             }
349             for (var i = 0; i < this.handlers.length; i = i + 1) {
350                 this.handlers[i].detach();
351             }
352             this.scrollView.destroy();
353         },
354         // *** Private Methods *** //
355         /**Generate index indicator
356          * @private
357          * @function
358          * @returns {undefined}
359          */
360         genScrollIndicator: function() {
361             var container = this.scrollView.get(BOUNDING_BOX).one(".gallery-scroll-indicator"), i;
362             container.empty();
363             if (this.get("gallery").length > 1) {
364                 for (i = 0; i < this.get("gallery").length; i += 1) {
365                     container.append("<span data-index='" + i + "'></span>");
366                 }
367                 this.setSelected(+this.scrollView.pages.get("index"));
368             }
369         },
370         /**
371          * @function
372          * @private
373          * @description if this is in fullscreen, do sync properly
374          */
375         windowResizeEvent: function() {
376             if (this.get("fullScreen")) {
377                 this.scrollView.hide();
378                 this.syncUI();
379                 this.scrollView.show();
380             }
381         },
382         /**
383          * @function
384          * @private
385          * @description pass to the next page in scrollview.
386          */
387         next: function() {
388             this.scrollView.pages.next();
389         },
390         /**
391          * @function
392          * @private
393          * @description pass to the previous page in scrollview.
394          */
395         prev: function() {
396             this.scrollView.pages.prev();
397         },
398         /**
399          * @function
400          * @private
401          * @param index
402          * @description call "loadImage fonction to load selected image and
403          *  hide/show next/previous picture's controls accordings with the
404          *  existence of a next/previous picture.
405          */
406         setSelected: function(index) {
407             var list = this.get(CONTENT_BOX).all("li");
408             list.each(function() {
409                 this.removeClass("gallery-selected");
410                 this.removeClass("gallery-before-selected");
411             });
412             if (+index > 0) {
413                 list.item(index - 1).addClass("gallery-before-selected");
414                 this.scrollView.get(BOUNDING_BOX).one(".gallery-mask-left > div").show();
415             } else {
416                 this.scrollView.get(BOUNDING_BOX).one(".gallery-mask-left > div").hide();
417             }
418             if (+index === list.size() - 1) {
419                 this.scrollView.get(BOUNDING_BOX).one(".gallery-mask-right > div").hide();
420             } else {
421                 this.scrollView.get(BOUNDING_BOX).one(".gallery-mask-right > div").show();
422             }
423             if (list.size() > +index) {
424                 list.item(index).addClass("gallery-selected");
425             }
426             if (this.images[+index] && !this.images[+index].loaded) {
427                 this.loadImage(+index);
428             }
429             this.scrollView.get(BOUNDING_BOX).one(".gallery-scroll-indicator").all("span").each(function(item, id) {
430                 if (+index === id) {
431                     item.addClass("gallery-current-index");
432                 } else {
433                     item.removeClass("gallery-current-index");
434                 }
435             });
436 
437 
438         },
439         /**
440          * @function
441          * @private
442          * @param i
443          * @description load selected image or display error's image if an error
444          * occur.
445          */
446         loadImage: function(i) {
447             var img;
448             if (this.get(CONTENT_BOX).all("li").item(i).hasChildNodes()) {
449                 return;
450             }
451             img = Y.Node.create("<img src='" + this.get("gallery")[i].srcUrl + "' ></img>");
452             img.index = i;
453             img.once("error", function(e) {
454                 e.target.get("parentNode").setStyles({
455                     background: "url('../../wegas-app/images/wegas-icon-error-48.png') no-repeat 50%"
456                 });
457             });
458             img.once("load", function(e) {
459                 e.target.get("parentNode").removeClass("img-loading");
460                 this.images[e.target.index].loaded = true;
461             }, this);
462             this.get(CONTENT_BOX).all("li").item(i).appendChild(img);
463             if (this.get("gallery")[i].description) {
464                 this.get(CONTENT_BOX).all("li").item(i).appendChild("<div class='gallery-text'>" + this.get("gallery")[i].description + "</div>");
465             }
466         }
467     }, {
468         /** @lends Y.Wegas.WegasGallery */
469         CSS_PREFIX: "wegas-gallery",
470         /**
471          * @field
472          * @static
473          * @description
474          * <p><strong>Attributes</strong></p>
475          * <ul>
476          *    <li>gallery: contains list of pictures</li>
477          *    <li>selectedWidth: width of the gallery</li>
478          *    <li>selectedHeight: height of the gallery</li>
479          *    <li>fullScreen: display mode of the gallery (boolean)</li>
480          *    <li>lightGallery: style mode of the gallery (boolean)</li>
481          * </ul>
482          */
483         ATTRS: {
484             /**
485              *  Contains list of pictures
486              */
487             gallery: {
488                 value: [],
489                 type: "array",
490                 lazyAdd: true,
491                 setter: function(o) {
492                     this.get(CONTENT_BOX).empty();
493                     this.images = {};
494                     for (var i in o) {
495                         if (Y.Lang.isString(o[i])) {
496                             o[i] = {// If the arguent is a string, treat it as an url
497                                 srcUrl: o[i]
498                             };
499                         }
500                         this.images[i] = {
501                             loaded: false
502                         };
503                         this.get(CONTENT_BOX).appendChild("<li class='img-loading'></li>");
504                     }
505                     return o;
506                 }
507             },
508             /**
509              * Width of the gallery
510              */
511             selectedWidth: {
512                 value: 250,
513                 type: "number",
514                 validator: Y.Lang.isNumber,
515                 setter: function(v) {
516                     if (this.scrollView) {
517                         this.scrollView.set("width", (parseInt(v) + 30));
518                     }
519                     return v;
520                 }
521             },
522             /**
523              * Height of the gallery
524              */
525             selectedHeight: {
526                 value: 250,
527                 type: "number",
528                 validator: Y.Lang.isNumber
529             },
530             /**
531              * Display mode of the gallery (boolean)
532              */
533             fullScreen: {
534                 value: false,
535                 "transient": "true",
536                 validator: Y.Lang.isBoolean
537             },
538             /**
539              * Style mode of the gallery (boolean)
540              */
541             lightGallery: {
542                 value: false,
543                 writeOnce: "initOnly",
544                 validator: Y.Lang.isBoolean
545             }
546 
547         }
548     });
549     Y.namespace("Wegas.util").WegasGallery = WegasGallery;

553     Y.Wegas.Gallery = Y.Base.create("wegas-gallery", WegasGallery, [Y.Wegas.Widget, Y.Wegas.Editable], {}, {});
554     /**
555      * @name Y.Wegas.FileExplorerGallery
556      * @extends Y.Wegas.WegasGallery
557      * @class class to create a loader-image
558      * @constructor
559      * @description create a loader-image and remove them when wanted image
560      *  is loaded.
561      */
562     var FileExplorerGallery = Y.Base.create("wegas-gallery", WegasGallery, [], {
563         /** @lends Y.Wegas.FileExplorerGallery */
564 
565 // *** Private Methods *** //
566         /**
567          * @function
568          * @private
569          * @param i
570          * @description create a loader-image and remove them when wanted image
571          *  is loaded. If an error occur, display an error image.
572          */
573         loadImage: function(i) {
574             var img, imgLoader;
575             if (this.get(CONTENT_BOX).all("li").item(i).hasChildNodes()) {
576                 return;
577             }
578 
579             img = Y.Node.create("<img />");
580             imgLoader = new Y.Wegas.ImgageLoader({
581                 target: img,
582                 srcUrl: Y.Plugin.CRDataSource.getFullpath(this.get("gallery")[i].srcUrl)
583             });
584             imgLoader.fetch();
585             imgLoader.once("load", function(e) {
586                 if (e.meta.description) {
587                     this.get(CONTENT_BOX).all("li").item(i).appendChild("<div class='gallery-text'>" + e.meta.description + "</div>");
588                 }
589             }, this);
590             img.index = i;
591             img.once("error", function(e) {
592                 e.target.get("parentNode").setStyles({
593                     background: "url('../../wegas-app/images/wegas-icon-error-48.png') no-repeat 50%"
594                 });
595             });
596             img.once("load", function(e) {
597                 e.target.get("parentNode").removeClass("img-loading");
598                 this.images[e.target.index].loaded = true;
599             }, this);
600             this.get(CONTENT_BOX).all("li").item(i).appendChild(img);
601         }
602     }, {CSS_PREFIX: "wegas-gallery"});
603     Y.namespace("Wegas.util").FileExplorerGallery = FileExplorerGallery;
604 });
