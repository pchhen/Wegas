 13 YUI.add('wegas-chart', function(Y) {
 14     var CONTENTBOX = 'contentBox',
 15             Chart = Y.Base.create("wegas-chart", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 16         renderUI: function() {
 17             this.handlers = [];
 18             var cb = this.get(CONTENTBOX);
 19             cb.setContent(
 20                     '<div style="width:' + this.checkType(this.get("chartWidth")) + '; height:' + this.checkType(this.get("chartHeight")) + ';" class="chart"></div>'
 21                     );
 22         },
 23         bindUI: function() {
 24             this.handlers.push(
 25                     Y.Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this));
 26         },
 27         syncUI: function() {
 28             var i;
 29             this.vdList = [];
 30             var variable = this.get("variables");
 31             for (i = 0; i < variable.length; i++) {
 32                 var vd = Y.Wegas.Facade.VariableDescriptor.cache.find("name", variable[i].name);
 33                 vd.position = i;
 34                 if (variable[i].label)
 35                     vd.label = variable[i].label;
 36                 else
 37                     vd.label = variable[i].name;
 38                 this.historyRequest(vd);
 39 
 40             }
 41         },
 42         destructor: function() {
 43             for (var i = 0; i < this.handlers.length; i = i + 1) {
 44                 this.handlers[i].detach();
 45             }
 46         },
 47         historyRequest: function(vd) {
 48             Y.Wegas.Facade.VariableDescriptor.sendRequest({
 49                 request: "/" + vd.get("id") + "/VariableInstance/" + vd.getInstance().get("id") + "?view=Export",
 50                 on: {
 51                     success: Y.bind(function(r) {
 52                         var a = Y.JSON.parse(r.data.responseText);    
 53                         a.entities[0].label = vd.label;
 54                         this.vdList.splice(vd.position, 0, a.entities[0]);
 55                         if (this.vdList.length === this.get("variables").length) {
 56                             this.clear(this.get(CONTENTBOX));
 57                             this.createChart();
 58                         }
 59                     }, this),
 60                     failure: function(r) {
 61                         Y.error("Error by loading history data");
 62                     }
 63                 }
 64             });
 65         },
 66         /**
 67          * Creat a YUI3 Charts combospline' with values of a resource's moral and confidence historic values.
 68          * If any resource is given, the chart will be not created.
 69          * @ Param NumberDescriptor numberDescriptor, the source of chart's values
 70          */
 71         createChart: function() {
 72             if (this.chart)
 73                 this.chart.destroy();
 74             if (this.vdList.length < 1)
 75                 return;
 76             var i,
 77                     seriesCollection = [],
 78                     rawSeries = [],
 79                     obj;
 80             for (i = 0; i < this.vdList.length; i++) {
 81                 obj = {
 82                     yDisplayName: this.vdList[i].label
 83                 };
 84                 seriesCollection.push(obj);
 85                 rawSeries.push(this.vdList[i].history);
 86             }
 87             
 88             this.chart = new Y.Chart({
 89                 type: this.get("chartType"),
 90                 seriesCollection: seriesCollection,
 91                 axes: {
 92                     values: {
 93                         minimum: this.get("minValue"),
 94                         maximum: this.get("maxValue")
 95                     }
 96                 },
 97                 legend: {
 98                     styles: {
 99                         gap: 0
100                     },
101                     position: this.get("legendPosition")
102                 },
103                 tooltip: this.chartTooltip,                
104                 dataProvider: this.getChartValues(this.get("numberOfValue"), rawSeries),
105                 horizontalGridlines: this.get("horizontalGridlines"),
106                 verticalGridlines: this.get("verticalGridlines")
107             });
108             this.chart.render(".chart");
109         },
110         /**
111          * Create series for the chart.
112          * i = numberOfValues
113          * For each series, If number of values is smaller than i, copy the last value to create a serie with i values.
114          * If number of values is greater than i, keep only the i last values.
115          * @param Integer numberOfValues, the number of value wanted in the series.
116          * @param Array rawSeries, an array of array of Integer.
117          */
118         getChartValues: function(numberOfValues, rawSeries) {
119             var i, j, fitSeries = new Array(), serieRawData = new Array(), serieFitData = new Array();
120             for (i = 0; i < numberOfValues; i++) {
121                 serieFitData.push(i+1);
122             }
123             fitSeries.push(serieFitData.slice());
124             for (i = 0; i < rawSeries.length; i++) {
125                 serieRawData = rawSeries[i];
126                 serieFitData.length = 0;
127                 for (j = numberOfValues - 1; j >= 0; j--) {
128                     if (serieRawData.length - 1 >= j) {
129                         serieFitData.push(serieRawData[serieRawData.length - (j + 1)]);
130                     }
131                 }
132                 fitSeries.push(serieFitData.slice());
133             }
134             return fitSeries;
135         },
136         chartTooltip: {
137             markerLabelFunction: function(categoryItem, valueItem, itemIndex, series, seriesIndex) {
138                 var msg = document.createElement("div"),
139                         boldTextBlock = document.createElement("div");
140                 boldTextBlock.appendChild(document.createTextNode(valueItem.displayName + ': ' + valueItem.axis.get("labelFunction").apply(this, [valueItem.value])));
141                 msg.appendChild(boldTextBlock);
142                 return msg;
143             }
144         },
145         clear: function(cb) {
146             this.chart = null;
147             cb.one('.chart').setHTML();
148         },
149         checkType: function(value){
150             if (value.substr(-2) !== "px" && value.substr(-2) !== "pt" && value.substr(-2) !== "em" && value.substr(-1) !== "%"){
151                 return value + "px";
152             } else {
153                 return value;
154             }
155             
156         }
157     }, {
158         ATTRS: {
159             /**
160              * The target variable, returned either based on the variableName attribute,
161              * and if absent by evaluating the expr attribute.
162              */
163             variables: {
164                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
165                 _inputex: {
166                     _type: "list",
167                     useButtons: true,
168                     elementType: {
169                         type: "variableselect",
170                         label: "variable"
171                     },
172                     index: 1
173                 }
174             },
175             chartType: {
176                 value: "combo",
177                 _inputex: {
178                     _type: "select",
179                     value: "bottom",
180                     label: "Chart type",
181                     choices: [{ value: 'combo' }, { value: 'line' }],
182                     index: 0
183                 }
184             },
185             minValue: {
186                 value: 0,
187                 _inputex: {
188                     _type: "integer",
189                     label: "Min. value",
190                     index: 2
191                 }
192             },
193             maxValue: {
194                 value: 100,
195                 _inputex: {
196                     _type: "integer",
197                     label: "Max. value",
198                     index: 3
199                 }
200             },
201             chartWidth: {
202                 value: "250px",
203                 _inputex: {
204                     _type: "string",
205                     label: "Width",
206                     index: 5
207                 }
208             },
209             chartHeight: {
210                 value: "200px",
211                 _inputex: {
212                     _type: "string",
213                     label: "Height",
214                     index: 6
215                 }
216             },
217             numberOfValue: {
218                 value: 5,
219                 _inputex: {
220                     _type: "integer",
221                     label: "Number of value",
222                     index: 4
223                 }
224             },
225             horizontalGridlines: {
226                 value: true,
227                 _inputex: {
228                     _type: "boolean",
229                     label: "Horizontal Gridlines",
230                     index: 8
231                 }
232             },
233             verticalGridlines: {
234                 value: true,
235                 _inputex: {
236                     _type: "boolean",
237                     label: "Vertical Gridlines",
238                     index: 9
239                 }
240             },
241             legendPosition: {
242                 value: "bottom",
243                 _inputex: {
244                     _type: "select",
245                     value: "bottom",
246                     label: "Legend position",
247                     choices: [{ value: 'bottom' }, { value: 'left' }, { value: 'right' }, { value: 'top' }],
248                     index: 7
249                 }            
250             }
251         }
252     });
253 
254     Y.namespace('Wegas').Chart = Chart;
255 });
