 12 YUI.add('wegas-layout-choicelist', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      * @name Y.Wegas.ChoiceList
 17      * @extends Y.Wegas.List
 18      * @class class to serialize widgets
 19      * @constructor
 20      * @description class to serialize widgets
 21      */
 22     var List = Y.Base.create("wegas-choicelist", Y.Wegas.List,
 23             [Y.Wegas.Editable, Y.Wegas.Layout], {
 24         /** @lends Y.Wegas.ChoiceList# */
 25         bindUI: function() {
 26             this.after("addChild", function() {
 27                 this.set("element", this.get("element"));
 28             });
 29         },
 30         syncUI: function() {
 31             List.superclass.syncUI.apply(this);
 32             this.set("element", this.get("element"));
 33         },
 34         getActiveElement: function() {
 35             return this.item(this.get("element"));
 36         }
 37 
 38     }, {
 39         EDITORNAME: "Choice List",
 40         CSS_PREFIX: "wegas-choicelist",
 41         /**
 42          * @lends Y.Wegas.ChoiceList
 43          */
 44         /**
 45          * @field
 46          * @static
 47          * @description
 48          * <p><strong>Attributes</strong></p>
 49          * <ul>
 50          *    <li>defaultChildType: default value for children. Transient.</li>
 51          *    <li>children: list of widget. Transient.</li>
 52          *    <li>direction: string-helper to add class and set style of the
 53          *     list (vertical or horizontal). Vertical by default</li>
 54          * </ul>
 55          */
 56         ATTRS: {
 57             defaultChildType: {
 58                 value: "Text",
 59                 "transient": true
 60             },
 61             children: {
 62                 "transient": true
 63             },
 64             direction: {
 65                 value: 'vertical',
 66                 readOnly: true,
 67                 "transient": true
 68             },
 69             element: {
 70                 value: 0,
 71                 "transient": true,
 72                 type: "number",
 73                 validator: Y.Lang.isNumber,
 74                 setter: function(val) {
 75                     if (this.size() > 0) {
 76                         val = val % this.size();
 77                         this.each(function(item) {
 78                             item.get("boundingBox").addClass("hiddenChild");
 79                         });
 80                         this.item(val).get("boundingBox").removeClass("hiddenChild");
 81                     }
 82                     return val;
 83                 }
 84             }
 85         }
 86     });
 87     Y.namespace('Wegas').ChoiceList = List;
 88 
 89 });
