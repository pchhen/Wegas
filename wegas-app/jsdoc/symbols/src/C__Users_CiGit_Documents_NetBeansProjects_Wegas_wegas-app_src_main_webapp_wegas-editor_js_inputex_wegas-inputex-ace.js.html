 13 YUI.add('wegas-inputex-ace', function(Y) {
 14     "use strict";
 15 
 16     var inputEx = Y.inputEx;
 17 
 18     /**
 19      * Ace code editor field
 20      */
 21     inputEx.AceField = function(options) {
 22         inputEx.AceField.superclass.constructor.call(this, options);
 23     };
 24 
 25     Y.extend(inputEx.AceField, inputEx.Textarea, {
 26         /**
 27          * Set the default values of the options
 28          * @param {Object} options Options object as passed to the constructor
 29          */
 30         setOptions: function(options) {
 31             inputEx.AceField.superclass.setOptions.call(this, options);
 32 
 33             this.options.language = options.language || "javascript";
 34             this.options.theme = options.theme || "textmate";
 35             this.options.height = options.height || "150px";
 36         },
 37         disable: function() {
 38             this.editor.setReadOnly(true);
 39         },
 40         enable: function() {
 41             this.editor.setReadOnly(false);
 42         },
 43         /**
 44          * Render the field using the YUI Editor widget
 45          */
 46         renderComponent: function() {
 47             if (window.ace) {                                                   // Ace is present, launch it
 48                 this.el = Y.Node.create('<div>' + (this.options.value || "") + '</div>');//@fixme
 49                 this.fieldContainer.appendChild(this.el.getDOMNode());
 50 
 51                 this.editor = ace.edit(this.el.getDOMNode());
 52                 this.editor.setTheme("ace/theme/" + this.options.theme);
 53                 this.editor.setShowPrintMargin(false);
 54                 //this.editor.setOptions({
 55                 //    //enableBasicAutocompletion: true,
 56                 //    enableSnippets: true
 57                 //});
 58                 //this.editor.setHighlightActiveLine(false);
 59                 //this.editor.renderer.setHScrollBarAlwaysVisible(false);
 60 
 61                 this.session = this.editor.getSession();
 62                 this.session.setMode("ace/mode/" + this.options.language);
 63 
 64                 Y.Wegas.app.after("layout:resize", function() {                 // Every time the layout is resized (wegas editor only)
 65                     Y.once('domready', this.resize, this);                      // resize ace viewport
 66                 }, this.editor);
 67 
 68                 Y.after('windowresize', Y.bind(this.editor.resize, this.editor));// Every time window is resized, resize ace viewport
 69             } else {                                                            // Fallback with textarea
 70                 Y.log("Unable to find Ace libraries, falling back to text field", "error", "Wegas.Inputex.Ace");
 71                 inputEx.AceField.superclass.renderComponent.call(this);
 72             }
 73         },
 74         /**
 75          * Set the html content
 76          * @param {String} value The html string
 77          * @param {boolean} [sendUpdatedEvt] (optional) Wether this setValue should fire the 'updated' event or not (default is true, pass false to NOT send the event)
 78          */
 79         setValue: function(value, sendUpdatedEvt) {
 80             if (this.session) {
 81                 this.session.setValue(value);
 82                 if (sendUpdatedEvt !== false) {
 83                     // fire update event
 84                     this.fireUpdatedEvt();
 85                 }
 86             } else {    // fallback
 87                 return inputEx.AceField.superclass.setValue.apply(this, arguments);
 88             }
 89         },
 90         /**
 91          * Get the ace content
 92          * @return {String} the ace area content string
 93          */
 94         getValue: function() {
 95             if (this.session) {
 96                 return this.session.getValue();
 97             } else {
 98                 return inputEx.AceField.superclass.getValue.apply(this);
 99             }
100         }
101     });
102 
103     inputEx.registerType("ace", inputEx.AceField, []);                          // Register this class as "html" type
104 });
