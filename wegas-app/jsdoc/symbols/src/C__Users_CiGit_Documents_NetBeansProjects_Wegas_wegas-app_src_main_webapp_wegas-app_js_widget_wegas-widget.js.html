 23     function Widget() {
 24         this.after("render", function() {
 25             var bb = this.get(BOUNDING_BOX);
 26             bb.addClass("wegas-widget");
 27             if (this.get("cssClass")) {
 28                 bb.addClass(this.get("cssClass"));
 29             }
 30         });
 31         this.constructor.CSS_PREFIX = this.constructor.CSS_PREFIX               // If no prefix is set, use the name (without
 32                 || this.constructor.NAME.toLowerCase();                         // the usual "yui3-" prefix)
 33         this._cssPrefix = this.constructor.CSS_PREFIX;
 34 
 35         this.publish("showOverlay", {// Add custom event
 36             emitFacade: true
 37         });
 38         this.publish("hideOverlay", {
 39             emitFacade: true
 40         });
 41         this.publish("message", {
 42             emitFacade: true
 43         });
 44         this.publish("AttributesChange", {
 45             defaultFn: function() {
 46                 var widget = this.rebuild();
 47                 if (Y.Plugin.EditEntityAction.currentEntity === this) {
 48                     Y.Plugin.EditEntityAction.currentEntity = widget;
 49                 }
 50             }
 51         });
 52     }

 57         /**
 58          * @function
 59          * @private
 60          */
 61         defaultFailureHandler: function(e) {
 62             this.hideOverlay();
 63             var error = e.response.message || e.response.results.message || "Error during request.",
 64                     test = error.match(/ConstraintViolationException: (.*) is out of bound/),
 65                     stringMessage = error.match(/Error: StringMessage: (.*)/);
 66             if (test) {
 67                 this.showMessageBis("error", "You don't have enough " + test[1] + ".");
 68             } else if (stringMessage) {
 69                 this.showMessageBis("error", stringMessage[1].split(" in <")[0]);
 70             } else {
 71                 this.showMessageBis("error", error);
 72             }
 73             // e.halt(true);
 74         },

 76          * @function
 77          * @private
 78          * @description show an loading - overlay on all the screen.

 81             this.fire("wegas:showOverlay")

 83         },

 85          * @function

 92         /**
 93          * Display a closable message with a status-image.
 94          * Status-image of message depends of level parameters
 95          * Txt parameters is the displayed text.
 96          * Timeout is the displaying time of this message.
 97          * @function
 98          * @private
 99          * @param level
100          * @param txt
101          * @param timeout
102          * @description
103          *
104          * @deprecated Will be replace by showMessageBis
105          */
106         showMessage: function(level, txt, timeout) {
107             this.emitDOMMessage(level, {content: txt, timeout: timeout});
108         },
109         emitDOMMessage: function() {
110             var bb = this.get("boundingBox");
111             bb.emitDOMMessage.apply(bb, arguments);
112         },
113         /**
114          * Will replace the original showMessage
115          */
116         showMessageBis: function(level, msg, timeout) {
117             this.fire("wegas:message", {
118                 level: level,
119                 content: msg,
120                 timeout: timeout
121             });
122         },
123         highlight: function(bool) {
124             if (!this.get("destroyed")) {
125                 bool ? this.get(BOUNDING_BOX).addClass("highlighted") : this.get(BOUNDING_BOX).removeClass("highlighted");
126             }
127         },
128         rebuild: function() {
129             var parent, index, cfg;
130             if (this.isRoot()) {
131                 parent = Y.Widget.getByNode(this.get(BOUNDING_BOX).get("parentNode"));
132                 parent.reload();
133                 return parent.get("widget"); // dependencies should (and must) be loaded by now that way we obtain the new widget
134             }
135             parent = this.get("parent");
136             index = parent.get("children").indexOf(this);
137             cfg = this.toObject();
138             this.destroy();
139             return parent.add(cfg, index).item(0);
140         }

509             var child, Fn, type = config.childType || config.type;
510             if (type) {
511                 Fn = Lang.isString(type) ? Y.Wegas[type] || Y[type] : type;
512             }
513 
514             if (Lang.isFunction(Fn)) {
515                 child = new Fn(config);
516             } else {
517                 Y.log("Could not create a child widget because its constructor is either undefined or invalid(" + type + ").", "error", "Wegas.Widget");
518             }
519

521         },

534          *  This getter is to be used for any object attribute that references a VariableDescriptor and

536          *
537          */
538         /**
539          * @function
540          * @private
541          * @param cfg, cb
542          * @return value
543          * @description This getter is to be used for any object attribute
544          *  that references a VariableDescriptor and has either an name, id
545          *  or expr parameter.
546          */
547         VARIABLEDESCRIPTORGETTER: function(val, fullName) {
548             var ds = Y.Wegas.Facade.VariableDescriptor;
549             if (val && fullName.split(".")[1] === "evaluated") {                // If evaluated value is required
550 
551                 if (val.name) {                                                 // Eval based on the name field
552                     val.evaluated = ds.cache.find('name', val.name);
553 
554                 } else if (val.expr) {                                          // if absent evaluate the expr field
555                     try {
556                         val.evaluated = ds.cache.findById(Y.Wegas.Facade.VariableDescriptor.script.localEval(val.expr));
557                     } catch (e) {
558                         Y.log("Unable to read expression: " + val.expr, "error", "Wegas.Widget");
559                         val.evaluated = null;
560                     }
561 
562                 } else if (val.id) {
563                     val.evaluated = ds.cache.findById(val.id);
564                 }
565             }
566 
567             if (val && fullName.indexOf(".") < 0) {                             // If the getter requires the full object (e.g. serialisation)
568                 delete val.evaluated;                                           // Remove the ref to the evaluated descriptor
569             }
570 
571             return val;
572         },

581     Y.WidgetParent.prototype._createChild = function(config) {
582         var altType = config.childType || config.type;
583         if (altType) {
584             config.childType = Y.Lang.isString(altType) ? Y.Wegas[altType] || Y[altType] : altType;
585         }
586         return baseCreateChild.call(this, config);                              //reroute
587     };

592     Y.Widget.prototype.plug = function(Plugin, config) {
593         if (!Lang.isArray(Plugin)) {
594             if (Plugin && !Lang.isFunction(Plugin)) {
595                 config = Plugin.cfg;
596                 Plugin = Plugin.fn;
597             }
598             if (Plugin && !Lang.isFunction(Plugin)) {                           // @hacked
599                 Plugin = Y.Plugin[Plugin];
600             }
601         }
602         basePlug.call(this, Plugin, config);                                    //reroute
603     };

605     /**
606      * Simulate a DOM Event bubbling up to a listener and stops.
607      * @param {String} type
608      * @param {Object} data
609      */
610     Y.Node.prototype.emitDOMMessage = function(type, data) {
611         var ev = "dom-message:" + type;
612         data = data || {};
613         data.type = type;
614         try {
615             this.ancestor(function(node) {
616                 return node.getEvent(ev) ? true : false;
617             }, true).fire(ev, data);
618         } catch (e) {
619             //no ancestor found
620         }
621     };
