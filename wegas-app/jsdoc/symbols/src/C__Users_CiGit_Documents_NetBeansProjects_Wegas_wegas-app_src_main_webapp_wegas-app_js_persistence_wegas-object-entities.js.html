 12 YUI.add('wegas-object-entities', function(Y) {
 13     "use strict";
 14 
 15     var STRING = "string", HIDDEN = "hidden", NAME = "name",
 16             HTML = "html", VALUE = "value", HASHLIST = "hashlist",
 17             COMBINE = "combine";
 18 
 19     /**
 20      * ObjectDescriptor mapper
 21      */
 22     Y.Wegas.persistence.ObjectDescriptor = Y.Base.create("ObjectDescriptor", Y.Wegas.persistence.VariableDescriptor, [], {
 23         getProperty: function(player, key) {
 24             return this.getInstance(player).get("properties." + key);
 25         }
 26     }, {
 27         ATTRS: {
 28             "@class": {
 29                 value: "ObjectDescriptor"
 30             },
 31             properties: {
 32                 optional: false,
 33                 _inputex: {
 34                     label: "Descriptor properties",
 35                     _type: HASHLIST,
 36                     useButtons: true,
 37                     keyField: NAME,
 38                     valueField: VALUE,
 39                     elementType: {
 40                         type: COMBINE,
 41                         fields: [{
 42                                 name: NAME,
 43                                 typeInvite: NAME
 44                             }, {
 45                                 name: VALUE,
 46                                 typeInvite: VALUE
 47                             }]
 48                     }
 49                 }
 50             },
 51             defaultInstance: {
 52                 type: "object",
 53                 properties: {
 54                     '@class': {
 55                         type: STRING,
 56                         _inputex: {
 57                             _type: HIDDEN,
 58                             value: 'ObjectInstance'
 59                         }
 60                     },
 61                     properties: {
 62                         optional: false,
 63                         _inputex: {
 64                             label: "Default properties",
 65                             _type: HASHLIST,
 66                             keyField: NAME,
 67                             useButtons: true,
 68                             valueField: VALUE,
 69                             elementType: {
 70                                 type: COMBINE,
 71                                 fields: [{
 72                                         name: NAME,
 73                                         typeInvite: NAME
 74                                     }, {
 75                                         name: VALUE,
 76                                         typeInvite: VALUE
 77                                     }]
 78                             }
 79                         }
 80                     }
 81                 }
 82             },
 83             description: {
 84                 type: STRING,
 85                 format: HTML,
 86                 optional: true
 87             }
 88         },
 89         METHODS: {
 90             size: {
 91                 label: "size",
 92                 returns: "number",
 93                 arguments: [{
 94                         type: "hidden",
 95                         value: "self"
 96                     }]
 97             },
 98             getProperty: {
 99                 label: "property equals",
100                 returns: "string",
101                 arguments: [{
102                         type: "hidden",
103                         value: "self"
104                     }, {
105                         value: "property name",
106                         scriptType: STRING
107                     }]
108             },
109             setProperty: {
110                 label: "set property",
111                 arguments: [{
112                         type: "hidden",
113                         value: "self"
114                     }, {
115                         value: "property name",
116                         scriptType: STRING
117                     }, {
118                         value: "value",
119                         scriptType: STRING
120                     }]
121             }
122         }
123     });

132             },
133             properties: {

136                     label: "Properties",
