 12 YUI.add('wegas-proggame-scriptfiles', function(Y) {
 13     "use strict";
 14     var CONTENTBOX = 'contentBox';
 15     /**
 16      *  @class Display the proggame files
 17      *  @name Y.Plugin.ScriptFiles
 18      *  @extends Y.Widget
 19      *  @constructor
 20      */
 21     var ScriptFiles = Y.Base.create("wegas-proggame-scriptfiles", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 22         handlers: null,
 23         initializer: function() {
 24             this.handlers = {};
 25 
 26             this.publish('openFile', {
 27                 broadcast: true,
 28                 emitFacade: true
 29             });
 30         },
 31         renderUI: function() {
 32             var cb = this.get(CONTENTBOX);
 33             this.addButton = new Y.Wegas.Button({
 34                 label: "Add file"
 35             });
 36             this.addButton.render(cb);
 37             cb.append("<div class='fileName'></div>");
 38 
 39             this.createPanel();
 40         },
 41         bindUI: function() {
 42             this.handlers.sync = Y.Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this);
 43 
 44             this.addButton.on("click", function(e) {
 45 //                this.newFilePanel.show();
 46                 this.createPanel().show();
 47             }, this);
 48         },
 49         syncUI: function() {
 50             var inboxDescriptor = this.get('variable.evaluated'),
 51                     messages = inboxDescriptor.getInstance().get("messages"), node,
 52                     cb = this.get(CONTENTBOX), fileNameDiv = cb.one(".fileName"), i;
 53 
 54             fileNameDiv.get('childNodes').remove();
 55 
 56             if (messages.length === 0) {
 57                 fileNameDiv.append("<p>No files to display</p>");
 58             } else {
 59                 for (i = 0; i < messages.length; i++) {
 60                     node = Y.Node.create('<p>' + messages[i].get("subject") + '</p>');
 61                     node.data = messages[i];
 62                     fileNameDiv.append(node);
 63 
 64                     node.on("click", function(e) {
 65                         //console.log("openFile event");
 66                         this.fire("openFile", {file: e.target.data});
 67                     }, this);
 68                 }
 69             }
 70         },
 71         createPanel: function() {
 72             var node = Y.Node.create('<p><input type="text" name="fileName" placeholder="File name"/></p>'), that = this;
 73             this.newFilePanel = new Y.Wegas.Panel({
 74                 srcNode: node,
 75                 headerContent: 'Add a new file',
 76                 width: 250,
 77                 modal: true,
 78                 centered: true,
 79                 visible: false,
 80                 render: true,
 81                 buttons: [{
 82                         value: 'Add',
 83                         section: Y.WidgetStdMod.FOOTER,
 84                         action: function(e) {
 85                             e.preventDefault();
 86                             this.exit();
 87                             that.addFile(this.get("srcNode").one("input").get("value"));
 88                             this.get("srcNode").one("input").set("value", "");
 89                         }
 90                     }, {
 91                         value: 'Cancel',
 92                         section: Y.WidgetStdMod.FOOTER,
 93                         action: function(e) {
 94                             e.preventDefault();
 95                             this.exit();
 96                         }
 97                     }
 98                 ]
 99             });
100             return this.newFilePanel;
101         },
102         addFile: function(fileName) {
103             Y.Wegas.Facade.VariableDescriptor.sendRequest({
104                 request: "/Script/Run/" + Y.Wegas.app.get('currentPlayer'),
105                 //'Managed-Mode': 'false',
106                 cfg: {
107                     method: "POST",
108                     data: Y.JSON.stringify({
109                         "@class": "Script",
110                         language: "JavaScript",
111                         content: "VariableDescriptorFacade.find(gameModel, 'files').sendMessage(self, '', '" + fileName + ".js', '', []);"
112                     })
113                 },
114                 on: {
115                     success: Y.bind(function(e) {
116                         this.fire("openFile", {file: e.response.entity});
117                     }, this)
118                 }
119             });
120         },
121         destructor: function() {
122             var k;
123             for (k in this.handlers) {
124                 this.handlers[k].detach();
125             }
126             this.addButton.destroy();
127         }
128     }, {
129         ATTRS: {
130             variable: {
131                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
132                 _inputex: {
133                     _type: "variableselect",
134                     label: "variable",
135                     classFilter: ["InboxDescriptor"]
136                 }
137             }
138         }
139     });
140     Y.namespace('Wegas').ScriptFiles = ScriptFiles;
141 });
