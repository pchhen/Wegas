 12 YUI.add('wegas-layout-list', function(Y) {
 13     "use strict";
 14 
 15     var BOUNDINGBOX = 'boundingBox',
 16             CONTENTBOX = 'contentBox',
 17             List;
 18 
 19     /**
 20      * @name Y.Wegas.List
 21      * @extends Y.Widget
 22      * @borrows Y.WidgetParent, Y.WidgetChild, Y.Wegas.Editable, Y.Wegas.Layout
 23      * @class class to serialize widgets
 24      * @constructor
 25      * @description class to serialize widgets
 26      */
 27     List = Y.Base.create("wegas-list", Y.Widget, [Y.WidgetParent, Y.WidgetChild, Y.Wegas.Editable, Y.Wegas.Layout], {
 28         /** @lends Y.Wegas.List# */
 29 
 30         // *** Private fields *** //
 31         // -
 32 
 33         // ** Lifecycle Methods ** /
 34         /**
 35          * @function
 36          * @private
 37          * @description set class of the contentbox (vertical or horizontal)
 38          * add class with "clear:both" style after the contentbox.
 39          */
 40         syncUI: function() {
 41             var cb = this.get(CONTENTBOX);
 42 
 43             if (this.get('direction') === 'vertical') {
 44                 cb.addClass(this.getClassName('vertical'));
 45                 cb.removeClass(this.getClassName('horizontal'));
 46             } else {
 47                 cb.addClass(this.getClassName('horizontal'));
 48                 cb.removeClass(this.getClassName('vertical'));
 49             }
 50             this.get(BOUNDINGBOX).append('<div style="clear:both"></div>');
 51         }
 52     }, {
 53         /** @lends Y.Wegas.List */
 54         EDITORNAME: "Folder",
 55         CSS_PREFIX: "wegas-list",
 56         /**
 57          * @field
 58          * @static
 59          * @description
 60          * <p><strong>Attributes</strong></p>
 61          * <ul>
 62          *    <li>defaultChildType: default value for children. Transient.</li>
 63          *    <li>children: list of widget. Transient.</li>
 64          *    <li>direction: string-helper to add class and set style of the
 65          *     list (vertical or horizontal). Vertical by default</li>
 66          * </ul>
 67          */
 68         ATTRS: {
 69             direction: {
 70                 value: 'vertical',
 71                 type: "string",
 72                 choices: [{
 73                         value: 'vertical'
 74                     }, {
 75                         value: 'horizontal'
 76                     }]
 77             }
 78 
 79             /**
 80              * Prevent widgetchild selection to be propagated through the hierarchy
 81              */
 82             //selected: {
 83             //    value: 2,
 84             //    readonly: true
 85             //}
 86         }
 87 
 88     });
 89     Y.namespace('Wegas').List = List;
 90 
 91 });
