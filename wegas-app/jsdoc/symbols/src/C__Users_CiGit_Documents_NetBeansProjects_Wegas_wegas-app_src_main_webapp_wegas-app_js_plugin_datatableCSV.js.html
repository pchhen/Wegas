 13 YUI.add('datatable-csv', function(Y) {
 14     "use strict";
 15     var DatatableCSV, imgTest = new Image();
 16 
 17     /**
 18      * Plugin to export datatable as CSV.
 19      * @constructor
 20      * @extends Y.Plugin.Base
 21      * @name Y.Wegas.DatatableCSV
 22      */
 23     DatatableCSV = Y.Base.create("datatable-csv", Y.Plugin.Base, [], {
 24         /* @lends Y.Wegas.DatatableCSV# */
 25         /**
 26          * lifecycle method
 27          * @function
 28          * @private
 29          * @returns {undefined}
 30          */
 31         initializer: function() {
 32             this.bcsv = new Y.Button({
 33                 label: "CSV",
 34                 on: {
 35                     click: Y.bind(function() {
 36                         DatatableCSV.dataToWindow(this._toCSV());
 37                     }, this)
 38                 }
 39             }).render(this.get("host").get("boundingBox"));
 40             this.bdcsv = new Y.Button({
 41                 label: "Download CSV",
 42                 on: {
 43                     click: Y.bind(function() {
 44                         var csv = this._toCSV(),
 45                                 //node = this.get("host").get("boundingBox").one(".datatable-csv-file"),
 46                                 gm_name = Y.Wegas.Facade.GameModel.cache.getCurrentGameModel().get("name"),
 47                                 game_name = Y.Wegas.Facade.Game.cache.getCurrentGame().get("name"),
 48                                 name = gm_name + "-" + game_name + "-" + Y.Lang.now() + ".csv";
 49 
 50                         DatatableCSV.download("text/csv", name, csv);
 51                         //Disabled due to IE allowing limited data URI
 52                         //if (DatatableCSV.DATA_URI_SUPPORT) {
 53                         //    node.setAttribute("href", "data:text/csv;header=true," + encodeURIComponent(csv));
 54                         //    node.setAttribute("download", name);
 55                         //} else {
 56                         //    //FALLBACK : get a real browser
 57                         //    download("text/csv", name, csv);
 58                         //}
 59                     }, this)
 60                 }
 61             }).render(this.get("host").get("boundingBox"));
 62         },
 63         /**
 64          * Transformation method, datatable to CSV
 65          * @function
 66          * @private
 67          * @returns {undefined}
 68          */
 69         _toCSV: function() {
 70             var headers = Y.Array.map(this.get("host").get("columns"), function(i) {
 71                 return "\"" + (i.label || i.key) + "\"";
 72             }), keys = Y.Array.map(this.get("host").get("columns"), function(i) {
 73                 return i.key;
 74             }),
 75                     data = this.get("host").data.toArray(), records = [headers], i, j, fields;
 76             //Build lignes following key order.
 77             for (i = 0; i < data.length; i += 1) {
 78                 fields = [];
 79                 for (j = 0; j < keys.length; j += 1) {
 80                     fields.push("\"" + (data[i].get(keys[j]) || "") + "\"");
 81                 }
 82                 records.push(fields);
 83 
 84             }
 85             return DatatableCSV.arraysToCSV(records);
 86 
 87         },
 88         /**
 89          * lifecycle method
 90          * @function
 91          * @private
 92          * @returns {undefined}
 93          */
 94         destructor: function() {
 95             if (this.bcsv){
 96                 this.bcsv.destroy();
 97             }
 98             if (this.bdcsv){
 99                 this.bdcsv.destroy();
100             }
101         }
102     }, {
103         /* @lends Y.Wegas.DatatableCSV */
104         NS: "datatablecsv",
105         /**
106          * Field delimiter. As of RFC 4180 : "," by default
107          * @field
108          * @static
109          */
110         FIELD_DELIMITER: ",",
111         /**
112          * Record (lines) delimiter. As of RFC 4180 : "\r\n" (CRLF) by default
113          * @static
114          * @field
115          */
116         RECORD_DELIMITER: "\r\n",
117         /**
118          * Writes given data to window
119          * @static
120          * @function
121          * @param {contentType} data to write to window
122          * @param {contentType} winHandle (optional) window handle to write to. A new window is create if this parameter is missing.
123          * @returns {undefined}
124          */
125         dataToWindow: function(data, winHandle) {
126             var win = winHandle || window.open(), range, selection;
127             win.document.write("<pre>");
128             win.document.write("</pre>");
129             Y.one(win.document).one("pre").set("text", data);
130             //Select data
131             if (document.createRange && window.getSelection) {
132                 range = win.document.createRange();
133                 selection = win.getSelection();
134                 selection.removeAllRanges();
135                 range.selectNodeContents(win.document.body.firstChild);
136                 selection.addRange(range);
137             }
138         },
139         /**
140          * Convert array of arrays to CSV
141          * @private
142          * @param {Array} arrays array containing arrays [[], []]
143          * @returns {String} CSV string
144          */
145         arraysToCSV: function(arrays) {
146             return Y.Array.map(arrays, function(i) {
147                 return i.join(DatatableCSV.FIELD_DELIMITER);
148             }).join(DatatableCSV.RECORD_DELIMITER);
149         },
150         /**
151          * Send data to server to download it.
152          * @function
153          * @private
154          * @param {String} contentType file's content-type
155          * @param {String} name file's name
156          * @param {String} data file's content
157          * @returns {undefined}
158          */
159         download: function(contentType, name, data) {
160             var url = Y.Wegas.app.get("base") + "rest/Download/" + name,
161                     form = Y.Node.create('<form enctype="multipart/form-data" method="post" action="' + url + '" ><input type="hidden" name="data"><input type="hidden" name="ctype"></form>');
162             form.one("input[name=data]").getDOMNode().value = data;
163             form.one("input[name=ctype]").getDOMNode().value = contentType;
164             form.submit();
165             form.destroy();
166         },
167         DATA_URI_SUPPORT: undefined
168     });
169     Y.namespace("Wegas").DatatableCSV = DatatableCSV;
170 
171     imgTest.onload = imgTest.onerror = function() {
172         if (imgTest.width === 1 && imgTest.height === 1) {
173             DatatableCSV.DATA_URI_SUPPORT = true;
174         } else {
175             DatatableCSV.DATA_URI_SUPPORT = false;
176         }
177         imgTest = null;
178     };
179     imgTest.src = "data:image/gif;base64,R0lGODlhAQABAAD/ACwAAAAAAQABAAACADs=";
180 
181 });
