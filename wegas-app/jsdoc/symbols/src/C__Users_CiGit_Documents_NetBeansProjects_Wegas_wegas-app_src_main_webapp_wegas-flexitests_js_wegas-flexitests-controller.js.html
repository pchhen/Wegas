 13 YUI.add("wegas-flexitests-controller", function(Y) {
 14     "use strict";
 15     Y.Wegas.FlexitestsController = Y.Base.create("wegas-flexitests-controller", Y.Wegas.AbsoluteLayout, [Y.Wegas.Editable, Y.Wegas.Layout], {
 16         BOUNDING_TEMPLATE: "<div><div class='flexi-mask'></div></div>",
 17         /**
 18          * Lifecycle method
 19          * @function
 20          * @private
 21          * @returns {undefined}
 22          */
 23         initializer: function() {
 24             this.leftElement = null;
 25             this.rightElement = null;
 26             this.centerElement = null;
 27             this.mcq = null;
 28             this.maxSize = 0;
 29             this.ongoing = false;
 30             this.config = {};
 31             this.currentQuestionId = -1;
 32             this.questionToDo = [];
 33             this.startTime = null;
 34             this.events = [];
 35             this.publish("visibility-timer:restart", {
 36                 broadcast: 1
 37             });
 38             //this.get("contentBox").hide();
 39         },
 40         /**
 41          * Lifecycle method
 42          * @function
 43          * @private
 44          * @returns {undefined}
 45          */
 46         bindUI: function() {
 47             this.constructor.superclass.bindUI.apply(this);
 48             this.set("tabIndex", -1);
 49             this.after("*:currentLoadingChange", function(e) {
 50                 var i, noready = false;
 51                 for (i in e.newVal) {
 52                     if (e.newVal.hasOwnProperty(i)) {
 53                         noready = (noready || e.newVal[i]);
 54                         if (noready) {
 55                             break;
 56                         }
 57                     }
 58                 }
 59                 if (!noready) {
 60                     this.get("contentBox").show();
 61                     this.startStimuli();
 62                 }
 63             });
 64             this.on("*:clientResponse", function(e) {
 65                 if (this.ongoing) {
 66                     this.responseGiven(e.value);
 67                 }
 68             });
 69         },
 70         /**
 71          * Lifecycle method
 72          * @function
 73          * @private
 74          * @returns {undefined}
 75          */
 76         syncUI: function() {
 77             this.mcq = this.getChildById("flexi-mcq");
 78 
 79             var targetDescriptor = this.mcq.get("variable.evaluated"),
 80                     props, exist = false, i, j, done = 0;
 81 
 82             if (!targetDescriptor) {
 83                 this.showMessage("Unable to find target variable");
 84                 return;
 85             }
 86 
 87             this.mask();
 88             this.leftElement = this.getChildById("leftElement");
 89             this.rightElement = this.getChildById("rightElement");
 90             this.centerElement = this.getChildById("centerElement");
 91             this.fixPoint = this.get("contentBox").all(".fix-point");
 92             this.fixPoint.hide();
 93             props = targetDescriptor.getInstance().get("properties");
 94             this.maxSize = Math.max(this.leftElement.size(), this.rightElement.size(), this.centerElement.size());
 95             for (i = 0; i < this.maxSize; i += 1) {
 96                 exist = false;
 97                 for (j in props) {
 98                     if (props.hasOwnProperty(j)) {
 99                         if (+(Y.JSON.parse(props[j]).id) === +i) {
100                             exist = true;
101                             done += 1;
102                             break;
103                         }
104                     }
105                 }
106                 if (!exist) {
107                     this.questionToDo[i - done] = i;
108                 }
109             }
110             this.onceAfter("render", this.next, this);
111         },
112         responseGiven: function(response) {
113             var responseTime = Y.Lang.now() - this.startTime,
114                     reponseElement,
115                     elements = this.collectElements();
116             this.ongoing = false;
117             //this.mask();
118             elements.index = +this.maxSize - this.questionToDo.length;
119             elements.id = this.currentQuestionId;
120             elements.response = response;
121             elements.delay = responseTime;
122             reponseElement = this.centerElement.getActiveElement().flexiresponse;
123 
124             if (reponseElement instanceof Y.Plugin.FlexiResponse &&
125                     reponseElement.get("value") === response) {
126                 elements.valid = true;
127                 this.mcq.success(responseTime);
128             } else {
129                 elements.valid = false;
130                 this.mcq.error(responseTime);
131             }
132             this.mcq.save(elements);
133 
134             Y.later(+this.mcq.get("feedback"), this, this.next);
135         },
136         next: function() {
137             var onSuccess = Y.bind(function() {
138                 Y.later(+this.get("fixPoint"), this, this.createLoadingEvent);
139                 this.fixPoint.show();
140             }, this);
141             if (this.questionToDo.length === 0) {
142                 this.runEndScript();
143                 this.unmask();
144                 return;
145             }
146             this.mask();
147             this.set("currentLoading", {"left": true, "center": true, "right": true});
148             this.currentQuestionId = this.generateNextId();
149             this.centerElement.set("element", +this.currentQuestionId);
150             this.leftElement.set("element", +this.currentQuestionId);
151             this.rightElement.set("element", +this.currentQuestionId);
152             if (this.get("popupAfter") > 0 &&
153                     (this.maxSize - this.questionToDo.length) !== 1 &&
154                     ((this.maxSize - this.questionToDo.length - 1) % this.get("popupAfter")) === 0) {
155                 this.get("boundingBox").emitDOMMessage("showPopup", {
156                     content: this.get("popupContent"),
157                     buttons: [{
158                             label: "Ok",
159                             action: function() {
160                                 this.exit();
161                                 onSuccess();
162                             }
163                         }]
164                 });
165             } else {
166                 onSuccess();
167             }
168 
169         },
170         collectElements: function() {
171             var elements = {},
172                     swaped = this.swapzone ? this.swapzone.swaped : false,
173                     left = this.leftElement.getActiveElement(),
174                     center = this.centerElement.getActiveElement(),
175                     right = this.rightElement.getActiveElement();
176             //if (swaped) {
177             //    elements.left = right.get("content") || right.get("url");
178             //    elements.right = left.get("content") || left.get("url");
179             //} else {
180             //    elements.left = left.get("content") || left.get("url");
181             //    elements.right = right.get("content") || right.get("url");
182             //}
183             //elements.center = center.get("content") || center.get("url");
184             /*Extract elements in results*/
185             if (swaped) {
186                 elements.left = "right";
187                 elements.right = "left";
188             } else {
189                 elements.left = "left";
190                 elements.right = "right";
191             }
192 
193             return elements;
194         },
195         /**
196          * Check elements to be loaded if they exist
197          * @returns {undefined}
198          */
199         createLoadingEvent: function() {
200             if (this.centerElement.getActiveElement()) {
201                 this.centerElement.getActiveElement().onceAfter("render", function(e) {
202                     this.set("currentLoading.center", false);
203                 }, this);
204             } else {
205                 this.set("currentLoading.center", false);
206             }
207             if (this.leftElement.getActiveElement()) {
208                 this.leftElement.getActiveElement().onceAfter("render", function(e) {
209                     this.set("currentLoading.left", false);
210                 }, this);
211             } else {
212                 this.set("currentLoading.left", false);
213             }
214             if (this.rightElement.getActiveElement()) {
215                 this.rightElement.getActiveElement().onceAfter("render", function(e) {
216                     this.set("currentLoading.right", false);
217                 }, this);
218             } else {
219                 this.set("currentLoading.right", false);
220             }
221         },
222         generateNextId: function() {
223             return this.get("random") ?
224                     this.questionToDo.splice(Math.round(Math.random() * (this.questionToDo.length - 1)), 1)[0] :
225                     this.questionToDo.shift();
226         },
227         startStimuli: function() {
228             this.get("boundingBox").focus();
229             this.fire("visibility-timer:restart");
230             Y.soon(Y.bind(this.unmask, this));
231             this.ongoing = true;
232         },
233         runEndScript: function() {
234             Y.Wegas.Facade.VariableDescriptor.sendRequest({
235                 request: "/Script/Run/" + Y.Wegas.app.get('currentPlayer'),
236                 cfg: {
237                     method: "POST",
238                     data: Y.JSON.stringify(this.get("endImpact"))
239                 },
240                 on: {
241                     failure: Y.bind(function(e) {
242                         Y.log("error", "Failed to store data", "Y.Wegas.FlexitestsController");
243                     }, this)
244                 }
245             });
246         },
247         mask: function() {
248             this.get("boundingBox").one(".flexi-mask").show();
249         },
250         unmask: function() {
251             this.fixPoint.hide();
252             this.get("boundingBox").one(".flexi-mask").hide();
253             this.startTime = Y.Lang.now();
254         },
255         getChildById: function(id) {
256             var returnItem = null;
257             this.some(function(item) {
258                 if (item.get("id") === id) {
259                     returnItem = item;
260                     return true;
261                 }
262             });
263             return returnItem;
264         },
265         toObject: Y.Wegas.AbsoluteLayout.prototype.toObject,
266         /**
267          * Lifecycle method
268          * @function
269          * @private
270          * @returns {undefined}
271          */
272         destructor: function() {
273             var i;
274             this.constructor.superclass.destructor.apply(this);
275             for (i = 0; i < this.events.length; i += 1) {
276                 this.events[i].detach();
277             }
278         }
279     }, {
280         EDITORNAME: "Flexitests Controller",
281         ATTRS: {
282             currentLoading: {
283                 value: {},
284                 "transient": true
285             },
286             fixPoint: {
287                 value: 2000,
288                 type: "number",
289                 _inputex: {
290                     label: "Show fix-point (ms)",
291                     description: "Time to wait before next question"
292                 }
293             },
294             random: {
295                 value: true,
296                 type: "boolean",
297                 _inputex: {
298                     label: "Play lists randomly"
299                 }
300             },
301             popupAfter: {
302                 value: 0,
303                 type: "number",
304                 _inputex: {
305                     label: "Show popup message",
306                     description: "After # question(s).0 or less to disable"
307                 }
308             },
309             popupContent: {
310                 value: "",
311                 type: "string",
312                 optional: true,
313                 format: "html",
314                 _inputex: {
315                     label: "Popup content"
316                 }
317             },
318             endImpact: {
319                 label: "End impact",
320                 _inputex: {
321                     _type: "script",
322                     description: "What should happen once there is no more questions",
323                     optional: true
324                 }
325             }
326         }
327     });
328     /*Inject specific plugins*/
329     Y.Wegas.FlexitestsController.ATTRS.plugins = Y.clone(Y.Wegas.FlexitestsController.ATTRS.plugins);
330     Y.Wegas.FlexitestsController.ATTRS.plugins._inputex.items.push({
331         type: "Button",
332         label: "Swap top zones",
333         data: "SwapZone"
334     });
335     Y.Plugin.FlexiResponse = Y.Base.create("wegas-flexi-response", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
336     }, {
337         NS: "flexiresponse",
338         NAME: "flexiresponse",
339         EDITORNAME: "Flexitest Response",
340         ATTRS: {
341             "value": {
342                 value: "",
343                 type: "string",
344                 _inputex: {
345                     label: "Response value",
346                     description: "Needed to match with user input"
347                 }
348             }
349         }
350     });
351     Y.Plugin.SwapZone = Y.Base.create("wegas-flexi-swapzone", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
352         initializer: function() {
353             if (!(this.get("host") instanceof Y.Wegas.FlexitestsController)) {
354                 return;
355             }
356             this.swaped = false;
357             this.afterHostMethod("next", function() {
358                 var current = this.get("host").maxSize - this.get("host").questionToDo.length;
359                 if (current === 1 && Math.random() > 0.5) {
360                     this.swap();
361                 }
362                 if (current % this.get("after") === 0) {
363                     this.swap();
364                 }
365             });
366         },
367         swap: function() {
368             this.get("host").leftElement.get("contentBox").swap(this.get("host").rightElement.get("contentBox"));
369             this.swaped = !this.swaped;
370         },
371         destructor: function() {
372             if (this.swaped) {
373                 this.swap();
374             }
375         }
376     }, {
377         NS: "swapzone",
378         NAME: "swapzone",
379         EDITORNAME: "Flexitest swap zone",
380         ATTRS: {
381             "after": {
382                 value: 1,
383                 type: "number",
384                 _inputex: {
385                     label: "Swap after",
386                     description: "Top zone swap after # question(s)"
387                 },
388                 setter: function(v) {
389                     return +v > 0 ? +v : 1;
390                 }
391             }
392         }
393     });
394 });
