 12 YUI.add("wegas-inputex-rte", function(Y) {
 13     "use strict";
 14 
 15     var inputEx = Y.inputEx, RTEField;
 16 
 17     /**
 18      * @class Wrapper for the Rich Text Editor from YUI
 19      * @name Y.inputEx.Wegas.RTEField
 20      * @extends Y.inputEx.Textarea
 21      * @constructor
 22      * @param {Object} options
 23      */
 24     RTEField = function(options) {
 25         RTEField.superclass.constructor.call(this, options);
 26     };
 27 
 28     Y.extend(RTEField, inputEx.Textarea, {
 29         destroy: function() {
 30             tinymce.execCommand('mceRemoveEditor', false, this.el.id);
 31             RTEField.superclass.destroy.call(this);
 32         },
 33         /**
 34          * Set the default values of the options
 35          * @param {Object} options Options object as passed to the constructor
 36          */
 37         setOptions: function(options) {
 38             RTEField.superclass.setOptions.call(this, options);
 39 
 40             this.options.opts = options.opts || {};
 41             this.options.typeInvite = null;
 42         },
 43         /**
 44          * Render the field using the YUI Editor widget
 45          */
 46         renderComponent: function() {
 47             RTEField.superclass.renderComponent.call(this);
 48             if (!RTEField.init) {
 49                 RTEField.init = true;
 50                 tinymce.init({
 51                     plugins: [
 52                         "autolink autoresize link image lists ",
 53                         "code media table contextmenu paste"
 54                                 //textcolor
 55                                 //wordcount autosave advlist charmap print preview hr anchor pagebreak spellchecker directionality
 56                     ],
 57                     toolbar1: "bold italic bullist | link image media | code",
 58                     // formatselect removeformat underline unlink forecolor backcolor anchor previewfontselect fontsizeselect styleselectspellchecker template
 59                     // contextmenu: "link image inserttable | cell row column deletetable | formatselect forecolor",
 60                     menubar: false,
 61                     statusbar: false,
 62                     relative_urls: false,
 63                     toolbar_items_size: 'small',
 64                     file_browser_callback: this.onFileBrowserClick,
 65                     image_advtab: true,
 66                     autoresize_min_height: 50,
 67                     autoresize_max_height: 500,
 68                     content_css: [
 69                         // "http://yui.yahooapis.com/combo?3.12.0/build/cssreset/cssreset-min.css&3.12.0/build/cssfonts/cssfonts-min.css&3.12.0/build/cssgrids/cssgrids-min.css",
 70                         // Y.Wegas.app.get("base") + "wegas-app/css/wegas-app.css"
 71                         Y.Wegas.app.get("base") + "wegas-editor/css/wegas-inputex-rte.css"
 72                     ],
 73                     style_formats: [{// Style formats
 74                             title: 'Title 1',
 75                             block: 'h1'
 76                         }, {
 77                             title: 'Title 2',
 78                             block: 'h2'
 79                                     // styles : {
 80                                     //    color : '#ff0000'
 81                                     // }
 82                         }, {
 83                             title: 'Title 3',
 84                             block: 'h3'
 85                         }, {
 86                             title: 'Normal',
 87                             inline: 'span'
 88                         }]});
 89             }
 90             Y.once("domready", function() {
 91                 tinymce.execCommand('mceAddEditor', false, this.el.id);
 92             }, this);
 93         },
 94         onFileBrowserClick: function(field_name, url, type, win) {
 95             RTEField.filePanel = new Y.Wegas.FileSelect();
 96 
 97             RTEField.filePanel.on("*:fileSelected", function(e, path) {
 98                 e.stopImmediatePropagation();
 99                 e.preventDefault();
100                 RTEField.filePanel.hide();
101 
102                 var win = RTEField.filePanel.win,
103                         field_name = RTEField.filePanel.field_name,
104                         targetInput = win.document.getElementById(field_name);
105                 targetInput.value = Y.Plugin.CRDataSource.getFullpath(path);    // update the input field
106 
107                 if (typeof (win.ImageDialog) !== "undefined") {                 // are we an image browser
108                     if (win.ImageDialog.getImageData) {                         // we are, so update image dimensions...
109                         win.ImageDialog.getImageData();
110                     }
111 
112                     if (win.ImageDialog.showPreviewImage) {                     // ... and preview if necessary
113                         win.ImageDialog.showPreviewImage(Y.Plugin.CRDataSource.getFullpath(path));
114                     }
115                 }
116                 if (win.Media) {                                                // If in an editor window
117                     win.Media.formToData("src");                                // update the data
118                 }
119             });
120             RTEField.filePanel.win = win;
121             RTEField.filePanel.field_name = field_name;
122             return false;
123         },
124         /**
125          * Set the html content
126          * @param {String} value The html string
127          * @param {boolean} [sendUpdatedEvt] (optional) Wether this setValue should fire the 'updated' event or not (default is true, pass false to NOT send the event)
128          */
129         setValue: function(value) {
130             var tmceI = tinyMCE.get(this.el.id);
131 
132             if (value) {
133                 value = value.replace(
134                         new RegExp("data-file=\"([^\"]*)\"", "gi"),
135                         "src=\"" + Y.Plugin.CRDataSource.getFullpath("") + "$1\""
136                         + " href=\"" + Y.Plugin.CRDataSource.getFullpath("") + "$1\"");// @hack Place both href and src so it will work for both <a> and <img> elements
137             }
138             RTEField.superclass.setValue.call(this, value);
139 
140             if (tmceI) {
141                 tmceI.setContent(value);
142             }
143         },
144         /**
145          * Get the html string
146          * @return {String} the html string
147          */
148         getValue: function() {
149             //var path = Y.Plugin.CRDataSource.getFullpath("")
150             var reg = new RegExp("((src|href)=\".*/rest/File/GameModelId/.*/read([^\"]*)\")", "gi"); // Replace absolute path with injector style path
151             tinyMCE.triggerSave();
152             //return RTEField.superclass.getValue.call(this).replace(reg, "data-file=\"$3\" $1");
153             return RTEField.superclass.getValue.call(this).replace(reg, "data-file=\"$3\"");
154         },
155         /**
156          * @static
157          */
158         filePanel: null
159     });
160 
161     inputEx.registerType("html", RTEField, []);                                 // Register this class as "html" type
162 });
