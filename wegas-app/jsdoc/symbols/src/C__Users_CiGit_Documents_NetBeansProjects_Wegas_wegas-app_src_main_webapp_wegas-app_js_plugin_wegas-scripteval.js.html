  8 YUI.add('wegas-scripteval', function(Y) {
  9     "use strict";
 10     var ScriptEval, VariableDescriptorFacade;
 11 
 12     function buildItems(entity, acc) {                                  // Recursively build items lists
 13         var j, items;
 14         if (entity instanceof Y.Wegas.persistence.ListDescriptor) {
 15             items = entity.get("items");
 16             acc.items = [];
 17             for (j in items) {
 18                 if (items.hasOwnProperty(j)) {
 19                     acc.items.push(Y.JSON.parse(Y.JSON.stringify(items[j].getInstance())));
 20                     buildItems(items[j], acc.items[acc.items.length - 1]);
 21                 }
 22             }
 23         }
 24     }
 25 
 26     ScriptEval = Y.Base.create("ScriptEval", Y.Plugin.Base, [], {
 27         context: null,
 28         upToDate: false,
 29         initializer: function() {
 30             this.context = {};
 31             this.upToDate = false;
 32             this.afterHostEvent("response", function(e) {
 33                 this.upToDate = false;
 34             }, this);
 35             this.publish("evaluated");
 36             this.publish("failure");
 37         },
 38         /**
 39          *  A localEval with server fallback.
 40          *
 41          *  @param script The script to evaluate
 42          *  @param cb A callback object, containing success, failure function or just a function as success callback. First parameter passed will be result
 43          */
 44         eval: function(script, cb) {
 45             var result, url;
 46             if (cb instanceof Function) {
 47                 cb = {
 48                     success: cb
 49                 };
 50             }
 51             try {
 52                 result = this.localEval(script);
 53                 this.fire("evaluated", result);
 54                 if (cb && cb.success instanceof Function) {
 55                     cb.success(result);
 56                 }
 57             } catch (error) {
 58                 url = Y.Wegas.Facade.VariableDescriptor.get("source") + "/Script/Run/" + Y.Wegas.app.get('currentPlayer');
 59                 Y.io(url, {
 60                     headers: {
 61                         'Content-Type': 'application/json; charset=iso-8859-1',
 62                         'Managed-Mode': 'false'
 63                     },
 64                     sync: false,
 65                     method: "POST",
 66                     data: Y.JSON.stringify({
 67                         "@class": "Script",
 68                         "language": "JavaScript",
 69                         "content": script
 70                     }),
 71                     on: {
 72                         success: Y.bind(function(id, req_id, response) {
 73                             if (cb && cb.success instanceof Function) {
 74                                 cb.success(Y.JSON.parse(response.responseText));
 75                             }
 76                         }, this),
 77                         failure: Y.bind(function(id, req_id, response) {
 78                             var result;
 79                             try {
 80                                 result = Y.JSON.parse(response.responseText);
 81                             } catch (e) {
 82                                 result = null;
 83                             }
 84                             if (cb && cb.failure instanceof Function) {
 85                                 cb.failure(result);
 86                             }
 87                         }, this)
 88                     }
 89                 });
 90             }
 91         },
 92         /**
 93          * Tries to evaluate the script locally, using variables cache
 94          * @param {String} script The script to eval localy
 95          * @return {Any} value locally evaluated
 96          */
 97         localEval: function(script) {
 98             /*jslint evil: true */
 99             if (!this.upToDate) {                                               //Only compute if new value
100                 this._buildContext();
101             }
102             return (new Function("with(this) { return " + script + ";}")).call(this.context);
103         },
104         /**
105          * @function
106          * @private
107          * @returns {undefined}
108          */
109         _buildContext: function() {
110             var i, data = this.get("host").data;
111             this.upToDate = true;
112             this.context = {};
113 
114             for (i in data) {
115                 if (data.hasOwnProperty(i)) {
116                     this.context[data[i].get('name')] = Y.JSON.parse(Y.JSON.stringify(data[i].getInstance()));
117                     buildItems(data[i], this.context[data[i].get('name')]);
118 
119 //                if (data[i] instanceof Y.Wegas.persistence.ListDescriptor) {
120 //                    this.context[data[i].get('name')].items = [];
121 //                    for (j in data[i].get("items")) {
122 //                        this.context[data[i].get('name')].items.push(JSON.parse(JSON.stringify(data[i].get("items")[j].getInstance())));
123 //                    }
124 //                }
125                 }
126             }
127             /*SANDBOX*/
128             Y.mix(this.context, {
129                 window: undefined,
130                 Y: undefined,
131                 YUI: undefined
132             });
133             /*Extend functionalities (mirror server)*/
134             Y.mix(this.context, {
135                 VariableDescriptorFacade: VariableDescriptorFacade,
136                 self: Y.Wegas.Facade.Game.cache.getCurrentPlayer(),
137                 gameModel: Y.Wegas.Facade.GameModel.cache.getCurrentGameModel()
138             });
139         }
140 
141     }, {
142         NS: "script",
143         NAME: "scriptEval"
144     });
145     VariableDescriptorFacade = {
146         find: function(gameModel, name) {
147             return Y.Wegas.Facade.VariableDescriptor.cache.find("name", name);
148         }
149     };
150     Y.namespace('Plugin').ScriptEval = ScriptEval;
151 
152 });
