 12 YUI.add('wegas-action', function(Y) {
 13     "use strict";
 14 
 15     var HOST = "host", Plugin = Y.Plugin, Wegas = Y.namespace("Wegas");
 16 
 17     "use strict";
 18 
 19     /**
 20      *  @name Y.Wegas.Plugin
 21      *  @class Extension that adds editable capacities to plugins
 22      *  @extends Y.Plugin
 23      *  @constructor
 24      */
 25     function WPlugin() {}
 26     Y.mix(WPlugin.prototype, {});
 27     Y.mix(WPlugin, {
 28         ATTRS: {
 29             host: {
 30                 "transient": true
 31             },
 32             initialized: {
 33                 "transient": true
 34             },
 35             destroyed: {
 36                 "transient": true
 37             }
 38         }
 39     });
 40     Wegas.Plugin = WPlugin;
 41 
 42     /**
 43      *  @name Y.Plugin.Action
 44      *  @extends Y.Plugin.Base
 45      *  @augments Y.Wegas.Plugin
 46      *  @augments Y.Wegas.Editable
 47      *  @class
 48      *  @constructor
 49      */
 50     var Action = Y.Base.create("wegas-actionplugin", Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 51         /** @lends Y.Plugin.Action */
 52 
 53         /**
 54          * @function
 55          * @private
 56          */
 57         initializer: function() {
 58             this.onHostEvent(this.get("targetEvent"), this.execute, this);
 59         },
 60 
 61         /**
 62          * @function
 63          * @protected
 64          */
 65         execute: function() {
 66             Y.error("Y.Plugin.Action.execute() is abstract, should be overriddent");
 67         }
 68     }, {
 69         NS: "wegas",
 70         NAME: "Action",
 71         ATTRS: {
 72             targetEvent: {
 73                 value: "click"
 74             }
 75         }
 76     });
 77     Plugin.Action = Action;
 78 
 79     /**
 80      *  @class
 81      *  @name Y.Plugin.OpenGameAction
 82      *  @extends Y.Plugin.Action
 83      *  @constructor
 84      */
 85     var OpenUrlAction = function() {
 86         OpenUrlAction.superclass.constructor.apply(this, arguments);
 87     };
 88 
 89     Y.extend(OpenUrlAction, Action, {
 90 
 91         execute: function() {
 92             var targetUrl = Wegas.app.get("base") + this.get("url");
 93 
 94             if (this.get("target") === "blank") {
 95                 window.open(targetUrl);
 96             } else {
 97                 window.location.href = targetUrl;
 98             }
 99         }
100     }, {
101         NS: "openurlaction",
102         NAME: "OpenUrlAction",
103         ATTRS: {
104             url: {},
105             /**
106              * Can be "self" or "blank"
107              */
108             target: {
109                 value: "blank"
110             }
111         }
112     });
113     Plugin.OpenUrlAction = OpenUrlAction;
114 
115 
116     /**
117      *  @class
118      *  @name Y.Plugin.OpenPageAction
119      *  @extends Y.Plugin.Action
120      *  @module Wegas
121      *  @constructor
122      */
123     var OpenPageAction = function() {
124         OpenPageAction.superclass.constructor.apply(this, arguments);
125     };
126     Y.extend(OpenPageAction, Action, {
127 
128         initializer: function() {
129             OpenPageAction.superclass.initializer.apply(this, arguments);
130             this.afterHostEvent("render", function() {
131                 var targetPageLoader = Wegas.PageLoader.find(this.get('targetPageLoaderId'));
132                 if (targetPageLoader.get("pageId") === this.get("subpageId")) {
133                     this.get(HOST).set("selected", 1);
134                 }
135             }, this);
136         },
137 
138         execute: function() {
139             var targetPageLoader = Wegas.PageLoader.find(this.get('targetPageLoaderId'));
140             targetPageLoader.set("pageId", this.get("subpageId"));
141             this.get(HOST).set("selected", 1);
142         }
143     }, {
144         NS: "OpenPageAction",
145         NAME: "OpenPageAction",
146         ATTRS: {
147             subpageId: {
148                 type: "string",
149                 _inputex: {
150                     label: "Page to display",
151                     _type: "pageselect",
152                     required: false
153                 }
154             },
155             targetPageLoaderId: {
156                 type: "string",
157                 _inputex: {
158                     label: "Target page loader"
159                 }
160             }
161         }
162     });
163     Plugin.OpenPageAction = OpenPageAction;
164 
165     /**
166      *  @class
167      *  @name Y.Plugin.PopupPlg
168      *  @extends Y.Plugin.Action
169      *  @constructor
170      */
171     var ExecuteScriptAction = function() {
172         ExecuteScriptAction.superclass.constructor.apply(this, arguments);
173     };
174     Y.extend(ExecuteScriptAction, Action, {
175 
176         execute: function() {
177             var host = this.get(HOST), overlayGuest, guest = host.get("root");
178             if (guest.showOverlay && guest.hideOverlay) {
179                 overlayGuest = guest;
180                 overlayGuest.showOverlay();
181             }
182 
183             Wegas.VariableDescriptorFacade.rest.sendRequest({
184                 request: "/Script/Run/" + Wegas.app.get('currentPlayer'),
185                 cfg: {
186                     method: "POST",
187                     data: Y.JSON.stringify(this.get("onClick"))
188                 },
189                 on: {
190                     success: function(r) {
191                         if (overlayGuest) {
192                             overlayGuest.hideOverlay();
193                         }
194                     },
195                     failure: function(r) {
196                         if (overlayGuest) {
197                             overlayGuest.hideOverlay();
198                         }
199                         Y.bind(host.defaultExceptionHandler, host, r);
200                     }
201                 }
202             });
203         }
204     }, {
205         NS: "ExecuteScriptAction",
206         NAME: "ExecuteScriptAction",
207         ATTRS: {
208             onClick: {
209                 _inputex: {
210                     _type: "script",
211                     label: "On click"
212                 }
213             }
214         }
215     });
216     Plugin.ExecuteScriptAction = ExecuteScriptAction;
217 
218     /**
219      *  @class Show a message when the host widget is rendered, useful for welcome
220      *  messages
221      *  @name Y.Plugin.PopupPlg
222      *  @extends Y.Plugin.Base
223      *  @constructor
224      */
225     var PopupPlg = function() {
226         PopupPlg.superclass.constructor.apply(this, arguments);
227     };
228     Y.extend(PopupPlg, Plugin.Base, {
229 
230         initializer: function() {
231             this.afterHostEvent("render", function() {
232                 this.get(HOST).showMessage("info", this.get("content"));
233             });
234         }
235     }, {
236         NS: "PopupPlg",
237         NAME: "PopupPlg",
238         ATTRS: {
239             content: {
240                 type: "string",
241                 format: "text"
242             }
243         }
244     });
245     Plugin.PopupPlg = PopupPlg;
246 
247 });
