 14 YUI.add("wegas-template", function(Y) {
 15     "use strict";
 16     var TEMPLATE = {
 17         TEXT: "<div class='wegas-template-text'><span><%= this.label || '{label}' %></span><br/><span><%= this.value || '{value}' %></span></div>",
 18         BOX: "<div class='wegas-template-box'><label><%= this.label || '{label}'%></label><br/><span><% for(var i=0; i < this.value; i+=1){%>" +
 19                 "<div class='wegas-template-box-unit'></div><% } %></span><br/>" +
 20                 "<span class='wegas-template-box-value'>(<%= this.value || '{value}' %><% if(this.defaultValue != ''){ %><%= '/' + (this.defaultValue || '{defaultValue}') %><% } %>)</span></div>",
 21         VALUEBOX: "<div class='wegas-template-valuebox'><label><%= this.label || '{label}'%></label</div><br/></label><div class='wegas-template-valuebox-units'><% for(var i=+this.minValue; i < +this.maxValue + 1; i+=1){%>" +
 22                 "<div class='wegas-template-valuebox-unit <%= +i === +this.value ? ' wegas-template-valuebox-selected' : '' %>'><%= ''+i %></div><% } %></span>" +
 23                 "</div></div>",
 24         TITLE: "<div class='wegas-template-title'><%= this.label || '{label}'%></div>",
 25         FRACTION: "<div class='wegas-template-fraction'><%= (this.minValue || '{minValue}') + '/' + (this.value || '{label}') + '/' + (this.maxValue || '{maxValue}') %></div>"
 26     },
 27     engine = new Y.Template(Y.Template.Micro),
 28             undefinedToEmpty = function(value) {
 29         return Y.Lang.isUndefined(value) ? "" : "" + value;
 30     };
 31     /**
 32      * @name Y.Wegas.Template
 33      * @extends Y.Widget
 34      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 35      * @class
 36      * @constructor
 37      * @description  Display  Wegas variables instance (or/and descriptor) under
 38      * specifique templates : text, title, box, fraction and valuebox.
 39      * It is also possible to create a custom template.
 40      */
 41     Y.namespace("Wegas").Template = Y.Base.create("wegas-template", Y.Widget,
 42             [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 43         /*@lends Y.Wegas.Template#*/
 44         TEMPLATES: {
 45             text: engine.compile(TEMPLATE.TEXT),
 46             box: engine.compile(TEMPLATE.BOX),
 47             valuebox: engine.compile(TEMPLATE.VALUEBOX),
 48             title: engine.compile(TEMPLATE.TITLE),
 49             fraction: engine.compile(TEMPLATE.FRACTION)
 50         },
 51         renderUI: function() {
 52             this.set("variable", this.get("variable"));
 53         },
 54         syncUI: function() {
 55             var template = this.get("custom"), hashCode = "" + Y.Wegas.Helper.hashCode(template),
 56                     data = this.computeData();
 57             if (template === "" && this.TEMPLATES[this.get("template")]) {
 58                 this.get("contentBox").setHTML(this.TEMPLATES[this.get("template")](data));
 59             } else {
 60                 if (Y.Lang.isUndefined(this.TEMPLATES[hashCode])) {
 61                     this.TEMPLATES[hashCode] = engine.compile(template);
 62                 }
 63                 try {
 64                     this.get("contentBox").setHTML(this.TEMPLATES[hashCode](data));
 65                 } catch (e) {
 66                     Y.log("Error rendering template: " + template, "error", "Wegas.Template");
 67                 }
 68             }
 69 
 70         },
 71         bindUI: function() {
 72             this.after(["dataChange", "variableChange", "templateChange"], this.syncUI);
 73             this.vdUpdateHandler = Y.Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this);
 74         },
 75         computeData: function() {
 76             var data = {}, desc = this.get("variable.evaluated");
 77 
 78             if (desc) {
 79 
 80                 if (desc instanceof Y.Wegas.persistence.ListDescriptor) {       // If the widget is a list,
 81                     desc = desc.get("currentItem");                             // display it with the current list and the current element
 82                 }
 83 
 84                 data.label = undefinedToEmpty(desc.getPublicLabel());
 85                 data.value = undefinedToEmpty(desc.getInstance().get("value"));
 86                 data.maxValue = undefinedToEmpty(desc.get("maxValue"));
 87                 data.minValue = undefinedToEmpty(desc.get("minValue"));
 88                 data.defaultValue = undefinedToEmpty(desc.get("defaultValue"));
 89                 data.variable = desc;
 90             }
 91             return Y.mix(Y.merge(this.get("data")), data, false, null, 0, true);
 92         },
 93         destructor: function() {
 94             this.vdUpdateHandler.detach();
 95         }
 96 
 97     }, {
 98         /*@lends Y.Wegas.Template*/
 99         EDITORNAME: "Variable template",
100         ATTRS: {
101             /**
102              * The target variable, returned either based on the variableName attribute,
103              * and if absent by evaluating the expr attribute.
104              */
105             variable: {
106                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
107                 _inputex: {
108                     _type: "variableselect",
109                     label: "variable"
110                 }
111             },
112             template: {
113                 type: "string",
114                 value: "text",
115                 choices: [{
116                         value: "text"
117                     }, {
118                         value: "title"
119                     }, {
120                         value: "box"
121                     }, {
122                         value: "valuebox"
123                     }, {
124                         value: "fraction"
125                     }],
126                 _inputex: {
127                     label: "Predefined template"
128                 }
129             },
130             custom: {
131                 type: "string",
132                 optional: true,
133                 value: "",
134                 _inputex: {
135                     label: "Custom template",
136                     description: "Takes precedence over predefined templates",
137                     _type: "text"
138                 }
139             },
140             data: {
141                 value: {},
142                 _inputex: {
143                     _type: "object",
144                     useButtons: true,
145                     required: false
146                 }
147             }
148         }
149     });
150 });
