 21     /**
 22      * @name Y.Wegas.Button
 23      * @extends Y.Button
 24      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 25      * @class Custom Button implementation.
 26      * @constructor
 27      * @description Custom Button implementation. Adds Y.WidgetChild and
 28      * Y.Wegas.Widget extensions to the original Y.Button
 29      */
 30     Button = Y.Base.create("button", Y.Button, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 31         /** @lends Y.Wegas.Button# */
 32         // *** Private fields *** //
 33 
 34         // *** Lifecycle Methods *** //
 35         /**
 36          * @function
 37          * @private
 38          * @description Set variable with initials values.
 39          * Plug tooltip add given (by ATTRS) css class to contentbox
 40          */
 41         initializer: function() {
 42             Button.superclass.initializer.apply(this, arguments);
 43 
 44             this.publish("click", {
 45                 emitFacade: true,
 46                 bubbles: true,
 47                 defaultFn: function() {                                         // Force event activation by default
 48                 }
 49             });
 50             //this.constructor.CSS_PREFIX = "yui3-button";                      // Revert changes done by Y.Wegas.Widget so styling will work
 51             this._cssPrefix = "yui3-button";
 52 
 53             if (this.get("cssClass")) {
 54                 this.get(CONTENTBOX).addClass(this.get("cssClass"));
 55             }
 56 
 57             if (this.get("tooltip")) {
 58                 this.plug(Y.Plugin.Tooltip, {
 59                     content: this.get("tooltip")
 60                 });
 61             }
 62         },
 63         getEditorLabel: function() {
 64             return Y.Wegas.Helper.stripHtml(this.get("label"));
 65         },
 66         /**
 67          * @function
 68          * @private
 69          * @description Call widget parent to execute its proper render function.
 70          * add "wegas-button" class to bounding box.
 71          */
 72         renderUI: function() {
 73             Button.superclass.renderUI.apply(this, arguments);
 74             this.get(BOUNDINGBOX).addClass("wegas-button");
 75         }
 76     }, {
 77         /**
 78          * @lends Y.Wegas.Button
 79          */
 80         EDITORNAME: "Button",
 81         /**
 82          * @field
 83          * @static
 84          * @description
 85          * <p><strong>Attributes</strong></p>
 86          * <ul>
 87          *    <li>label: the label of the button</li>
 88          *    <li>data: the data used by the button</li>
 89          *    <li>tooltip: the tooltip of the button</li>
 90          *    <li>disabled: boolean to choose state of the button</li>
 91          *    <li>cssClass: cssClass of the button</li>
 92          *    <li>plugins: impact to bind at the buttons</li>
 93          * </ul>
 94          */
 95         ATTRS: {
 96             label: {
 97                 type: "string"
 98             },
 99             data: {},
100             tooltip: {
101                 type: "string",
102                 optional: true,
103                 "transient": true
104             },
105             disabled: {
106                 "transient": false,
107                 type: "boolean",
108                 _inputex: {
109                     wrapperClassName: 'inputEx-fieldWrapper wegas-advanced-feature'
110                 }
111             },
112             cssClass: {
113                 value: null
114             }
115         }
116     });

129     /**
130      * Plugin which adds an unread message counter to a widget.
131      *
132      * @class Y.Wegas.UnreadCount
133      * @extends Y.Plugin.Base
134      * @borrows Y.Wegas.Plugin, Y.Wegas.Editable
135      */
136     var UnreadCount = Y.Base.create("wegas-unreadCount", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
137         /** @lends Y.Plugin.UnreadCount# */
138 
139         // *** Private fields *** //
140         /**
141          * Reference to each used functions
142          */
143         handlers: null,
144         // *** Lifecycle methods *** //
145         /**
146          * @function
147          * @private
148          * @description Set variable with initials values.
149          */
150         initializer: function() {
151             this.handlers = {};
152             this.bindUI();
153         },
154         /**
155          * @function
156          * @private
157          * @description bind function to events.
158          * When VariableDescriptorFacade is updated, do sync.
159          * When plugin's host is render, do sync.
160          */
161         bindUI: function() {
162             this.handlers.update = Y.Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this);
163             this.afterHostEvent("render", this.syncUI, this);
164         },
165         /**
166          * @function
167          * @private
168          * @description call function 'getUnreadCount' to set the number of
169          * unread on the host.
170          */
171         syncUI: function() {
172             var cb = this.get('host').get(CONTENTBOX),
173                     target = cb.one(".unread-count"),
174                     unreadCount = this.getUnreadCount();
175 
176             if (!target) {                                                      // If the counter span has not been rendered, do it
177                 cb.append('<span class="unread-count"></span>');
178                 target = cb.one(".unread-count");
179             }
180 
181             if (unreadCount > 0) {                                              // Update the content
182                 target.setContent("<span class='value'>" + unreadCount + "</span>");
183             } else {
184                 target.setContent("");
185             }
186         },
187         /**
188          * @function
189          * @private
190          * @description Detach all functions created by this widget
191          */
192         destructor: function() {
193             for (var k in this.handlers) {
194                 this.handlers[k].detach();
195             }
196         },
197         // *** Private methods *** //
198         /**
199          * @function
200          * @private
201          * @return Number of unread.
202          * @description Count the number of unread reply in given variable.
203          */
204         getUnreadCount: function() {
205             var i, instance, messages, items, count = 0,
206                     descriptor = this.get('variable.evaluated');
207 
208             if (!descriptor) {
209                 return 0;
210             }
211 
212             if (descriptor instanceof Y.Wegas.persistence.ListDescriptor) {     // For ListDescriptors, we count the children instance's
213                 items = descriptor.flatten();
214                 for (i = 0; i < items.length; i = i + 1) {
215                     if (Y.Wegas.persistence.QuestionDescriptor
216                             && items[i] instanceof Y.Wegas.persistence.QuestionDescriptor) {
217                         instance = items[i].getInstance();
218                         //count += instance.get("unread") ? 1 : 0;
219                         if (instance.get("replies")) {
220                             count += instance.get("replies").length === 0 && instance.get("active") ? 1 : 0; // only count if it is active
221                         }
222                     }
223                 }
224             }
225 
226             if (descriptor instanceof Y.Wegas.persistence.InboxDescriptor) {
227                 messages = descriptor.getInstance().get("messages");            // For InboxVariableDescriptors, we count the replies
228                 for (i = 0; i < messages.length; i = i + 1) {
229                     count += messages[i].get("unread") ? 1 : 0;
230                 }
231             }
232 
233             return count;
234         }
235     }, {
236         NS: "UnreadCount",
237         NAME: "UnreadCount",
238         /**
239          * @lends Y.Plugin.UnreadCount
240          */
241         /**
242          * @field
243          * @static
244          * @description
245          * <p><strong>Attributes</strong></p>
246          * <ul>
247          *    <li>variable: The target variable, returned either based on the name
248          *     attribute, and if absent by evaluating the expr attribute.</li>
249          * </ul>
250          */
251         ATTRS: {
252             /**
253              * The target variable, returned either based on the variableName attribute,
254              * and if absent by evaluating the expr attribute.
255              */
256             variable: {
257                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
258                 _inputex: {
259                     _type: "variableselect",
260                     label: "List to count unread"
261                 }
262             }
263         }
264     });
265     Y.namespace('Plugin').UnreadCount = UnreadCount;
