 14 YUI.add('wegas-pageloader', function(Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = 'contentBox', PageLoader;
 18 
 19     /**
 20      * @name Y.Wegas.PageLoader
 21      * @extends Y.Widget
 22      * @borrows Y.WidgetChild, Y.WidgetParent, Y.Wegas.Widget, Y.Wegas.Editable
 23      * @class  class loader of wegas's pages
 24      * @constructor
 25      * @description Load pages and request widget to render.
 26      */
 27     PageLoader = Y.Base.create("wegas-pageloader", Y.Widget,
 28             [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 29         /** @lends Y.Wegas.PageLoader# */
 30 
 31         // *** Private fields *** //
 32         /**
 33          * Current page id
 34          */
 35         currentPageId: null,
 36         /**
 37          * Reference to each used functions
 38          */
 39         handlers: null,
 40         // *** Lifecycle Methods ***/
 41         /**
 42          * @function
 43          * @private
 44          * @description Set variable with initials values.
 45          * Set page to default page
 46          * Keep a references of all loaded PageLoaders in PageLoader.pageLoaderInstances.
 47          */
 48         initializer: function() {
 49             this.handlers = [];
 50             PageLoader.pageLoaderInstances[this.get("pageLoaderId")] = this;    // We keep a references of all loaded PageLoaders
 51             if (this.get("defaultPageId")) {
 52                 this.set("pageId", this.get("defaultPageId"));
 53             }
 54 
 55         },
 56         /**
 57          * @function
 58          * @private
 59          * @description bind function to events.
 60          * When page is updated, syncUI
 61          * When a response is fire by the VariableDescriptor, do sync
 62          * When the current player change, do sync
 63          * When an exception in fire, stop loading page, show error message.
 64          */
 65         bindUI: function() {
 66             var onUpdate = function(e) {
 67                 if ("" + this.get("variable.evaluated") !== "" + this.get('pageId')) {
 68                     this.syncUI();
 69                 }
 70             };
 71             //Y.Wegas.Facade.Page.after("response", this.syncUI, this);
 72 //            this.handlers.push(Y.Wegas.Facade.Page.cache.after("pageUpdated", function(e) {
 73 //                if (e.page && ("" + e.page["@pageId"] === "" + this.get("pageId"))) {
 74 //                    this.currentPageId = null; // @hack force update
 75 //                    this.syncUI();
 76 //                }
 77 //            }, this));
 78 
 79             this.handlers.push(Y.Wegas.Facade.VariableDescriptor.after("update", onUpdate, this));
 80 
 81             this.on("*:exception", function(e) {
 82                 var test;
 83 
 84                 e.halt(true);
 85 
 86                 if (test = e.message.match(/ConstraintViolationException: (.*) is out of bound/)) {
 87                     this.showMessage("error", "Insufficient " + test[1] + ".");
 88                 } else {
 89                     this.showMessage("error", e.message);
 90                 }
 91             });
 92         },
 93         /**
 94          * @function
 95          * @private
 96          * @description Set pageId and displayed new page if the id is
 97          *  different that the current page id
 98          */
 99         syncUI: function() {
100             var val = this.get("variable.evaluated");
101             if (val && val.getInstance().get('value')) {                        // If there is a variable to refresh
102                 this.set("pageId", val.getInstance().get('value'));
103             } else {                                                            // Otherwise use pageId (in case the setter has not been called yet)
104                 this.set("pageId", this.get("pageId"));
105             }
106         },
107         /**
108          * @function
109          * @private
110          * @description Destroy widget and detach all functions created by this widget
111          * remove instance kept in PageLoader.pageLoaderInstances.
112          */
113         destructor: function() {
114             var i;
115             if (this.get("widget")) {
116                 this.get("widget").destroy();
117             }
118             for (i = 0; i < this.handlers.length; i += 1) {
119                 this.handlers[i].detach();
120             }
121             delete PageLoader.pageLoaderInstances[this.get("pageLoaderId")];
122         },
123         // *** Private Methods ***/
124         /**
125          * @function
126          * @private
127          * @param {String} pageId check for this page's ID.
128          * @return boolean
129          * @description Return true if an ancestor already loads pageId
130          */
131         ancestorWithPage: function(pageId) {                                     //Page loader mustn't load the page who contain itself.
132             var same = false;
133             this.get("boundingBox").ancestors("." + this.getClassName(), false).some(function(node) {
134                 var widget = Y.Widget.getByNode(node);
135                 if (pageId === widget.get('pageId') || pageId === widget.get('variable.evaluated')) {
136                     same = true;
137                     Y.log("Pageloader [" + this.get("pageLoaderId") + "] tries to load page (" + pageId + ") already loaded by one of its ancestor.", 'warn', 'Y.Wegas.PageLoader');
138                     return true;
139                 }
140             }, this);
141             return same;
142         }
143 
144     }, {
145         /** @lends Y.Wegas.PageLoader */
146         EDITORNAME: "Pageloader",
147         /**
148          * @field
149          * @static
150          * @description
151          * <p><strong>Attributes</strong></p>
152          * <ul>
153          *    <li>pageLoaderId: the id of this pageLoader</li>
154          *    <li>defaultPageId: the id of the default page to load</li>
155          *    <li>pageId: the page id to load</li>
156          *    <li>variable: a variable (or expression) which contain the id of the page to load
157          *    The target variable, returned either based on the name attribute,
158          *    and if absent by evaluating the expr attribute.
159          *    </li>
160          *    <li>widget: A widget to render in current page (transient)</li>
161          * </ul>
162          */
163         ATTRS: {
164             /**
165              * the id of this pageLoader
166              */
167             pageLoaderId: {
168                 type: "string",
169                 value: "PageLoader" + Y.Lang.now(), //generate a default pageLoaderId
170                 _inputex: {
171                     label: "Zone id"
172                 }
173             },
174             /**
175              * the id of the default page to load
176              */
177             defaultPageId: {
178                 type: "string",
179                 _inputex: {
180                     label: "Default page",
181                     _type: "pageselect",
182                     required: false
183                 }
184             },
185             /**
186              * the page id to load
187              */
188             pageId: {
189                 type: "string",
190                 "transient": true,
191                 setter: function(val) {
192                     if (!val || val === this.currentPageId || this.ancestorWithPage(val)) {// If the widget is currently being loaded, escape
193                         return val;
194                     }
195                     this.currentPageId = val;
196                     Y.Wegas.Facade.Page.cache.getPage(val, Y.bind(function(widgetCfg) {
197                         if (!widgetCfg) {
198                             return val;
199                         }
200                         if (this.get("widget")) {
201                             this.get("widget").destroy();                           // @fixme we should remove the widget instead of destroying it
202                         }
203                         this.get(CONTENTBOX).empty();
204                         this.showOverlay();
205                         try {
206                             Y.Wegas.Widget.use(widgetCfg, Y.bind(function(cfg) {    // Load the subwidget dependencies
207                                 var widget = Y.Wegas.Widget.create(cfg);            // Render the subwidget
208                                 widget.render(this.get(CONTENTBOX));
209                                 widget['@pageId'] = cfg['@pageId'];
210                                 this.set("widget", widget);
211                                 widget.addTarget(this);                             // Event on the loaded widget will be forwarded
212                                 this.hideOverlay();
213                             }, this, widgetCfg));
214                         } catch (e) {
215                             Y.log('renderUI(): Error rendering widget: ' + (e.stack || e), 'error', 'Wegas.PageLoader');
216                         }
217                     }, this));
218 
219 
220                     return val;
221                 }
222             },
223             /**
224              * A variable (or expression) which contain the id of the page to load
225              * The target variable, returned either based on the name attribute,
226              * and if absent by evaluating the expr attribute.
227              */
228             variable: {
229                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER
230             },
231             /**
232              * A widget to render in current page (transient)
233              */
234             widget: {
235                 "transient": true
236             }
237         },
238         pageLoaderInstances: {},
239         find: function(id) {
240             return PageLoader.pageLoaderInstances[id];
241         }
242     });
243     Y.namespace('Wegas').PageLoader = PageLoader;
244 
245 });
