 12 YUI.add('wegas-monopoly-controller', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class Monopoly controller
 17      *  @name Y.Plugin.MonopolyController
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var Wegas = Y.Wegas,
 22     MonopolyController = Y.Base.create("wegas-monopoly-controller", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 23         /** @lends Y.Plugin.MonopolyController */
 24 
 25         /**
 26          * @function
 27          * @private
 28          */
 29         initializer: function() {
 30             this.afterHostEvent("render", function(){
 31                 this.dice = this.get("host").item(0);
 32                 this.diceAfter();
 33                
 34                 this.buy = this.get("host").item(1);
 35                 this.boxValue = Y.Wegas.Facade.VariableDescriptor.cache.find("name", "boxValue").getAttrs().items;
 36                 this.buyProperty();
 37                
 38                 this.display = this.get("host").item(5);
 39                 this.display.checkState();
 40                
 41                 this.restart = this.get("host").item(3);
 42                 this.checkRestart();
 43                 this.clickRestart();
 44                 
 45                 this.next = this.get("host").item(2);
 46                 this.clickNext();
 47             });
 48         },
 49         
 50         diceAfter: function(){
 51             this.dice.after("diceRolling",function(){
 52                 this.dice.rollButton.disable();
 53             }, this);
 54             this.dice.after("diceRolled", this.setPion, this);
 55         },
 56         
 57         setPion : function(e){
 58             this.display.setPion(e.target.result);
 59             this.display.setState("afterRoll");
 60         },
 61         
 62         clickNext: function(){
 63             var turn = Y.Wegas.Facade.VariableDescriptor.cache.find("name", "turnOf"),
 64                 id, player;
 65             this.next.on("click", function(){
 66                 if (turn.get("scope") instanceof Y.Wegas.persistence.GameModelScope){ //@fixme when gameScope works
 67                     player = Y.Wegas.Facade.Game.cache.getCurrentGame().get("teams");
 68                     this.nextPlayer("teams", turn, player);
 69                 } else if (turn.get("scope") instanceof Y.Wegas.persistence.TeamScope) {
 70                     this.nextPlayer("players", turn, player);
 71                     player = Y.Wegas.Facade.Game.cache.getCurrentTeam().get("players");
 72                 }
 73             }, this);
 74         },
 75         
 76         nextPlayer: function (scope, turn, player){
 77             var i;
 78             for (i = 0; i < player.length; i++){
 79                 if (turn.get("scope").getInstance().get("value") == Y.Wegas.Facade.Game.cache.getCurrentGame().get(scope)[i].get("id")){
 80                     i++
 81                     if (i == player.length){
 82                         this.display.setCurrentPlayer(Y.Wegas.Facade.Game.cache.getCurrentGame().get(scope)[0].get("id"));
 83                     } else {
 84                         this.display.setCurrentPlayer(Y.Wegas.Facade.Game.cache.getCurrentGame().get(scope)[i].get("id"));
 85                     }
 86                 }
 87             }
 88             this.display.setState("wait"); 
 89         },
 90         
 91         checkRestart : function(){
 92             this.restartValue = Y.Wegas.Facade.VariableDescriptor.cache.find("name", "restart").getInstance().get("value");
 93             if (this.restartValue == "true"){
 94                 Y.one('.game .yui3-togglebutton').addClass('yui3-button-selected');
 95             } else {
 96                 Y.one('.game .yui3-togglebutton').removeClass('yui3-button-selected');
 97             }
 98         },
 99         
100         clickRestart : function (e){
101             this.restart.on("click", function(){
102                 // check if all player have true TODO
103                 
104                 // change restart status
105                 if (this.restartValue == "true"){
106                     this.restartValue = "false";
107                 } else {
108                     this.restartValue = "true";
109                 } 
110                 Y.Wegas.Facade.VariableDescriptor.sendRequest({
111                     request: "/Script/Run/" + Y.Wegas.app.get('currentPlayer'),
112                     headers:{
113                         'Content-Type': 'application/json; charset=ISO-8859-1',
114                         'Managed-Mode':'true'
115                     },
116                     cfg: {
117                         method: "POST",
118                         data: Y.JSON.stringify({
119                             "@class": "Script",
120                             "language": "JavaScript",
121                             "content": "importPackage(com.wegas.core.script);\nrestart.value ="+ this.restartValue +";"
122                         })
123                     }
124                 });
125             })
126         },
127         
128         buyProperty: function(){
129             this.buy.on("click", function(){
130                 var position, money;
131                 position = Y.Wegas.Facade.VariableDescriptor.cache.find("name", "position").getInstance().get("value");
132                 position--;
133                 // check if property is free
134                 if (this.boxValue[position].getInstance().get("properties").playerId != "" ||
135                 this.boxValue[position].getInstance().get("properties").playerId == "notBuyable"){
136                     alert("this property is not buyable");
137                     return;
138                 }
139                 // check if have enough money               
140                 money = Y.Wegas.Facade.VariableDescriptor.cache.find("name", "money").getInstance().get("value");
141                 if (this.boxValue[position].getInstance().get("properties").value <= money){
142                     this.setMoney(money - this.boxValue[position].getInstance().get("properties").value);
143                 // add property
144                     this.boxValue[position].getInstance().get("properties").playerId = Y.Wegas.app.get('currentPlayer');
145                     Y.Wegas.Facade.VariableDescriptor.sendRequest({
146                         request: "/" + this.boxValue[position].getInstance().get("descriptorId") +"/VariableInstance/" + this.boxValue[position].getInstance().get("id"),
147                         headers:{
148                             'Content-Type': 'application/json; charset=ISO-8859-1',
149                             'Managed-Mode':'true'
150                         },
151                         cfg: {
152                             method: "PUT",
153                             data: this.boxValue[position].getInstance().toJSON()
154                         },
155                         on: {
156                             success: Y.bind(function (e) {
157                                 this.display.checkPropertyBuyable();
158                             }, this)
159                         }
160                     });
161                 } else {
162                     alert("You have not enought money");
163                 }
164             }, this);
165             
166         },
167         
168         setMoney: function(value){
169             Y.Wegas.Facade.VariableDescriptor.sendRequest({
170                 request: "/Script/Run/" + Y.Wegas.app.get('currentPlayer'),
171                 headers:{
172                     'Content-Type': 'application/json; charset=ISO-8859-1',
173                     'Managed-Mode':'true'
174                 },
175                 cfg: {
176                     method: "POST",
177                     data: Y.JSON.stringify({
178                         "@class": "Script",
179                         "language": "JavaScript",
180                         "content": "importPackage(com.wegas.core.script);\nmoney.value ="+ value +";"
181                     })
182                 }
183             });
184         }
185         
186     }, {
187         NS: "MonopolyController",
188         NAME: "MonopolyController"
189     });
190     Y.namespace("Plugin").MonopolyController = MonopolyController;
191 
192 });
