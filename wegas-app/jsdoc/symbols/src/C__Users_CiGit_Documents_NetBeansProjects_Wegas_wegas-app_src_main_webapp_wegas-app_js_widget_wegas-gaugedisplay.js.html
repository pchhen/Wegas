 13 YUI.add('wegas-gaugedisplay', function(Y) {
 14     "use strict";
 15 
 16     var CONTENTBOX = 'contentBox', GaugeDisplay;
 17 
 18     /**
 19      * @name Y.Wegas.GaugeDisplay
 20      * @extends Y.Widget
 21      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 22      * @class Manage a canevas gauge
 23      * @constructor
 24      * @description Manage a canevas gauge based on a instance's value
 25      */
 26     GaugeDisplay = Y.Base.create("wegas-gauge", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 27         /**
 28          * @lends Y.Wegas.GaugeDisplay#
 29          */
 30         // *** Private fields *** //
 31         /**
 32          * Content box of this widget, static
 33          */
 34         CONTENT_TEMPLATE: '<div style="text-align: center;line-height:3px"><canvas height="50px" width="100px"></canvas><center class="label"></center><center class="percent"></center></div>',
 35         /**
 36          * Maximum value displayed by the gauge, static
 37          */
 38         MAXVAL: 200,
 39         /**
 40          * Reference to each used functions
 41          */
 42         handlers: null,
 43         /**
 44          * reference to the gauge object
 45          */
 46         gauge: null,
 47         /**
 48          * reference to the gauge status
 49          */
 50         disable: null,
 51         // ** Lifecycle Methods ** //
 52         /**
 53          * @function
 54          * @private
 55          * @description Set variables with initials values.
 56          */
 57         initializer: function() {
 58             this.handlers = [];
 59             this.disable = false;
 60         },
 61         /**
 62          * @function
 63          * @private
 64          * @description create  and render a gauge with a defined configuration
 65          *  or given cfg (in ATTRS)
 66          */
 67         renderUI: function() {
 68             this.setValue(this.get("cfg"));
 69         },
 70         /**
 71          * @function
 72          * @private
 73          * @description Create and set value to gauge.
 74          */
 75         setValue: function(cfg) {
 76             var opts = {
 77 //                lines: cfg.lines || 1, //don't work with this new version of gauge library
 78                 // The number of lines to draw
 79                 angle: cfg.angle || 0.15,
 80                 // The length of each line
 81                 lineWidth: cfg.lineWidth || 0.44,
 82                 // The line thickness
 83                 pointer: cfg.pointer || {
 84                     length: 0.5, // The radius of the inner circle
 85                     strokeWidth: 0.035, // The rotation offset
 86                     color: '#000000'                                            // Fill color
 87                 },
 88 //                colorStart: cfg.colorStart || '#0981A9', // Colors, don't work with this new version of gauge library
 89 //                colorStop: cfg.colorStop || '#000000',
 90                 strokeColor: cfg.strokeColor || '#FFFFFF',
 91                 percentColors: cfg.percentColors || [[0.0, "#0981A9"]],
 92 //              generateGradient: cfg.generateGradient || false //don't work with this new version of gauge library
 93             };
 94             this.gauge = new Gauge(this.get("contentBox").one("canvas").
 95                     getDOMNode());// create the  gauge!conso
 96             this.gauge.setOptions(opts);
 97             this.gauge.maxValue = this.MAXVAL;                                  // set max gauge value
 98             this.gauge.animationSpeed = 32;                                     // set animation speed (32 is default value)
 99         },
100         /**
101          * @function
102          * @private
103          * @description bind function to events.
104          * When VariableDescriptorFacade is updated, do sync.
105          */
106         bindUI: function() {
107             this.handlers.push(Y.Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this));
108             this.after('disabledChange', function(e) {
109                 this.disable = e.newVal;
110                 this.syncUI();
111             }, this);
112 
113             this.after("cfgChange", function(e) {
114                 this.setValue(e.newVal);
115             });
116         },
117         /**
118          * @function
119          * @private
120          * @description Set the label, the min value, the max value, the
121          *  value and display the percentage based on these value. This values
122          *   are based on the descriptor/instance variable given in ATTRS.
123          */
124         syncUI: function() {
125             var maxVal, minVal, value, label,
126                     variableDescriptor = this.get("variable.evaluated");
127             if (!variableDescriptor) {
128                 return;
129             }
130 
131             label = this.get("label") || variableDescriptor.getLabel();
132             minVal = variableDescriptor.get("minValue");
133             maxVal = variableDescriptor.get("maxValue") - minVal;
134             value = (variableDescriptor.getInstance().
135                     get("value") - minVal) / maxVal * this.MAXVAL;
136             if (!value || this.disable) {
137                 value = 0.1;                                                    // @hack @fixme unkown bug, value seams to be treated by gauge as false...
138             }
139 
140             this.gauge.set(value);                                              // set actual value
141             this.get(CONTENTBOX).one(".label").setContent(label);
142             this.get(CONTENTBOX).one(".percent").
143                     setContent(Math.round(value / this.MAXVAL * this.get("percentMaxValue")) + "%");
144         },
145         getEditorLabel: function() {
146             var variable = this.get("variable.evaluated");
147             if (variable) {
148                 return variable.getEditorLabel();
149             }
150             return null;
151         },
152         /**
153          * @function
154          * @private
155          * @description Detach all functions created by this widget.
156          */
157         destructor: function() {
158             for (var i = 0; i < this.handlers.length; i += 1) {
159                 this.handlers[i].detach();
160             }
161         }
162     }, {
163         /**
164          * @lends Y.Wegas.GaugeDisplay#
165          */
166         EDITORNAME: "Gauge",
167         /**
168          * @field
169          * @static
170          * @description
171          * <p><strong>Attributes</strong></p>
172          * <ul>
173          *    <li>variable: The target variable, returned either based on the name attribute,
174          * and if absent by evaluating the expr attribute.</li>
175          *    <li>label: A label for the gauge, if no one is given, take the
176          *     public label of the variable</li>
177          *    <li>cfg: the configuration of the gauge (object)</li>
178          * </ul>
179          */
180         ATTRS: {
181             /**
182              * The target variable, returned either based on the name attribute,
183              * and if absent by evaluating the expr attribute.
184              */
185             variable: {
186                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
187                 _inputex: {
188                     _type: "variableselect",
189                     label: "variable"
190                 }
191             },
192             /**
193              * A label for the gauge, if no one is given, take the public
194              *  label of the variable
195              */
196             label: {
197                 type: "string",
198                 optional: true,
199                 validator: Y.Lang.isString,
200                 _inputex: {
201                     label: "Label"
202                 }
203             },
204             percentMaxValue: {
205                 type: "number",
206                 optional: true,
207                 validator: Y.Lang.isNumber,
208                 value: 100,
209                 _inputex: {
210                     label: "Percent max value"
211                 }
212             },
213             /**
214              * The configuration of the gauge (object)
215              */
216             cfg: {
217                 value: {},
218                 _inputex: {
219                     _type: "wegasobject",
220                     useButtons: true,
221                     elementType: {
222                         type: "wegaskeyvalue",
223                         availableFields: [{
224                                 name: "angle",
225                                 type: "number"
226                             }, {
227                                 name: "lineWidth",
228                                 type: "number"
229                             }, {
230                                 name: "strokeColor",
231                                 type: "colorpicker"
232                             }, {
233                                 name: "pointer",
234                                 type: "group",
235                                 fields: [
236                                     {type: "number", name: 'length', typeInvite: "length"},
237                                     {type: "number", name: 'strokeWidth', typeInvite: "stroke width"},
238                                     {type: "colorpicker", name: 'color'}
239                                 ]
240                             }, {
241                                 name: "percentColors",
242                                 type: "list",
243                                 useButtons: true,
244                                 elementType: {
245                                     type: "combine",
246                                     fields: [
247                                         {type: "string", name: "value", typeInvite: "value"},
248                                         {type: "colorpicker", name: "color"}
249                                     ]
250                                 },
251                                 palette: 3
252                             }]
253                     }
254                 }
255             }
256         }
257     });
258 
259     Y.namespace('Wegas').GaugeDisplay = GaugeDisplay;
260 });
