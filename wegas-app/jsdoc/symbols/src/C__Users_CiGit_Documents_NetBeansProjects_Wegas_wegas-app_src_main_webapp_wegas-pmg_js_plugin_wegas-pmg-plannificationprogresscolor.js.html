 12 YUI.add('wegas-pmg-plannificationprogresscolor', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class color plannification progress in datatable
 17      *  @name Y.Plugin.PlannificationProgressColor
 18      *  @extends Y.Plugin.PlannificationActivityColor
 19      *  @constructor
 20      */
 21     var Wegas = Y.Wegas,
 22             PlannificationProgressColor = Y.Base.create("wegas-pmg-plannificationprogresscolor", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 23         /** @lends Y.Plugin.PlannificationProgressColor */
 24         initializer: function() {
 25             //this.taskTable;
 26             this.onceAfterHostEvent("render", function() {
 27                 this.sync();
 28 
 29                 this.afterHostMethod("syncUI", this.sync);
 30 
 31                 this.get("host").datatable.after("sort", this.sync, this);
 32             });
 33         },
 34         sync: function() {
 35             this.taskTable = {};
 36             this.fillTaskTable();
 37             this.pertValue();
 38             this.findCell();
 39         },
 40         fillTaskTable: function() {
 41             var i, taskDesc, taskInst, dt = this.get("host").datatable,
 42                     properties;
 43 
 44             for (i = 0; i < dt.data._items.length; i++) {
 45                 taskDesc = Y.Wegas.Facade.VariableDescriptor.cache.find("id", dt.getRecord(i).get("id"));
 46                 taskInst = taskDesc.getInstance();
 47                 properties = taskInst.get("properties");
 48                 if (parseInt(properties.completeness) < 100) {
 49                     taskDesc.timeSolde = this.timeSolde(taskDesc);
 50                     taskDesc.startPlannif = this.startPlannif(taskDesc);
 51 
 52                     this.taskTable[taskDesc.get("id")] = taskDesc;
 53                 }
 54             }
 55         },
 56         timeSolde: function(taskDesc) {
 57             var taskInst = taskDesc.getInstance(), properties = taskInst.get("properties"), timeSolde;
 58             if (taskInst.get("plannification").length > 0) {
 59                 timeSolde = (1 - parseInt(properties.completeness) / 100) * taskInst.get("plannification").length;
 60             } else {
 61                 timeSolde = (1 - parseInt(properties.completeness) / 100) * taskInst.get("duration");
 62             }
 63             return timeSolde;
 64         },
 65         startPlannif: function(taskDesc) {
 66             var taskInst = taskDesc.getInstance(), plannification = taskInst.get("plannification"), min;
 67             min = Math.min.apply(Math, plannification);
 68             if (min !== Infinity) {
 69                 return min;
 70             } else {
 71                 return 0;
 72             }
 73         },
 74         pertValue: function() {
 75             var predecessors, taskId, taskDesc, i, maxPert, predecessorDuration,
 76                     allPredDefine, countPertValue = 0, treated = [], predecessorId;
 77 
 78             while (countPertValue < Y.Object.size(this.taskTable)) {
 79                 for (taskId in this.taskTable) {
 80                     taskDesc = this.taskTable[taskId];
 81 
 82                     if (treated.indexOf(taskId) > -1) {
 83                         continue;
 84                     }
 85 
 86                     maxPert = this.get("host").schedule.currentPeriod();
 87                     allPredDefine = true;
 88                     predecessors = taskDesc.get("predecessors");
 89 
 90                     for (i = 0; i < predecessors.length; i++) {
 91                         predecessorId = predecessors[i].get("id");
 92                         if (!this.taskTable[predecessorId]) {
 93                             continue;
 94                         }
 95                         // verifie si le pr�decesseur possede le debut pert
 96                         if (this.taskTable[predecessorId].startPert) {
 97                             // defini la dur�e du pr�decesseur
 98                             predecessorDuration = this.taskTable[predecessorId].startPert + this.taskTable[predecessorId].timeSolde;
 99                             // verifie si la dur�e du pr�decesseur et la plus grande
100                             if (predecessorDuration > maxPert) {
101                                 maxPert = predecessorDuration;
102                             }
103                         } else {
104                             allPredDefine = false;
105                         }
106                     }
107                     // si tous les pr�decesseur on un debut pert alors on l'ajoute dans la liste
108                     if (allPredDefine) {
109                         treated.push(taskId);
110                         taskDesc.startPert = maxPert;
111                         taskDesc.startMax = Math.max(taskDesc.startPert, taskDesc.startPlannif);
112                         taskDesc.end = taskDesc.startMax + taskDesc.timeSolde;
113                         countPertValue++;
114                     }
115                 }
116             }
117         },
118         findCell: function() {
119             var taskId, taskDesc, host = this.get("host"),
120                     dt = this.get("host").datatable, i, ii, cell;
121 
122             for (i = 0; i < dt.data.size(); i++) {
123                 for (taskId in this.taskTable) {
124                     taskDesc = this.taskTable[taskId];
125                     if (dt.getRecord(i).get("id") === taskDesc.get("id")) {
126                         for (ii = 1; ii <= host.schedule.get("columnToAdd"); ii++) {
127                             cell = host.schedule.getCell(i, ii);
128                             this.findCssClass(ii, taskDesc.startMax, taskDesc.end, cell);
129                         }
130                         break;
131                     }
132                 }
133             }
134 
135         },
136         findCssClass: function(time, start, end, cell) {
137             var decimal;
138             if (time < this.get("host").schedule.currentPeriod()) {
139                 return;
140             }
141             if (start - parseInt(start) === 0 && end - parseInt(end) === 0) {
142                 end--;
143             }
144             if (time === parseInt(end) || time === parseInt(start) && parseInt(start) === parseInt(end)) {
145                 decimal = end - parseInt(end);
146                 if (decimal === 0) {
147                     this.addColor(cell, "fill100");
148                 } else if (decimal > 0 && decimal <= 0.3) {
149                     this.addColor(cell, "fill0to25");
150                 } else if (decimal > 0.6 && decimal <= 0.99999) {
151                     this.addColor(cell, "fill0to75");
152                 } else {
153                     this.addColor(cell, "fill0to50");
154                 }
155             } else if (time === parseInt(start)) {
156                 decimal = start - parseInt(start);
157                 if (decimal === 0) {
158                     this.addColor(cell, "fill100");
159                 } else if (decimal > 0 && decimal <= 0.3) {
160                     this.addColor(cell, "fill25to100");
161                 } else if (decimal > 0.6 && decimal <= 0.99999) {
162                     this.addColor(cell, "fill75to100");
163                 } else {
164                     this.addColor(cell, "fill50to100");
165                 }
166             } else if (time > parseInt(start) && time < parseInt(end)) {
167                 this.addColor(cell, "fill100");
168             }
169         },
170         addColor: function(cell, cssClass) {
171             cell.append("<span class='progress " + cssClass + "'></span>");
172         }
173     }, {
174         NS: "plannificationprogresscolor",
175         NAME: "PlannificationProgressColor"
176     });
177     Y.namespace("Plugin").PlannificationProgressColor = PlannificationProgressColor;
178 });
