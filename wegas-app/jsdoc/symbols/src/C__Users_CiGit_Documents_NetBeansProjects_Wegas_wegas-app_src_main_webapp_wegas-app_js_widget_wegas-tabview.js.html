152                 Y.mix(tabCfg, {

178             var nTab = TabView.createTab(id, tabViewSelector, tabCfg);          // create a new one

183             nTab.get("panelNode").one("div").empty();                           // @fixme since the above method is not enough
184             nTab.load(widgetCfg, fn);                                           // Load target widget
185             return nTab;
186         }

191     /**
192      * Extension enabling a Tab to be a parent of another Widget.
193      *
194      * @modified from original WidgetParent module
195      *
196      * @module widget-parent
197      */
198     function Parent(config) {
199         //Y.WidgetParent.call(this, config);
200         this.publish("addChild", {
201             defaultTargetOnly: true,
202             defaultFn: this._defAddChildFn
203         });
204         this.publish("removeChild", {
205             defaultTargetOnly: true,
206             defaultFn: this._defRemoveChildFn
207         });
208 
209         this._items = [];
210 
211         var children, handle;
212 
213         if (config && config.children) {
214 
215             children = config.children;
216 
217             handle = this.after("initializedChange", function(e) {
218                 this._add(children);
219                 handle.detach();
220             });
221 
222         }
223 
224         //  Widget method overlap
225         Y.after(this._renderChildren, this, "renderUI");
226         Y.after(this._bindUIParent, this, "bindUI");
227 
228         //this.after("selectionChange", this._afterSelectionChange);
229         //this.after("selectedChange", this._afterParentSelectedChange);
230         //this.after("activeDescendantChange", this._afterActiveDescendantChange);
231 
232         this._hDestroyChild = this.after("*:destroy", this._afterDestroyChild);
233         this.after("*:focusedChange", this._updateActiveDescendant);
234     }

243     /**
244      * @name Y.Wegas.Tab
245      * @extends Y.Tab
246      * @borrows Y.Wegas.Widget, Parent, Y.WidgetChild
247      * @class class to manage a tabspecific to Wegas

251     Tab = Y.Base.create("tab", Y.Tab, [Parent, Y.WidgetChild, Y.Wegas.Editable, Y.Wegas.Layout], {
252         /** @lends Y.Wegas.Tab# */
253         PANEL_TEMPLATE: '<div><div class=\"panel-inner\"></div></div>',
254         // *** Private Fields *** //
255         /**
256          * Array of widget items.
257          */
258         _witems: null,
259         // *** Lifecycle Methods *** //
260         /**
261          * @function
262          * @private
263          * @description Set variables with initials values.
264          * call explicitly initializer method of widget parent.
265          * assign this tab in owner TabView.
266          */
267         initializer: function(cfg) {
268             Tab.superclass.initializer.apply(this, arguments);
269             TabView.tabs[cfg.id || cfg.label] = this;
270             this._witems = [];
271             this.on("addChild", function(e) {
272                 this._witems.push(e.child);
273             });
274 
275             //this.plug(Closable);
276             this.plug(Y.Plugin.PopupListener, {
277                 targetAttr: "panelNode",
278                 alignAttr: "panelNode"
279             });
280         },
281         /**
282          * @function
283          * @private
284          * @description call render method of widget parent.
285          */
286         renderUI: function() {
287             Tab.superclass.renderUI.apply(this, arguments);
288         },
289         /**
290          * @function
291          * @private
292          * @description call sync method of widget parent.
293          */
294         syncUI: function() {
295             Tab.superclass.syncUI.apply(this, arguments);
296         },
297         /**
298          * @function
299          * @private
300          * @description delete Tab in owner Tabview.
301          *  call explicitly destructor method of widget parent.
302          */
303         destructor: function() {
304             var i;
305             for (i = 0; i < this._witems.length; i += 1) {
306                 this._witems[i].destroy();
307             }
308 
309             delete TabView.tabs[this.get("id")];
310             Tab.superclass.destructor.apply(this, arguments);
311         },
312         _onActivate: function(e) {
313             if (e.target.get("disabled")) {
314                 return;
315             }
316             Tab.superclass._onActivate.apply(this, arguments);
317         },
318         // *** Private Methods *** //
319         /**
320          * @function
321          * @private
322          * @param cfg
323          * @param callback
324          * @description Retrieves the given widget configuration and add it
325          * to the tab
326          */
327         load: function(cfg, callback) {
328             this.showOverlay();
329             Y.Wegas.Widget.use(cfg, Y.bind(function(cfg, callback) {            // Load the subpage dependencies
330                 this.hideOverlay();
331                 var widgets = this.add(cfg);                                    // Render the subpage
332                 if (callback) {
333                     callback(widgets.item(0), this);                            // Trigger the callback
334                 }
335             }, this, cfg, callback));
336         },
337         /**
338          * @function
339          * @private
340          * @description Override Y.WidgetParent to render children in the panel node;
341          * to the tab
342          */
343         _renderChildren: function() {
344             var renderTo = this._childrenContainer || this.get("panelNode").one("div");    // @modified
345 
346             this._childrenContainer = renderTo;
347 
348             this.each(function(child) {
349                 child.render(renderTo);
350             }, this);
351         },
352         /**
353          * @function
354          * @private
355          * @param index
356          * @description return the widget item at given index
357          */
358         witem: function(index) {
359             return this._witems[index];
360         }
361     }, {
362         CSS_PREFIX: "yui3-tab",
363         /** @lends Y.Wegas.Tab */
364 
365         /**
366          * @field
367          * @static
368          * @description
369          * <p><strong>Attributes</strong></p>
370          * <ul>
371          *    <li>Content: Overrides the panelNode management</li>
372          * </ul>
373          */
374         ATTRS: {
375             /**
376              * Overrides the panelNode management
377              */
378             label: {
379                 type: "string"
380             },
381             content: {
382                 "transient": true,
383                 setter: function() {
384                 }
385             },
386             panelNode: {
387                 "transient": true
388             }
389         }
390     });

464      * @class plugin to toggle visibility of a tab
465      * @constructor

467     var LayoutToggleTab = function() {

469     };
470     Y.extend(LayoutToggleTab, Y.Plugin.Base, {
471         /** @lends Y.Wegas.LayoutToggleTab# */
472 
473         // *** Private fields *** //
474         /**
475          * Html template added in host's contentbox
476          */
477         REMOVE_TEMPLATE: '<a class="yui3-tab-remove" title="remove tab">x</a>',

479          * @function
480          * @private
481          * @description if "removeChild" is fired by host, hide tab.
482          *  If "addChild" is fired by host, show tab.
483          */
484         initializer: function() {
485             this.afterHostEvent("removeChild", function() {
486                 Y.later(100, this, function() {
487                     if (this.get("host").isEmpty()) {
488                         Y.Wegas.app.widget.hidePosition("right");
489                     }
490                 });
491             });
492             this.onHostEvent("addChild", function() {
493                 if (this.get("host").isEmpty()) {
494                     Y.Wegas.app.widget.showPosition("right");
495                 }
496             });
497         }
498     }, {
499         NS: "LayoutToggleTab",
500         NAME: "LayoutToggleTab"
501     });

505     /**
506      * Override to add support for disabled tabs
507      * @param {type} e
508      */
509     Y.Tab.prototype._onActivate = function(e) {
510         if (e.target === this && !this.get("disabled")) {   /* modified */
511             //  Prevent the browser from navigating to the URL specified by the
512             //  anchor's href attribute.
513             e.domEvent.preventDefault();
514             e.target.set('selected', 1);
515         }
