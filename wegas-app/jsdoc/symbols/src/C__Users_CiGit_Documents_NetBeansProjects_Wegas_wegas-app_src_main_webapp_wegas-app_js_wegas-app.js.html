 12 YUI.add('wegas-app', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      * Create a new wegas-app
 17      *
 18      * <p><strong>Attributes</strong></p>
 19      * <ul>
 20      *    <li>base {String} base Url for app</li>
 21      *    <li>layoutSrc {String} location for the json config of the current page</li>
 22      *    <li>dataSources {Object[]} the list of datasource to be loaded on startup</li>
 23      *    <li>cssStylesheets {String[]} a list of stylesheets to be loaded<i>default: []</i></li>
 24      *    <li>currentGameModel {Number} current game model id</li>
 25      *    <li>currentGame {Number} current game  id</li>
 26      *    <li>currentTeam {Number} current team id</li>
 27      *    <li>currentPlayer {Number} current player id</li>
 28      *    <li>currentUse {Number} current game model idObject litteral representing current user</li>
 29      *    <li>editorMenus {Object[]}
 30      *        This field is used to globally override Entities edition menus.
 31      *        Use the target class name as the key.
 32      *    </li>
 33      *    <li>editorForms {Object[]}
 34      *         This field is used to globally override Entities edition forms.
 35      *        Use the target class name as the key.
 36      *   </li>
 37      * </ul>
 38      *
 39      * @name Y.Wegas.App
 40      * @class Base class for wegas, handle initialisation of datasources and rendering
 41      * @extends Y.Base
 42      * @constructor
 43      * @param Object Will be used to fill attributes field
 44      */
 45     var App = Y.Base.create("wegas-app", Y.Base, [], {
 46         /** @lends Y.Wegas.App# */
 47 
 48         // ** Private fields ** //
 49         /**
 50          * Holds a reference to all the dataSources used.
 51          * @field
 52          * @private
 53          */
 54         dataSources: [],
 55 
 56         /**
 57          * Lifecycle methods
 58          * @function
 59          * @private
 60          */
 61         initializer: function() {
 62             Y.Wegas.app = this;
 63             /**
 64              * @name Cocktail#shake
 65              * @event
 66              * @param {MyEventObject} e
 67              * @param {Boolean} [e.withIce=false]
 68              */
 69             this.publish("render", {});
 70         },
 71 
 72         /**
 73          * Destructor methods.
 74          * @function
 75          * @private
 76          */
 77         destructor: function() {
 78             var i;
 79             for (i = 0; i < this.dataSources.length; i = i + 1) {
 80                 this.dataSources[i].destroy();
 81             }
 82         },
 83 
 84         /**
 85          * Render function
 86          * @function
 87          * @private
 88          */
 89         render: function() {
 90             Y.io.header("Accept-Language", Y.config.lang);                      // Set the language for all requests
 91             Y.JSON.useNativeParse = true;                                       // @todo Shall we use browser native parser ?
 92 
 93             this.on("render", function() {                                     // Remove loading overlay on render
 94                 Y.one("body").removeClass("wegas-loading-overlay");
 95             });
 96 
 97             Y.on("io:failure", function(tId, e) {
 98                 if (!e) {
 99                     return;
100                 }
101                 var exception = e.responseText.substring(e.responseText.indexOf('"exception'), e.responseText.length);
102                 exception = exception.split(",");
103                 if (e.status === 400 && exception[0] === '"exception":"org.apache.shiro.authz.UnauthorizedException"' ||
104                     exception[0] === '"exception":"org.apache.shiro.authz.UnauthenticatedException"') {
105                     // Y.config.win.location.href = Y.Wegas.app.get("base") + 'wegas-app/view/login.html';   //Redirect to login
106                     alert("You have been logged out or does not have permissions");
107                 }
108             }, this);
109 
110             this.initDataSources();
111 
112             /**
113              * Shortcut to activate developper mode. Allow access to Y instance. Toggle.
114              * Event keypress 'ï¿½'
115              */
116             Y.one("body").on("keypress", function(e) {
117                 if (e.charCode === 176 && e.target === this) {
118                     Y.Wegas.app.set("devMode", !Y.Wegas.app.get("devMode"));
119                 }
120             });
121         },
122 
123         // *** Private methods ** //
124         /**
125          * @function
126          * @private
127          * @description initilize DataSources
128          */
129         initDataSources: function() {
130             var k, dataSource, dataSources = this.get('dataSources'),
131             onInitialRequest = function() {
132                 this.requestCounter -= 1;
133                 if (this.requestCounter === 0) {
134                     this.initPage();                                       // Run the initPage() method when they all arrived.
135                 }
136             };
137 
138             this.requestCounter = 0;
139 
140             for (k in dataSources) {
141                 if (dataSources.hasOwnProperty(k)) {
142                     dataSources[k].source = this.get("base") + dataSources[k].source; // Set up datasource path
143                     dataSource = new Y.Wegas.DataSource(dataSources[k]);        // Instantiate the datasource
144                     this.dataSources[k] = Y.Wegas[k + "Facade"] = this[k + "Facade"] = dataSource; // Set up global references
145                     dataSource.once("response", onInitialRequest, this);        // Listen to the datasources initial requests
146                     if (Y.Lang.isNumber(dataSource.sendInitialRequest())) {     // Send the initial request
147                         this.requestCounter += 1;                               // If the request was sent, update the counter
148                     }
149                 }
150             }
151 
152             if (this.requestCounter === 0) {                                    // If no request was sent, render directly
153                 this.initPage();
154             }
155         },
156 
157         /**
158          * @function
159          * @private
160          * @description initPage methods
161          */
162         initPage: function() {
163             Y.io(this.get('base') + this.get('layoutSrc') + '?id=' + Y.Wegas.Helper.genId(), {
164                 context: this,
165                 on: {
166                     success: function(id, o) {
167                         var cfg;
168                         try {
169                             cfg = Y.JSON.parse(o.responseText);		// Process the JSON data returned from the server
170                         } catch (e) {
171                             alert("Wegas.App.initUI(): Layout json parse failed failed!");
172                             Y.error("Layout parse failed", e, "Y.Wegas.App");
173                             return;
174                         }
175 
176                         Y.Wegas.Widget.use(cfg, Y.bind(function(cfg) {         // Load the subwidget dependencies
177                             this.widget = Y.Wegas.Widget.create(cfg);           // Render the subwidget
178                             this.widget.render();
179                             this.fire("render");                         // Fire a render event for some eventual post processing
180                         }, this, cfg));
181 
182                     //this.pageLoader = new Y.Wegas.PageLoader();               // Load the subwidget using pageloader
183                     //this.pageLoader.render();
184                     // cfg.id = -100;
185                     // this.dataSources.Page.data.push(cfg);
186                     //try {
187                     //    this.pageLoader.set("pageId", -100);
188                     //} catch (renderException) {
189                     //    Y.log('initUI(): Error rendering UI: ' + ((renderException.stack)
190                     //     ? renderException.stack : renderException), 'error', 'Wegas.App');
191                     //}
192 
193                     }
194                 }
195             });
196         }
197     }, {
198         /** @lends Y.Wegas.App */
199 
200         /**
201          * @field
202          * @static
203          */
204         ATTRS: {
205             /**
206              * Base url for app
207              */
208             base: {
209                 getter: function() {
210                     return Y.config.groups.wegas.base;
211                 }
212             },
213             layoutSrc: {},
214             dataSources: {
215                 value: {}
216             },
217             cssStylesheets: {
218                 value: []
219             },
220             currentGameModel: {},
221             currentGame: {},
222             currentTeam: {},
223             currentPlayer: {},
224             /**
225              * Object litteral representing current user.
226              */
227             currentUser: {},
228             /**
229              *
230              */
231             devMode: {
232                 value: false,
233                 setter: function(val) {
234                     if (val) {
235                         Y.one("body").addClass("wegas-devmode");
236                         //if (YUI_config.debug) {
237                         window.Y = Y;
238                     //}
239                     } else {
240                         Y.one("body").removeClass("wegas-devmode");
241                         // if (YUI_config.debug) {
242                         delete window.Y;
243                     //}
244                     }
245                 }
246             },
247             /**
248              * This field is used to globally override Entities edition menus.
249              * Use the target class name as the key.
250              */
251             editorMenus: {
252                 value: {}
253             },
254             /**
255              * This field is used to globally override Entities edition forms.
256              * Use the target class name as the key.
257              */
258             editorForms: {
259                 value: {}
260             }
261         }
262 
263     });
264     Y.namespace('Wegas').App = App;
265 });
