39   * This servlet allows to retrieve several resources in a single request. Used

54      private static final Logger logger = LoggerFactory.getLogger(ComboController.class);
55      private final static String CACHE_NAME = "combo";
56      /**
57       *
58       */
59      final static public String MediaTypeCss = "text/css; charset=ISO-8859-1";
60      final static public String MediaTypeJs = "text/javascript; charset=ISO-8859-1";
61      /**
62       *
63       */
64      @Context
65      protected UriInfo uriInfo;
66      /**
67       *

69      @Context
70      private ServletContext servletContext;

72      /**
73       * Retrieve
74       *
75       * @param req
76       * @return
77       * @throws IOException
78       */
79      @GET
80      @Produces({MediaTypeJs, MediaTypeCss})
81      @CacheMaxAge(time = 3, unit = TimeUnit.HOURS)
82      public Response index(@Context Request req) throws IOException {
83          Ehcache cache = cacheManagerHolder.getInstance().getEhcache(CACHE_NAME);
84  
85          final int hash = this.uriInfo.getRequestUri().getQuery().hashCode();
86          // Select the content-type based on the last file extension
87  
88          final Element combo = cache.get(hash);
89          CacheObject comboCache;
90  
91          if (combo != null) { // Get from cache
92              comboCache = (CacheObject) combo.getObjectValue();
93          } else { // build Cache.
94              //final Set<String> files = this.uriInfo.getQueryParameters().keySet(); // Old version, removed cause query parameters where in the wrong order
95              ArrayList<String> files = new ArrayList<>();                            // New version, with parameters in the right order
96              for (String parameter : this.uriInfo.getRequestUri().getQuery().split("&")) {
97                  String split = parameter.split("=")[0];
98                  if (split != null) {
99                      files.add(split);
100                 }
101             }
102             files.remove("v");
103             files.remove("version");
104             final String mediaType = (files.iterator().next().endsWith("css"))
105                     ? MediaTypeCss : MediaTypeJs;                            // Select the content-type based on the first file extension
106             comboCache = new CacheObject(this.getCombinedFile(files, mediaType), mediaType);
107             cache.put(new Element(hash, comboCache));
108 
109         }
110         ResponseBuilder rb = req.evaluatePreconditions(new EntityTag(comboCache.getETag()));
111         if (rb != null) {
112             return rb.tag(comboCache.getETag()).build();
113         }
114         // MediaType types[] = {"application/json", "application/xml"};
115         // List<Variant> vars = Variant.mediaTypes(types).add().build();
116         // Variant var = req.selectVariant(vars);
117         //EntityTag etag = new EntityTag();
118         //Response.ResponseBuilder responseBuilder = request.evaluatePreconditions(updateTimestamp, etag);
119         return Response.ok(comboCache.getFiles())
120                 .type(comboCache.getMediaType())
121                 //    .expires(new Date(System.currentTimeMillis() + (1000 * 60 * 60 * 24 * 3)))
122                 .tag(new EntityTag(comboCache.getETag()))
123                 .build();
124     }

126     private String getCombinedFile(List<String> fileList, String mediaType) throws IOException {
127         StringBuilder acc = new StringBuilder();
128         for (String fileName : fileList) {
129             try {
130                 InputStream fis = (InputStream) servletContext.getResourceAsStream(fileName);
131                 String content = IOUtils.toString(fis, Helper.getWegasProperty("encoding"));
132                 //String content = new Scanner(fis, Helper.getWegasProperty("encoding"))
133                 //.useDelimiter("\\A").next();                              // Use a fake delimiter to read all lines at once
134 
135                 if (mediaType.equals(MediaTypeCss)) {                       // @hack for css files, we correct the path
136                     String dir = fileName.substring(0, fileName.lastIndexOf('/') + 1);
137                     content = content.replaceAll("url\\(\"?\'?([^:\\)\"\']+)\"?\'?\\)",
138                             "url(" + servletContext.getContextPath()
139                             + dir + "$1)");              //Regexp to avoid rewriting protocol guess they contain ':' (http: data:)
140                 }
141                 acc.append(content);
142             } catch (NullPointerException e) {
143                 logger.error("Resource not found : {}", fileName);
144             }
145         }
146         return acc.toString();
147     }

149     private static class CacheObject implements Serializable {
150 
151         private String ETag;
152         private String files;
153         private String mediaType;
154 
155         public CacheObject(String files, String mediaType) {
156             try {
157                 this.files = files;
158                 this.mediaType = mediaType;
159                 MessageDigest md = MessageDigest.getInstance("MD5");
160                 byte[] digest = md.digest(files.getBytes());
161                 this.ETag = String.format("%1$032X", new BigInteger(1, digest));
162             } catch (NoSuchAlgorithmException ex) {
163                 logger.error("No MD5 algorithm found");
164             }
165         }
166 
167         public String getETag() {
168             return ETag;
169         }
170 
171         public String getFiles() {
172             return files;
173         }
174 
175         public String getMediaType() {
176             return mediaType;
177         }
178 
179         @Override
180         public int hashCode() {
181             return this.ETag.hashCode();
182         }
183 
184         @Override
185         public boolean equals(Object obj) {
186             if (obj == null) {
187                 return false;
188             }
189             if (getClass() != obj.getClass()) {
190                 return false;
191             }
192             final CacheObject other = (CacheObject) obj;
193             return Objects.equals(this.ETag, other.ETag);
194         }

196     }
