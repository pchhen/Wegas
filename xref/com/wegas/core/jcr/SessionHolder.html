8   package com.wegas.core.jcr;
9   
10  import com.wegas.core.Helper;
11  import java.util.HashMap;
12  import java.util.Map;
13  import javax.annotation.PreDestroy;
14  import javax.ejb.DependsOn;
15  import javax.ejb.Singleton;
16  import javax.jcr.Repository;
17  import javax.jcr.RepositoryException;
18  import javax.jcr.Session;
19  import javax.jcr.SimpleCredentials;
20  
21  /**
22   *
23   * @author Cyril Junod <cyril.junod at gmail.com>
24   */
25  @Singleton
26  @DependsOn("JackrabbitConnector")
27  public class SessionHolder {
28  
29      final static private SimpleCredentials admin = new SimpleCredentials(Helper.getWegasProperty("jcr.admin.username"), Helper.getWegasProperty("jcr.admin.password").toCharArray());
30      private static Map<String, Session> sessionMap = new HashMap<>();
31  
32      /**
33       *
34       * @param repository
35       * @return
36       * @throws RepositoryException
37       */
38      public static Session getSession(String repository) throws RepositoryException {
39          Session session;
40          if (!SessionHolder.sessionMap.containsKey(repository) || !SessionHolder.sessionMap.get(repository).isLive()) {
41              Repository repo = (Repository) new JackrabbitConnector().getRepo();
42              sessionMap.remove(repository);
43              try {
44                  session = repo.login(admin, repository);
45              } catch (javax.jcr.NoSuchWorkspaceException ex) {
46                  createWorkspace(repository);
47                  session = repo.login(admin, repository);
48              }
49              sessionMap.put(repository, session);
50          }
51          return sessionMap.get(repository);
52      }
53  
54      /**
55       *
56       * @param repository
57       * @throws RepositoryException
58       */
59      public static void closeSession(String repository) throws RepositoryException {
60          if (SessionHolder.sessionMap.containsKey(repository) && SessionHolder.sessionMap.get(repository).isLive()) {
61              SessionHolder.sessionMap.get(repository).save();
62              SessionHolder.sessionMap.get(repository).logout();
63          }
64          SessionHolder.sessionMap.remove(repository);
65      }
66  
67      /**
68       *
69       * @param repository
70       * @throws RepositoryException
71       */
72      protected static void createWorkspace(String repository) throws RepositoryException {
73          Session s;
74  
75          s = getSession(null);
76          s.getWorkspace().createWorkspace(repository);
77      }
78  
79      @PreDestroy
80      private void onDestroy() {
81          for (Session s : sessionMap.values()) {
82              if (s.isLive()) {
83                  s.logout();
84              }
85          }
86          sessionMap.clear();
87      }
88  }
